8/// Defines how an object-tree is to be traversed.
9/// Used in `CollectionProxy`.

11#[repr(u32)]
12pub enum TreeTraversalType {
13	/// When traversing the tree, restrict to children of the current object.
14	RestrictChildren,
15
16	/// When traversing the tree, restrict to siblings of the current object.

18

24/// Definition of match rules for accessible objects.
25/// Rule(s) against which we can match an  object or a collection thereof.
26///
27/// # Examples
28/// ```rust
29/// # use zbus::MatchRule;
30/// let builder = MatchRule::builder();
31/// ```
32///
33#[derive(Debug, Clone, Default, Serialize, Deserialize, PartialEq, Eq)]
34pub struct ObjectMatchRule {
35	pub states: StateSet,
36	pub states_mt: MatchType,
37	pub attr: HashMap<String, String>,
38	pub attr_mt: MatchType,
39	pub roles: Vec<Role>,
40	pub roles_mt: MatchType,
41	pub ifaces: InterfaceSet,
42	pub ifaces_mt: MatchType,
43	pub invert: bool,
44	// Private phantom, gets compiled away.
45	// Here to ensure the builder is the only route to obtain a `MatchRule`
46	#[serde(skip)]
47	_marker: std::marker::PhantomData<()>,
48}
49
50// !!! WARNING !!! :
51//
52// State and Role are defined as u32 in Accessible.xml but as i32 in Collection.xml
53//
54// The signature on StateSet is defined inconsistently in the XMLs
55// Accessible.xml: GetState type="au"
56// Collection.xml: GetMatches argument name="rule" type="(aiia{ss}iaiiasib)"
57// The latter starts with ai, which is a signature for an array of signed ints i32.
58//
59// https://gitlab.gnome.org/federico/at-spi2-core/-/commit/4885efedeef71e0df8210622771a0b1bb10e194d
60impl Type for ObjectMatchRule {
61	const SIGNATURE: &'static Signature = &Signature::static_structure(&[
62		&Signature::Array(Child::Static { child: &Signature::I32 }),
63		&Signature::I32,
64		&Signature::Dict {
65			key: Child::Static { child: &Signature::Str },
66			value: Child::Static { child: &Signature::Str },
67		},
68		&Signature::I32,
69		&Signature::Array(Child::Static { child: &Signature::I32 }),
70		&Signature::I32,
71		&Signature::Array(Child::Static { child: &Signature::Str }),
72		&Signature::I32,
73		&Signature::Bool,
74	]);
75}

85/// The 'builder' type for `MatchRule`.  
86/// Use its methods to set match criteria.
87#[derive(Debug, Clone, Default)]
88pub struct ObjectMatchRuleBuilder {
89	states: StateSet,
90	states_mt: MatchType,
91	attr: HashMap<String, String>,
92	attr_mt: MatchType,
93	roles: Vec<Role>,
94	roles_mt: MatchType,
95	ifaces: InterfaceSet,
96	ifaces_mt: MatchType,
97	invert: bool,
98}
99
100impl ObjectMatchRuleBuilder {
101	/// Insert a `StateSet` to the builder
102	#[must_use]
103	pub fn states<I>(mut self, states: I, mt: MatchType) -> Self
104	where
105		I: IntoIterator,
106		I::Item: Borrow<State>,
107	{
108		self.states = states.into_iter().map(|state| *state.borrow()).collect();
109		self.states_mt = mt;
110		self
111	}
112
113	/// Insert a map of attributes
114	#[must_use]
115	pub fn attributes(mut self, attributes: HashMap<String, String>, mt: MatchType) -> Self {
116		self.attr = attributes;
117		self.attr_mt = mt;
118		self
119	}
120
121	/// Insert a slice of `Role`s
122	#[must_use]
123	pub fn roles(mut self, roles: &[Role], mt: MatchType) -> Self {
124		self.roles = roles.into();
125		self.roles_mt = mt;
126		self
127	}
128
129	/// Insert an `InterfaceSet` from a collection of `Interface`s
130	#[must_use]
131	pub fn interfaces<I>(mut self, interfaces: I, mt: MatchType) -> Self
132	where
133		I: IntoIterator,
134		I::Item: Borrow<Interface>,
135	{
136		self.ifaces = interfaces.into_iter().map(|iface| *iface.borrow()).collect();
137		self.ifaces_mt = mt;
138		self
139	}
140
141	/// Sets the inversion of the `MatchRule`, defaults to `false`, no inversion.
142	#[must_use]
143	pub fn invert(mut self, invert: bool) -> Self {
144		self.invert = invert;
145		self
146	}
147
148	/// Builds the [`ObjectMatchRule`]
149	///
150	/// [`ObjectMatchRule``]: crate::object_match::ObjectMatchRule
151	#[must_use]
152	pub fn build(self) -> ObjectMatchRule {
153		ObjectMatchRule {
154			states: self.states,
155			states_mt: self.states_mt,
156			attr: self.attr,
157			attr_mt: self.attr_mt,
158			roles: self.roles,
159			roles_mt: self.roles_mt,
160			ifaces: self.ifaces,
161			ifaces_mt: self.ifaces_mt,
162			invert: self.invert,
163			_marker: PhantomData,
164		}
165	}
166}

168/// Enumeration used by [`ObjectMatchRule`] to specify how to interpret [`ObjectRef`] objects.
169///
170/// [`ObjectRef`]: crate::object_ref::ObjectRef
171#[derive(Clone, Copy, Debug, PartialEq, Eq, Serialize, Deserialize, Type, Default)]
172#[repr(i32)]
173pub enum MatchType {
174	/// Invalidates match criterion.
175	Invalid,
176
177	#[default]
178	/// All of the criteria must be met.
179	All,
180
181	/// Any of the criteria must criteria must be met.
182	Any,
183
184	/// None of the criteria must be met.
185	NA,
186
187	/// Same as [`Self::All`] if the criterion item is non-empty - All of the criteria must be met.
188	/// For empty criteria this rule requires the returned value to also have empty set.
189	Empty,
190}
