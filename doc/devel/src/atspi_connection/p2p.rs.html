1//! Extends the `AccessibilityConnection` with P2P capabilities.
2//!
3//! # Considerations on using executors and P2P
4//!
5//! Every connection has a zbus `Executor` instance, on which tasks can be launched. Internally, zbus uses this executor for all sorts of tasks: listening for D-Bus signals,
6//! keeping property caches up to date, handling timeout errors, etc.
7//!
8//! When zbus users use `tokio` (zbus feature "tokio" set), zbus will latch onto the tokio runtime.
9//! The `Executor` instance will be empty and all zbus tasks are run on the user's tokio runtime.
10//! However, when using any other executor (smol, glommio, etc.), each `Connection` will spin up a thread with an `async_executor::Executor`.
11//!
12//! Typically an application will have a single connection, but with P2P, your application will have a connection with each application that supports it.
13//! Consequently, on anything but tokio, applications will get an extra thread with an `async_executor` for each connection!
14//! (So picking smol won't necessarily make your application small in the context of P2P.)
15

41/// Represents a peer with the name, path and connection for the P2P peer.
42#[derive(Clone, Debug)]
43pub struct Peer {
44	unique_name: OwnedUniqueName,
45	well_known_name: Option<OwnedWellKnownName>,
46	socket_address: Address,
47	p2p_connection: zbus::Connection,
48}
49
50impl Peer {
51	/// Creates a new `Peer` with the given bus name and socket path.
52	///
53	/// # Note
54	/// This function is intended for use in building the initial list of peers.
55	///
56	/// If given a `UniqueName`, it will check if the peer also owns a well-known name.
57	/// If given a `WellKnownName`, it will query the D-Bus for the unique name of the peer.
58	///
59	/// # Errors
60	/// - `DBusProxy` cannot be created.
61	/// - The socket address cannot be parsed.
62	///
63	pub(crate) async fn try_new<B, S>(
64		bus_name: B,
65		socket: S,
66		conn: &zbus::Connection,
67	) -> Result<Self, AtspiError>
68	where
69		B: Into<OwnedBusName>,
70		S: TryInto<Address>,
71	{
72		let dbus_proxy = DBusProxy::new(conn).await?;
73		let owned_bus_name: OwnedBusName = bus_name.into();
74
75		let socket_address = socket
76			.try_into()
77			.map_err(|_| AtspiError::ParseError("Invalid address string"))?;
78
79		// Because D-Bus does not let us query whether a unique name is the owner of a well-known name,
80		// we need to query all well-known names and their owners, and then check if the unique name is one of them.
81
82		// Get all well-known names from D-Bus
83		let well_known_names: Vec<OwnedWellKnownName> = dbus_proxy
84			.list_names()
85			.await?
86			.into_iter()
87			.filter_map(|name| {
88				if let BusName::WellKnown(well_nown_name) = name.clone().inner() {
89					Some(OwnedWellKnownName::from(well_nown_name.clone()))
90				} else {
91					None
92				}
93			})
94			.collect();
95
96		// We are creating a mapping of unique names to well-known names.
97		let mut unique_to_well_known: Vec<(OwnedUniqueName, OwnedWellKnownName)> = Vec::new();
98
99		// For each well-known name, we get the unique name of the owner.
100		// Note: not all well-known names on the bus will have an accessible owner.
101		// For instance, the `org.freedesktop.DBus` well-known name does not have an accessible connection.
102		for well_known_name in &well_known_names {
103			let bus_name = BusName::from(well_known_name.clone());
104			if let Ok(unique_name) = dbus_proxy.get_name_owner(bus_name).await {
105				unique_to_well_known.push((unique_name, well_known_name.clone()));
106			}
107		}
108
109		// Now we have a mapping of unique names to well-known names.
110		//
111		// A `Peer` instance requires a unique name which may or may not have a well-known name.
112		// We can build a `Peer` instance from either a unique name or a well-known name.
113		// If the argument name is a unique name, we look up whether this peer also owns a well-known name in our mapping.
114		// If the argument name is a well-known name, we _must_ get its unique owner too to create a `Peer`.
115		let (unique_name, well_known_name) = match owned_bus_name.inner() {
116			BusName::Unique(name) => {
117				// The argument name is the mandatory `UniqueName`, we do want check whether this peer also owns a well-known name.
118				let owned_well_known_name = unique_to_well_known.iter().find_map(|(u, w)| {
119					if u == name {
120						Some(w.clone())
121					} else {
122						None
123					}
124				});
125				let owned_unique_name = OwnedUniqueName::from(name.clone());
126				(owned_unique_name, owned_well_known_name)
127			}
128			BusName::WellKnown(well_known_name) => {
129				let bus_name = BusName::from(well_known_name.clone());
130				let owned_unique_name = dbus_proxy.get_name_owner(bus_name).await?;
131				let owned_well_known_name = OwnedWellKnownName::from(well_known_name.clone());
132				(owned_unique_name, Some(owned_well_known_name))
133			}
134		};
135
136		let p2p_connection = Builder::address(socket_address.clone())?.p2p().build().await?;
137
138		Ok(Peer { unique_name, well_known_name, socket_address, p2p_connection })
139	}

164	/// Try to create a new `Peer` from a bus name.
165	///
166	/// # Errors
167	/// Returns an error if the application proxy cannot be created or if it does not support `get_application_bus_address`.\
168	/// A non-existent bus name will also return an error.
169	pub async fn try_from_bus_name(
170		bus_name: BusName<'_>,
171		conn: &zbus::Connection,
172	) -> AtspiResult<Self> {
173		// Get the application proxy for the bus name
174		let application_proxy = ApplicationProxy::builder(conn)
175			.destination(&bus_name)?
176			.cache_properties(CacheProperties::No)
177			.build()
178			.await?;
179
180		let socket_path = application_proxy.get_application_bus_address().await?;
181		Self::try_new(bus_name, socket_path.as_str(), conn).await
182	}

233// A trait is needed to extend functionality on `BusName` for P2P address lookup.
234pub(crate) trait BusNameExt {
235	/// Looks up a `BusName`'s P2P address, if available.
236	async fn get_p2p_address(&self, conn: &zbus::Connection) -> AtspiResult<Address>;
237}
238
239impl BusNameExt for BusName<'_> {
240	async fn get_p2p_address(&self, conn: &zbus::Connection) -> AtspiResult<Address> {
241		let application_proxy = application::ApplicationProxy::builder(conn)
242			.destination(self)?
243			.cache_properties(CacheProperties::No)
244			.build()
245			.await?;
246
247		application_proxy
248			.get_application_bus_address()
249			.await
250			.map_err(|e| {
251				AtspiError::Owned(format!("Failed to get application bus address for {self}: {e}"))
252			})
253			.and_then(|address| {
254				Address::try_from(address.as_str())
255					.map_err(|_| AtspiError::ParseError("Invalid address string"))
256			})
257	}
258}

265impl Peers {
266	/// Returns a `Peers` containing the initial peers that support P2P connections.
267	///
268	/// # Note
269	/// Intended for internal use with `AccessibilityConnection::new()`.
270	///
271	/// # Errors
272	/// This function can return an error in the following cases:
273	/// - the `AccessibleProxy` to the registry cannot be created.
274	/// - the registry returns an error when querying for children.
275	/// - for any child, the `AccessibleProxy` cannot be created or the `ApplicationProxy` cannot be created.
276	pub(crate) async fn initialize_peers(conn: &zbus::Connection) -> AtspiResult<Self> {
277		let registry_well_known_name = RegistryProxy::DESTINATION
278			.as_ref()
279			.expect("RegistryProxy `default_destination` is not set");
280		let reg_accessible = AccessibleProxy::builder(conn)
281			.destination(registry_well_known_name)?
282			.cache_properties(CacheProperties::No)
283			.build()
284			.await?;
285
286		let accessible_applications = reg_accessible.get_children().await?;
287		let mut peers = Vec::with_capacity(accessible_applications.len());
288
289		for app in accessible_applications {
290			let accessible_proxy = app.as_accessible_proxy(conn).await?;
291			let proxies = accessible_proxy.proxies().await?;
292			let application_proxy = proxies.application().await?;
293
294			// Get the application bus address
295			// aka: Does the application support P2P connections?
296			if let Ok(address) = application_proxy.get_application_bus_address().await {
297				let name = app.name().ok_or(AtspiError::MissingName)?;
298				let bus_name = BusName::from(name.clone());
299
300				match Peer::try_new(bus_name, address.as_str(), conn).await {
301					Ok(peer) => peers.push(peer),
302
303					#[cfg(feature = "tracing")]
304					Err(e) => {
305						tracing::warn!("Failed to create peer for {:?}: {}", app.name_as_str(), e);
306					}
307
308					#[cfg(all(debug_assertions, not(feature = "tracing")))]
309					Err(e) => {
310						eprintln!("Failed to create peer for {:?}: {}", app.name_as_str(), e);
311					}
312
313					#[cfg(not(any(feature = "tracing", debug_assertions)))]
314					Err(_) => {
315						// Ignore error creating peer
316					}
317				}
318			}
319		}
320
321		Ok(Peers { peers: Arc::new(Mutex::new(peers)) })
322	}

324	/// Returns a [`Peer`] by its bus name.
325	fn get_peer(&self, bus_name: &BusName<'_>) -> Option<Peer> {
326		let peers = self.peers.lock().expect("already locked by current thread");
327
328		let matched = match bus_name {
329			BusName::Unique(unique_name) => {
330				peers.iter().find(|peer| peer.unique_name() == unique_name)
331			}
332			BusName::WellKnown(well_known_name) => {
333				let owned_well_known_name = OwnedWellKnownName::from(well_known_name.clone());
334				peers
335					.iter()
336					.find(|peer| peer.well_known_name() == Some(&owned_well_known_name))
337			}
338		};
339		matched.cloned()
340	}

347	/// Inserts a new `Peer` into the list of peers.
348	async fn insert_unique(
349		&self,
350		unique_name: &zbus::names::UniqueName<'_>,
351		conn: &zbus::Connection,
352	) -> AtspiResult<()> {
353		let bus_name = BusName::Unique(unique_name.as_ref());
354		let address = bus_name.get_p2p_address(conn).await?;
355		let p2p_connection = Builder::address(address.clone())?.p2p().build().await?;
356
357		let unique_name = OwnedUniqueName::from(unique_name.clone());
358
359		let peer =
360			Peer { unique_name, well_known_name: None, socket_address: address, p2p_connection };
361
362		let mut guard = self.peers.lock().expect("lock already held by current thread");
363		guard.push(peer);
364		Ok(())
365	}

367	/// Removes a `Peer` from the list of peers by its unique name.
368	fn remove_unique(&self, unique_name: &zbus::names::UniqueName<'_>) {
369		let mut peers = self.peers.lock().expect("lock already held by current thread");
370		peers.retain(|peer| peer.unique_name() != unique_name);
371	}

373	/// Inserts a new `Peer` with a well-known name into the list of peers.
374	async fn insert_well_known(
375		&self,
376		well_known_name: &WellKnownName<'_>,
377		name_owner: &UniqueName<'_>,
378		conn: &zbus::Connection,
379	) -> AtspiResult<()> {
380		let bus_name = BusName::WellKnown(well_known_name.clone());
381		let address = bus_name.get_p2p_address(conn).await?;
382		let p2p_connection = Builder::address(address.clone())?.p2p().build().await?;
383
384		let well_known_name = OwnedWellKnownName::from(well_known_name.clone());
385		let unique_name = OwnedUniqueName::from(name_owner.clone());
386
387		let peer = Peer {
388			unique_name,
389			well_known_name: Some(well_known_name),
390			socket_address: address,
391			p2p_connection,
392		};
393
394		let mut guard = self.peers.lock().expect("lock already held by current thread");
395		guard.push(peer);
396		Ok(())
397	}

399	/// Removes a `Peer` with a well-known name from the list of peers.
400	fn remove_well_known(&self, well_known_name: &WellKnownName<'_>, name_owner: &UniqueName<'_>) {
401		let mut peers = self.peers.lock().expect("lock already held by current thread");
402		let owned_well_known_name = OwnedWellKnownName::from(well_known_name.clone());
403		peers.retain(|peer| {
404			(peer.well_known_name() != Some(&owned_well_known_name))
405				&& peer.unique_name() == name_owner
406		});
407	}

409	/// Update a `Peer` with a new owner of it's well-known name in the list of peers.
410	async fn update_well_known_owner(
411		&self,
412		well_known_name: &WellKnownName<'_>,
413		old_name_owner: &UniqueName<'_>,
414		new_name_owner: &UniqueName<'_>,
415		conn: &zbus::Connection,
416	) -> AtspiResult<()> {
417		let socket_address = BusName::from(new_name_owner.clone()).get_p2p_address(conn).await?;
418		let p2p_connection = Builder::address(socket_address.clone())?.p2p().build().await?;
419
420		let well_known_name = Some(OwnedWellKnownName::from(well_known_name.clone()));
421		let old_name_owner = OwnedUniqueName::from(old_name_owner.clone());
422		let unique_name = OwnedUniqueName::from(new_name_owner.clone());
423
424		let peer = Peer {
425			unique_name,
426			well_known_name: well_known_name.clone(),
427			socket_address,
428			p2p_connection,
429		};
430
431		let mut peers = self.peers.lock().expect("lock already held by current thread");
432		if let Some(existing_peer) = peers.iter_mut().find(|p| {
433			p.well_known_name() == well_known_name.as_ref() && p.unique_name() == &old_name_owner
434		}) {
435			*existing_peer = peer;
436		} else {
437			return Err(AtspiError::Owned(format!(
438                "Owner swap failed: well-known name {well_known_name:?} with owner: {old_name_owner} not found"
439            )));
440		}
441		Ok(())
442	}

444	/// Spawns a task which listens for peer mutations.
445	///
446	/// This task listens for `NameOwnerChanged` signals and updates the list of peers accordingly.
447	///
448	/// # executor
449	/// The task is spawned on the executor of the `zbus::Connection`.
450	///
451	/// # Note
452	/// This function is called internally by `AccessibilityConnection::new()`.
453	pub(crate) fn spawn_peer_listener_task(&self, conn: &zbus::Connection) {
454		// Clone the `Peers` and `Connection` to move them into the async task.
455		// This is necessary because the async task needs to own these values.
456		let peers = self.clone();
457		let conn = conn.clone();
458		let dbus_proxy = futures_lite::future::block_on(DBusProxy::new(&conn))
459			.expect("Failed to create DBusProxy");
460
461		let executor = conn.executor().clone();
462
463		executor.spawn(async move {
464			let Ok(mut name_owner_changed_stream) = dbus_proxy.receive_name_owner_changed().await.inspect_err(|#[allow(unused_variables)] err| {
465				#[cfg(feature = "tracing")]
466				debug!("Failed to receive `NameOwnerChanged` stream: {err}");
467			}) else {
468				return;
469			};
470
471			while let Some(name_owner_event) = name_owner_changed_stream.next().await {
472					let Ok(args) = name_owner_event.args() else {
473						#[cfg(feature = "tracing")]
474						tracing::debug!("Received name owner changed event without args, skipping.");
475						continue;
476					};
477					let name = args.name().clone();
478					let new = args.new_owner().clone();
479					let old = args.old_owner().clone();
480
481					// `NameOwnerChanged` table (U = Unique, W = Well-Known):
482					// | Name | Old Owner | New Owner | Operation |
483					// |------|-----------|-----------|----------|
484					// |   U  |   None    |  Some(U)  |  Add     |
485					// |   U  |   Some(U) |    None   |  Remove  |
486					// |   W  |   None    |  Some(U)  |  Add     |
487					// |   W  |   Some(U) |    None   |  Remove  |
488					// |   W  |   Some(U) |  Some(U)  |  Replace |
489
490					match name {
491						BusName::Unique(unique_name) => {
492							// `zvariant:Optional` has deref target `Option`.
493							match (&*old, &*new) {
494								// Application appeared on the bus.
495								(None, Some(new_owner)) => {
496									debug_assert_eq!(new_owner, &unique_name, "When a name appears on the bus, the new owner must be the unique name itself.");
497
498									if let Ok(()) = peers.insert_unique(&unique_name, &conn).await.inspect_err(|#[allow(unused_variables)] err| {
499										#[cfg(feature = "tracing")]
500										warn!("Failed to insert unique name: {unique_name}: {err}");
501									}) {
502										#[cfg(feature = "tracing")]
503										info!("Inserted unique name: {unique_name} into the peer list.");
504									};
505
506								}
507								// Unique name left the bus.
508								(Some(old), None) => {
509									debug_assert!(old == &unique_name, "When a unique name is removed from the bus, the old owner must be the unique name itself.");
510									peers.remove_unique(&unique_name);
511
512									#[cfg(feature = "tracing")]
513									info!("Peer with unique name: {unique_name} left the bus - removed from peer list.");
514								}
515
516								// Unknown combination.
517								(_, _) => {
518									#[cfg(feature = "tracing")]
519									debug!("NameOwnerChanged` with unique name: {unique_name} has unknown argument combination ({old:?}, {new:?}).");
520								}
521							}
522						}
523						BusName::WellKnown(well_known_name) => {
524							match (&*old, &*new) {
525								// Unknown mutatuion. Well-known names should always have at least a new or old owner.
526								(None, None) => {
527									#[cfg(feature = "tracing")]
528									debug!("Received `NameOwnerChanged` event with no old or new owner for well-known name: {}", well_known_name);
529								}
530
531								// Well-known name appeared on the bus.
532								(None, Some(new_owner_unique_name)) => {
533									if let Ok(()) = peers.insert_well_known(
534										&well_known_name,
535										new_owner_unique_name,
536										&conn,
537									).await.inspect_err(|#[allow(unused_variables)] err| {
538										#[cfg(feature = "tracing")]
539										warn!("Failed to insert well-known name: {} with owner: {} - {}", &well_known_name, &new_owner_unique_name, err);
540									}) {
541										#[cfg(feature = "tracing")]
542										info!("Well-known name: {} with owner: {} inserted into the peer list.", &well_known_name, &new_owner_unique_name);
543									}
544								}
545
546								// Well-known name left the bus.
547								(Some(old_owner_unique_name), None) => {
548									peers.remove_well_known(
549										&well_known_name,
550										old_owner_unique_name,
551									);
552
553									#[cfg(feature = "tracing")]
554									info!(
555										"Well-known name: {} with owner: {} removed from the peer list.",
556										&well_known_name,
557										&old_owner_unique_name
558									);
559								},
560
561								// Well-known name received a new owner on the bus.
562								(Some(old_owner_unique_name), Some(new_owner_unique_name)) => {
563									if let Ok(()) = peers.update_well_known_owner(&well_known_name, old_owner_unique_name, new_owner_unique_name, &conn).await.inspect_err(|#[allow(unused_variables)] err| {
564										#[cfg(feature = "tracing")]
565										warn!("Failed to update well-known name: {} owner from: {} to: {} - {}", &well_known_name, &old_owner_unique_name, &new_owner_unique_name, err);
566									}) {
567										#[cfg(feature = "tracing")]
568										info!("Well-known name: {} updated owner from: {} to: {}", &well_known_name, &old_owner_unique_name, &new_owner_unique_name);
569									};
570								}
571							}
572						}
573					} // End of match on `name`
574				} // End of while loop
575
576				#[cfg(feature = "tracing")]
577				tracing::warn!("Peer listener task stopped, clearing peers list.");
578				peers.clear();
579			}, "PeerListenerTask")
580			.detach();
581	}
582
583	/// Clears the list of peers.
584	///
585	/// # Note
586	/// This is used to reset the list of peers when the D-Bus connection is lost.
587	fn clear(&self) {
588		let mut peers = self.peers.lock().expect("lock already held by current thread");
589		peers.clear();
590	}
591}

616impl P2P for crate::AccessibilityConnection {
617	/// Returns a P2P connected `AccessibleProxy` for the object, _if available_.\
618	/// If the application does not support P2P, an `AccessibleProxy` with a bus connection is returned.
619	///
620	/// # Examples
621	/// ```rust
622	/// # use tokio_test::block_on;
623	/// use zbus::names::UniqueName;
624	/// use zbus::zvariant::ObjectPath;
625	/// use atspi_proxies::accessible::AccessibleProxy;
626	/// use atspi_common::ObjectRef;
627	/// use atspi_connection::{P2P, Peer};
628	/// use atspi_connection::AccessibilityConnection;
629	///
630	/// # block_on(async {
631	/// let conn = AccessibilityConnection::new().await.unwrap();
632	///
633	/// let name = UniqueName::from_static_str_unchecked(":1.1");
634	/// let path = ObjectPath::from_static_str_unchecked("/org/freedesktop/accessible/root");
635	///
636	/// let object_ref = ObjectRef::new_owned(name, path);
637	/// let accessible_proxy = conn.object_as_accessible(&object_ref).await;
638	/// assert!(
639	///    accessible_proxy.is_ok(),
640	///    "Failed to get accessible proxy: {:?}",
641	///    accessible_proxy.err()
642	/// );
643	/// # });
644	/// ```
645	///
646	/// Handling `ObjectRef::Null` case:
647	///
648	/// ```rust
649	/// # use tokio_test::block_on;
650	/// use atspi_proxies::accessible::AccessibleProxy;
651	/// use atspi_common::{AtspiError, ObjectRef, ObjectRefOwned};
652	/// use atspi_connection::P2P;
653	/// use atspi_connection::AccessibilityConnection;
654	///
655	/// # block_on(async {
656	/// let conn = AccessibilityConnection::new().await.unwrap();
657	/// let object_ref = ObjectRef::Null;
658	/// let object_ref = ObjectRefOwned::new(object_ref); // Assume we received this from `Accessible.Parent`
659	///
660	/// let res = conn.object_as_accessible(&object_ref).await;
661	/// match res {
662	///     Ok(proxy) => {
663	///         // Use the proxy
664	///         let _proxy: AccessibleProxy<'_> = proxy;
665	///     }
666	///     Err(AtspiError::NullRef(_msg)) => {
667	///         // Handle null-reference case
668	///     }
669	///     Err(_other) => {
670	///         // Handle other error types
671	///     }
672	/// }
673	/// # });
674	/// ```
675	///
676	/// # Errors
677	/// If the method is called with a null-reference `ObjectRef`, it will return an `AtspiError::NullRef`.
678	/// Users should ensure that the `ObjectRef` is non-null before calling this method or handle the result.
679	/// If the `AccessibleProxy` cannot be created, or if the object path is invalid.
680	///
681	/// # Note
682	/// This function will first try to find a [`Peer`] with a P2P connection
683	async fn object_as_accessible(&self, obj: &ObjectRefOwned) -> AtspiResult<AccessibleProxy<'_>> {
684		if obj.is_null() {
685			return Err(AtspiError::NullRef(
686				"`p2p::object_as_accessible` called with null-reference ObjectRef",
687			));
688		}
689
690		let name = obj.name().ok_or(AtspiError::MissingName)?.to_owned();
691		let name = OwnedUniqueName::from(name);
692		let path = obj.path();
693
694		let lookup = self
695			.peers
696			.peers
697			.lock()
698			.expect("lock already held by current thread")
699			.iter()
700			.find(|peer| &name == peer.unique_name())
701			.cloned();
702
703		if let Some(peer) = lookup {
704			// If a peer is found, create an `AccessibleProxy` with a P2P connection
705			AccessibleProxy::builder(peer.connection())
706				.path(path)?
707				.cache_properties(CacheProperties::No)
708				.build()
709				.await
710				.map_err(Into::into)
711		} else {
712			// If _no_ peer was found, fall back to the bus connection
713			let conn = self.connection();
714			AccessibleProxy::builder(conn)
715				.path(path)?
716				.cache_properties(CacheProperties::No)
717				.build()
718				.await
719				.map_err(Into::into)
720		}
721	}
722
723	/// Returns a P2P connected [`AccessibleProxy`] to the root accessible object for the given bus name _if available_.\
724	/// If the P2P connection is not available, it returns an `AccessibleProxy` with a bus connection.
725	///
726	/// # Examples
727	///
728	/// ```rust
729	/// # use tokio_test::block_on;
730	/// use zbus::names::BusName;
731	/// use atspi_proxies::accessible::AccessibleProxy;
732	/// use atspi_common::ObjectRef;
733	/// use atspi_connection::{AccessibilityConnection, P2P};
734	///
735	/// # block_on(async {
736	///   let conn = AccessibilityConnection::new().await.unwrap();
737	///   let bus_name = BusName::from_static_str("org.a11y.atspi.Registry").unwrap();
738	///   let _accessible_proxy = conn.bus_name_as_root_accessible(&bus_name).await.unwrap();
739	///   // Use the accessible proxy as needed
740	/// # });
741	/// ```
742	///
743	/// # Errors
744	/// In case of an invalid connection or object path.
745	async fn bus_name_as_root_accessible(
746		&'_ self,
747		name: &BusName<'_>,
748	) -> AtspiResult<AccessibleProxy<'_>> {
749		// Look up peer by bus name
750		let lookup = self
751			.peers
752			.peers
753			.lock()
754			.expect("lock already held by current thread")
755			.iter()
756			.find(|peer| {
757				// Check if the peer's unique name matches the bus name
758				match name {
759					BusName::Unique(unique_name) => peer.unique_name() == unique_name,
760					BusName::WellKnown(well_known_name) => {
761						peer.well_known_name().is_some_and(|w| w == well_known_name)
762					}
763				}
764			})
765			.cloned();
766
767		if let Some(peer) = lookup {
768			// If a peer is found, create an AccessibleProxy with a P2P connection
769			AccessibleProxy::builder(peer.connection())
770				.cache_properties(CacheProperties::No)
771				.build()
772				.await
773				.map_err(Into::into)
774		} else {
775			// If no peer is found, fall back to the bus connection
776			let conn = self.connection();
777			AccessibleProxy::builder(conn)
778				.cache_properties(CacheProperties::No)
779				.build()
780				.await
781				.map_err(Into::into)
782		}
783	}
