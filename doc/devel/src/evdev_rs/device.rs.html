22    ///
23    /// This function only initializesthe struct to sane default values.
24    /// To actually hook up the device to a kernel device, use `set_fd`.
25    pub fn new() -> Option<Device> {
26        let libevdev = unsafe {
27            raw::libevdev_new()
28        };
29
30        if libevdev.is_null() {
31            None
32        } else {
33            Some(Device {
34                _file: None,
35                raw: libevdev,
36            })
37        }
38    }

41    ///
42    /// This is a shortcut for
43    ///
44    /// ```
45    /// use evdev_rs::Device;
46    /// # use std::fs::File;
47    ///
48    /// let mut device = Device::new().unwrap();
49    /// # let fd = File::open("/dev/input/event0").unwrap();
50    /// device.set_fd(fd);

52    pub fn new_from_fd(file: File) -> io::Result<Device> {
53        let mut libevdev = 0 as *mut _;
54        let result = unsafe {
55            raw::libevdev_new_from_fd(file.as_raw_fd(), &mut libevdev)
56        };
57
58        match result {
59            0 => Ok(Device { _file: Some(file), raw: libevdev }),
60            error => Err(io::Error::from_raw_os_error(-error)),
61        }
62    }

71    /// Returns the file associated with the device
72    ///
73    /// if the `set_fd` hasn't been called yet then it return `None`
74    pub fn fd(&self) -> Option<File> {
75        let result = unsafe {
76            raw::libevdev_get_fd(self.raw)
77        };
78
79        if result == 0 {
80            None
81        } else {
82            unsafe {
83                let f = File::from_raw_fd(result);
84                Some(f)
85            }
86        }
87    }

111    /// Change the fd for this device, without re-reading the actual device.
112    ///
113    /// If the fd changes after initializing the device, for example after a
114    /// VT-switch in the X.org X server, this function updates the internal fd
115    /// to the newly opened. No check is made that new fd points to the same
116    /// device. If the device has changed, evdev's behavior is undefined.
117    ///
118    /// evdev device does not sync itself after changing the fd and keeps the current
119    /// device state. Use next_event with the FORCE_SYNC flag to force a re-sync.
120    ///
121    /// # Example
122    ///
123    /// ```rust,ignore
124    /// dev.change_fd(new_fd);
125    /// dev.next_event(evdev::FORCE_SYNC);
126    /// while dev.next_event(evdev::SYNC).ok().unwrap().0 == ReadStatus::SYNC
127    ///                             {} // noop
128    /// ```
129    /// After changing the fd, the device is assumed ungrabbed and a caller must
130    /// call libevdev_grab() again.
131    ///
132    /// It is an error to call this function before calling set_fd().
133    pub fn change_fd(&mut self, file: File) -> io::Result<()>  {
134        let result = unsafe {
135            raw::libevdev_change_fd(self.raw, file.as_raw_fd())
136        };
137
138        match result {
139            0 => {
140                self._file = Some(file);
141                Ok(())
142            },
143            error => Err(io::Error::from_raw_os_error(-error))
144        }
145    }

148    ///
149    /// This prevents other clients (including kernel-internal ones such as
150    /// rfkill) from receiving events from this device. This is generally a
151    /// bad idea. Don't do this.Grabbing an already grabbed device, or
152    /// ungrabbing an ungrabbed device is a noop and always succeeds.
153    ///

159            raw::libevdev_grab(self.raw, grab as c_int)
160        };
161
162        match result {
163            0 => Ok(()),
164            error => Err(io::Error::from_raw_os_error(-error)),
165        }
166    }

208    /// Returns `true` if device support the InputProp/EventType/EventCode and false otherwise
209    pub fn has(&self, blob: &dyn Any) -> bool {
210        if let Some(ev_type) = blob.downcast_ref::<EventType>() {
211            self.has_event_type(ev_type)
212        } else if let Some(ev_code) = blob.downcast_ref::<EventCode>() {
213            self.has_event_code(ev_code)
214        } else if let Some(prop) = blob.downcast_ref::<InputProp>() {
215            self.has_property(prop)
216        } else {
217            false
218        }
219    }

339    ///
340    /// This function does not read an event off the fd and may not access the
341    /// fd at all. If there are events queued internally this function will
342    /// return non-zero. If the internal queue is empty, this function will poll
343    /// the file descriptor for data.
344    ///
345    /// This is a convenience function for simple processes, most complex programs

349    /// poll(2) return is safe. You do not need `has_event_pending` if
350    /// you're using select(2) or poll(2).

352        unsafe {
353            raw::libevdev_has_event_pending(self.raw) > 0

488    /// Forcibly enable an event type on this device, even if the underlying
489    /// device does not support it. While this cannot make the device actually
490    /// report such events, it will now return true for libevdev_has_event_code().
491    ///
492    /// The last argument depends on the type and code:
493    /// If type is EV_ABS, data must be a pointer to a struct input_absinfo
494    /// containing the data for this axis.
495    /// If type is EV_REP, data must be a pointer to a int containing the data
496    /// for this axis.
497    /// For all other types, the argument must be NULL.
498    ///
499    /// Note: Please use the `enable` function instead. This function is only
500    /// available for the sake of maintaining compatibility with libevdev.
501    pub fn enable_event_code(&self, ev_code: &EventCode, blob: Option<&dyn Any>)
502                             -> io::Result<()> {
503        let (ev_type, ev_code) = event_code_to_int(ev_code);
504
505        let data = blob
506            .map(|data| {
507                data.downcast_ref::<AbsInfo>()
508                    .map(|absinfo| &absinfo.as_raw() as *const _ as *const c_void)
509                    .unwrap_or_else(|| data as *const _ as *const c_void)
510            })
511            .unwrap_or_else(|| ptr::null() as *const _ as *const c_void);
512
513        let result = unsafe {
514            raw::libevdev_enable_event_code(self.raw,
515                                            ev_type as c_uint,
516                                            ev_code as c_uint,
517                                            data)
518        };
519
520        match result {
521            0 => Ok(()),
522            error => Err(io::Error::from_raw_os_error(-error))
523        }
524    }

640    /// This is a modification only affecting this representation of
641    /// this device.
642    pub fn set_clock_id(&self, clockid: i32) -> io::Result<()> {
643         let result = unsafe {
644            raw::libevdev_set_clock_id(self.raw,
645                                       clockid as c_int)
646        };
647
648        match result {
649            0 => Ok(()),
650            error => Err(io::Error::from_raw_os_error(-error))
651        }
652    }
653
654    /// Get the next event from the device. This function operates in two different
655    /// modes: normal mode or sync mode.
656    ///
657    /// In normal mode (when flags has `evdev::NORMAL` set), this function returns
658    /// `ReadStatus::Success` and returns the event. If no events are available at
659    /// this time, it returns `-EAGAIN` as `Err`.
660    ///
661    /// If the current event is an `EV_SYN::SYN_DROPPED` event, this function returns
662    /// `ReadStatus::Sync` and is set to the `EV_SYN` event.The caller should now call
663    /// this function with the `evdev::SYNC` flag set, to get the set of events that
664    /// make up the device state delta. This function returns ReadStatus::Sync for
665    /// each event part of that delta, until it returns `-EAGAIN` once all events
666    /// have been synced.
667    ///
668    /// If a device needs to be synced by the caller but the caller does not call
669    /// with the `evdev::SYNC` flag set, all events from the diff are dropped after
670    /// evdev updates its internal state and event processing continues as normal.
671    /// Note that the current slot and the state of touch points may have updated
672    /// during the `SYN_DROPPED` event, it is strongly recommended that a caller
673    /// ignoring all sync events calls `current_slot` and checks the
674    /// `ABS_MT_TRACKING_ID` values for all slots.
675    ///
676    /// If a device has changed state without events being enqueued in evdev,
677    /// e.g. after changing the file descriptor, use the `evdev::FORCE_SYNC` flag.
678    /// This triggers an internal sync of the device and `next_event` returns
679    /// `ReadStatus::Sync`.
680    pub fn next_event(&self, flags: ReadFlag)
681                      -> io::Result<(ReadStatus, InputEvent)> {
682        let mut ev = raw::input_event {
683            time: raw::timeval {
684                tv_sec: 0,
685                tv_usec: 0,
686            },
687            type_: 0,
688            code: 0,
689            value: 0,
690        };
691
692        let result = unsafe {
693            raw::libevdev_next_event(self.raw, flags.bits as c_uint, &mut ev)
694        };
695
696        let event = InputEvent {
697            time: TimeVal {
698                tv_sec: ev.time.tv_sec,
699                tv_usec: ev.time.tv_usec,
700            },
701            event_type: int_to_event_type(ev.type_ as u32).unwrap(),
702            event_code: int_to_event_code(ev.type_ as u32, ev.code as u32),
703            value: ev.value,
704        };
705
706        match result {
707            raw::LIBEVDEV_READ_STATUS_SUCCESS => Ok((ReadStatus::Success, event)),
708            raw::LIBEVDEV_READ_STATUS_SYNC => Ok((ReadStatus::Sync, event)),
709            error => Err(io::Error::from_raw_os_error(-error)),
710        }
711    }
712}
