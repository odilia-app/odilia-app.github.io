1//! A lock-free, eventually consistent, concurrent multi-value map.
2//!
3//! This map implementation allows reads and writes to execute entirely in parallel, with no
4//! implicit synchronization overhead. Reads never take locks on their critical path, and neither
5//! do writes assuming there is a single writer (multi-writer is possible using a `Mutex`), which
6//! significantly improves performance under contention.
7//!
8//! The trade-off exposed by this module is one of eventual consistency: writes are not visible to
9//! readers except following explicit synchronization. Specifically, readers only see the
10//! operations that preceeded the last call to `WriteHandle::refresh` by a writer. This lets
11//! writers decide how stale they are willing to let reads get. They can refresh the map after
12//! every write to emulate a regular concurrent `HashMap`, or they can refresh only occasionally to
13//! reduce the synchronization overhead at the cost of stale reads.
14//!
15//! For read-heavy workloads, the scheme used by this module is particularly useful. Writers can
16//! afford to refresh after every write, which provides up-to-date reads, and readers remain fast
17//! as they do not need to ever take locks.
18//!
19//! The map is multi-value, meaning that every key maps to a *collection* of values. This
20//! introduces some memory cost by adding a layer of indirection through a `Vec` for each value,
21//! but enables more advanced use. This choice was made as it would not be possible to emulate such
22//! functionality on top of the semantics of this map (think about it -- what would the operational
23//! log contain?).
24//!
25//! To faciliate more advanced use-cases, each of the two maps also carry some customizeable
26//! meta-information. The writers may update this at will, and when a refresh happens, the current
27//! meta will also be made visible to readers. This could be useful, for example, to indicate what
28//! time the refresh happened.
29//!
30//! # Examples
31//!
32//! Single-reader, single-writer
33//!
34//! ```
35//! // new will use the default HashMap hasher, and a meta of ()
36//! // note that we get separate read and write handles
37//! // the read handle can be cloned to have more readers
38//! let (book_reviews_r, mut book_reviews_w) = evmap::new();
39//!
40//! // review some books.
41//! book_reviews_w.insert("Adventures of Huckleberry Finn",    "My favorite book.");
42//! book_reviews_w.insert("Grimms' Fairy Tales",               "Masterpiece.");
43//! book_reviews_w.insert("Pride and Prejudice",               "Very enjoyable.");
44//! book_reviews_w.insert("The Adventures of Sherlock Holmes", "Eye lyked it alot.");
45//!
46//! // at this point, reads from book_reviews_r will not see any of the reviews!
47//! assert_eq!(book_reviews_r.len(), 0);
48//! // we need to refresh first to make the writes visible
49//! book_reviews_w.refresh();
50//! assert_eq!(book_reviews_r.len(), 4);
51//! // reads will now return Some() because the map has been initialized
52//! assert_eq!(book_reviews_r.get("Grimms' Fairy Tales").map(|rs| rs.len()), Some(1));
53//!
54//! // remember, this is a multi-value map, so we can have many reviews
55//! book_reviews_w.insert("Grimms' Fairy Tales",               "Eh, the title seemed weird.");
56//! book_reviews_w.insert("Pride and Prejudice",               "Too many words.");
57//!
58//! // but again, new writes are not yet visible
59//! assert_eq!(book_reviews_r.get("Grimms' Fairy Tales").map(|rs| rs.len()), Some(1));
60//!
61//! // we need to refresh first
62//! book_reviews_w.refresh();
63//! assert_eq!(book_reviews_r.get("Grimms' Fairy Tales").map(|rs| rs.len()), Some(2));
64//!
65//! // oops, this review has a lot of spelling mistakes, let's delete it.
66//! // empty deletes *all* reviews (though in this case, just one)
67//! book_reviews_w.empty("The Adventures of Sherlock Holmes");
68//! // but again, it's not visible to readers until we refresh
69//! assert_eq!(book_reviews_r.get("The Adventures of Sherlock Holmes").map(|rs| rs.len()), Some(1));
70//! book_reviews_w.refresh();
71//! assert_eq!(book_reviews_r.get("The Adventures of Sherlock Holmes").map(|rs| rs.len()), None);
72//!
73//! // look up the values associated with some keys.
74//! let to_find = ["Pride and Prejudice", "Alice's Adventure in Wonderland"];
75//! for book in &to_find {
76//!     if let Some(reviews) = book_reviews_r.get(book) {
77//!         for review in &*reviews {
78//!             println!("{}: {}", book, review);
79//!         }
80//!     } else {
81//!         println!("{} is unreviewed.", book);
82//!     }
83//! }
84//!
85//! // iterate over everything.
86//! for (book, reviews) in &book_reviews_r.read().unwrap() {
87//!     for review in reviews {
88//!         println!("{}: \"{}\"", book, review);
89//!     }
90//! }
91//! ```
92//!
93//! Reads from multiple threads are possible by cloning the `ReadHandle`.
94//!
95//! ```
96//! use std::thread;
97//! let (book_reviews_r, mut book_reviews_w) = evmap::new();
98//!
99//! // start some readers
100//! let readers: Vec<_> = (0..4).map(|_| {
101//!     let r = book_reviews_r.clone();
102//!     thread::spawn(move || {
103//!         loop {
104//!             let l = r.len();
105//!             if l == 0 {
106//!                 thread::yield_now();
107//!             } else {
108//!                 // the reader will either see all the reviews,
109//!                 // or none of them, since refresh() is atomic.
110//!                 assert_eq!(l, 4);
111//!                 break;
112//!             }
113//!         }
114//!     })
115//! }).collect();
116//!
117//! // do some writes
118//! book_reviews_w.insert("Adventures of Huckleberry Finn",    "My favorite book.");
119//! book_reviews_w.insert("Grimms' Fairy Tales",               "Masterpiece.");
120//! book_reviews_w.insert("Pride and Prejudice",               "Very enjoyable.");
121//! book_reviews_w.insert("The Adventures of Sherlock Holmes", "Eye lyked it alot.");
122//! // expose the writes
123//! book_reviews_w.refresh();
124//!
125//! // you can read through the write handle
126//! assert_eq!(book_reviews_w.len(), 4);
127//!
128//! // the original read handle still works too
129//! assert_eq!(book_reviews_r.len(), 4);
130//!
131//! // all the threads should eventually see .len() == 4
132//! for r in readers.into_iter() {
133//!     assert!(r.join().is_ok());
134//! }
135//! ```
136//!
137//! If multiple writers are needed, the `WriteHandle` must be protected by a `Mutex`.
138//!
139//! ```
140//! use std::thread;
141//! use std::sync::{Arc, Mutex};
142//! let (book_reviews_r, mut book_reviews_w) = evmap::new();
143//!
144//! // start some writers.
145//! // since evmap does not support concurrent writes, we need
146//! // to protect the write handle by a mutex.
147//! let w = Arc::new(Mutex::new(book_reviews_w));
148//! let writers: Vec<_> = (0..4).map(|i| {
149//!     let w = w.clone();
150//!     thread::spawn(move || {
151//!         let mut w = w.lock().unwrap();
152//!         w.insert(i, true);
153//!         w.refresh();
154//!     })
155//! }).collect();
156//!
157//! // eventually we should see all the writes
158//! while book_reviews_r.len() < 4 { thread::yield_now(); };
159//!
160//! // all the threads should eventually finish writing
161//! for w in writers.into_iter() {
162//!     assert!(w.join().is_ok());
163//! }
164//! ```
165//!
166//! `ReadHandle` is not `Sync` as it is not safe to share a single instance
167//! amongst threads. A fresh `ReadHandle` needs to be created for each thread
168//! either by cloning a `ReadHandle` or from a `ReadHandleFactory`.
169//!
170//! The reason for this is that each `ReadHandle` assumes that only one
171//! thread operates on it at a time. For details, see the implementation
172//! comments on `ReadHandle`.
173//!
174//!```compile_fail
175//! use evmap::ReadHandle;
176//!
177//! fn is_sync<T: Sync>() {
178//!   // dummy function just used for its parameterized type bound
179//! }
180//!
181//! // the line below will not compile as ReadHandle does not implement Sync
182//!
183//! is_sync::<ReadHandle<u64, u64>>()
184//!```
185//!
186//! `ReadHandle` **is** `Send` though, since in order to send a `ReadHandle`,
187//! there must be no references to it, so no thread is operating on it.
188//!
189//!```
190//! use evmap::ReadHandle;
191//!
192//! fn is_send<T: Send>() {
193//!   // dummy function just used for its parameterized type bound
194//! }
195//!
196//! is_send::<ReadHandle<u64, u64>>()
197//!```
198//!
199//! For further explanation of `Sync` and `Send` [here](https://doc.rust-lang.org/nomicon/send-and-sync.html)
200//!
201//! # Implementation
202//!
203//! Under the hood, the map is implemented using two regular `HashMap`s, an operational log,
204//! epoch counting, and some pointer magic. There is a single pointer through which all readers
205//! go. It points to a `HashMap`, which the readers access in order to read data. Every time a read
206//! has accessed the pointer, they increment a local epoch counter, and they update it again when
207//! they have finished the read (see #3 for more information). When a write occurs, the writer
208//! updates the other `HashMap` (for which there are no readers), and also stores a copy of the
209//! change in a log (hence the need for `Clone` on the keys and values). When
210//! `WriteHandle::refresh` is called, the writer, atomically swaps the reader pointer to point to
211//! the other map. It then waits for the epochs of all current readers to change, and then replays
212//! the operational log to bring the stale map up to date.
213//!
214//! Since the implementation uses regular `HashMap`s under the hood, table resizing is fully
215//! supported. It does, however, also mean that the memory usage of this implementation is
216//! approximately twice of that of a regular `HashMap`, and more if writes rarely refresh after
217//! writing.
218//!
219//! # Value storage
220//!
221//! The values for each key in the map are stored in [`Values`]. Conceptually, each `Values` is a
222//! _bag_ or _multiset_; it can store multiple copies of the same value. `evmap` applies some
223//! cleverness in an attempt to reduce unnecessary allocations and keep the cost of operations on
224//! even large value-bags small. For small bags, `Values` uses the `smallvec` crate. This avoids
225//! allocation entirely for single-element bags, and uses a `Vec` if the bag is relatively small.
226//! For large bags, `Values` uses the `hashbag` crate, which enables `evmap` to efficiently look up
227//! and remove specific elements in the value bag. For bags larger than one element, but smaller
228//! than the threshold for moving to `hashbag`, we use `smallvec` to avoid unnecessary hashing.
229//! Operations such as `Fit` and `Replace` will automatically switch back to the inline storage if
230//! possible. This is ideal for maps that mostly use one element per key, as it can improvate
231//! memory locality with less indirection.

253/// Unary predicate used to retain elements.
254///
255/// The predicate function is called once for each distinct value, and `true` if this is the
256/// _first_ call to the predicate on the _second_ application of the operation.
257pub struct Predicate<V>(pub(crate) Box<dyn FnMut(&V, bool) -> bool + Send>);
258
259impl<V> Predicate<V> {
260    /// Evaluate the predicate for the given element
261    #[inline]
262    pub fn eval(&mut self, value: &V, reset: bool) -> bool {
263        (*self.0)(value, reset)
264    }
265}
266
267impl<V> PartialEq for Predicate<V> {
268    #[inline]
269    fn eq(&self, other: &Self) -> bool {
270        // only compare data, not vtable: https://stackoverflow.com/q/47489449/472927
271        &*self.0 as *const _ as *const () == &*other.0 as *const _ as *const ()
272    }
273}
274
275impl<V> Eq for Predicate<V> {}
276
277impl<V> fmt::Debug for Predicate<V> {
278    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
279        f.debug_tuple("Predicate")
280            .field(&format_args!("{:p}", &*self.0 as *const _))
281            .finish()
282    }
283}

285/// A pending map operation.
286#[non_exhaustive]
287#[derive(PartialEq, Eq, Debug)]
288pub enum Operation<K, V> {
289    /// Replace the set of entries for this key with this value.
290    Replace(K, V),
291    /// Add this value to the set of entries for this key.
292    Add(K, V),
293    /// Remove this value from the set of entries for this key.
294    Remove(K, V),
295    /// Remove the value set for this key.
296    Empty(K),
297    #[cfg(feature = "indexed")]
298    /// Drop a key at a random index
299    EmptyRandom(usize),
300    /// Remove all values in the value set for this key.
301    Clear(K),
302    /// Remove all values for all keys.
303    ///
304    /// Note that this will iterate once over all the keys internally.
305    Purge,
306    /// Retains all values matching the given predicate.
307    Retain(K, Predicate<V>),
308    /// Shrinks [`Values`] to their minimum necessary size, freeing memory
309    /// and potentially improving cache locality.
310    ///
311    /// If no key is given, all `Values` will shrink to fit.
312    Fit(Option<K>),
313    /// Reserves capacity for some number of additional elements in [`Values`]
314    /// for the given key. If the given key does not exist, allocate an empty
315    /// `Values` with the given capacity.
316    ///
317    /// This can improve performance by pre-allocating space for large bags of values.
318    Reserve(K, usize),
319}

330/// Options for how to initialize the map.
331///
332/// In particular, the options dictate the hashing function, meta type, and initial capacity of the
333/// map.
334pub struct Options<M, S>
335where
336    S: BuildHasher,
337{
338    meta: M,
339    hasher: S,
340    capacity: Option<usize>,
341}
342
343impl<M, S> fmt::Debug for Options<M, S>
344where
345    S: BuildHasher,
346    M: fmt::Debug,
347{
348    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
349        f.debug_struct("Options")
350            .field("meta", &self.meta)
351            .field("capacity", &self.capacity)
352            .finish()
353    }
354}
355
356impl Default for Options<(), RandomState> {
357    fn default() -> Self {
358        Options {
359            meta: (),
360            hasher: RandomState::default(),
361            capacity: None,
362        }
363    }
364}
365
366impl<M, S> Options<M, S>
367where
368    S: BuildHasher,
369{
370    /// Set the initial meta value for the map.
371    pub fn with_meta<M2>(self, meta: M2) -> Options<M2, S> {
372        Options {
373            meta,
374            hasher: self.hasher,
375            capacity: self.capacity,
376        }
377    }
378
379    /// Set the hasher used for the map.
380    pub fn with_hasher<S2>(self, hash_builder: S2) -> Options<M, S2>
381    where
382        S2: BuildHasher,
383    {
384        Options {
385            meta: self.meta,
386            hasher: hash_builder,
387            capacity: self.capacity,
388        }
389    }
390
391    /// Set the initial capacity for the map.
392    pub fn with_capacity(self, capacity: usize) -> Options<M, S> {
393        Options {
394            meta: self.meta,
395            hasher: self.hasher,
396            capacity: Some(capacity),
397        }
398    }
399
400    /// Create the map, and construct the read and write handles used to access it.
401    #[allow(clippy::type_complexity)]
402    pub fn construct<K, V>(self) -> (ReadHandle<K, V, M, S>, WriteHandle<K, V, M, S>)
403    where
404        K: Eq + Hash + Clone,
405        S: BuildHasher + Clone,
406        V: Eq + Hash + ShallowCopy,
407        M: 'static + Clone,
408    {
409        let epochs = Default::default();
410        let inner = if let Some(cap) = self.capacity {
411            Inner::with_capacity_and_hasher(self.meta, cap, self.hasher)
412        } else {
413            Inner::with_hasher(self.meta, self.hasher)
414        };
415
416        let mut w_handle = inner.clone();
417        w_handle.mark_ready();
418        let r = read::new(inner, Arc::clone(&epochs));
419        let w = write::new(w_handle, epochs, r.clone());
420        (r, w)
421    }
422}
