55#[tracing::instrument(skip(state, shutdown))]
56async fn notifications_monitor(
57	state: Arc<ScreenReaderState>,
58	shutdown: CancellationToken,
59) -> eyre::Result<()> {
60	let mut stream = listen_to_dbus_notifications()
61		.instrument(tracing::info_span!("creating notification listener"))
62		.await?;
63	loop {
64		tokio::select! {
65		    Some(notification) = stream.next() => {
66		      let notification_message =
67			format!("new notification: {}, {}, {}.", notification.app_name, notification.title, notification.body);
68		      state.say(Priority::Important, notification_message).await;
69		    },
70		    () = shutdown.cancelled() => {
71		      tracing::debug!("Shutting down notification task.");
72		      break;
73		    },
74		}
75	}
76	Ok(())
77}
78#[tracing::instrument]
79async fn sigterm_signal_watcher(
80	token: CancellationToken,
81	tracker: TaskTracker,
82) -> eyre::Result<()> {
83	let timeout_duration = Duration::from_millis(500); //todo: perhaps take this from the configuration file at some point
84	let mut c = signal(SignalKind::interrupt())?;
85	c.recv().instrument(tracing::debug_span!("Watching for Ctrl+C")).await;
86	tracing::debug!("Asking all processes to stop.");
87	tracing::debug!("cancelling all tokens");
88	token.cancel();
89	tracing::debug!(?timeout_duration, "waiting for all tasks to finish");
90	timeout(timeout_duration, tracker.wait()).await?;
91	tracing::debug!("All listeners have stopped.");
92	tracing::debug!("Goodbye, Odilia!");
93	Ok(())
94}

119#[tracing::instrument(ret)]
120async fn focused(
121	state_changed: CacheEvent<Focused>,
122	EventProp(name): EventProp<Name>,
123	EventProp(description): EventProp<Description>,
124	EventProp(relation_set): EventProp<RelationSet>,
125) -> impl TryIntoCommands {
126	//because the current command implementation doesn't allow for multiple speak commands without interrupting the previous utterance, this is more or less an accumulating buffer for that utterance
127	let mut utterance_buffer = String::new();
128	//does this have a text or a name?
129	// in order for the borrow checker to not scream that we move ownership of item.text, therefore making item partially moved, we only take a reference here, because in truth the only thing that we need to know is if the string is empty, because the extending of the buffer will imply a clone anyway
130	let text = &state_changed.item.text;
131	if text.is_empty() {
132		//then the label can either be the accessible name, the description, or the relations set, aka labeled by another object
133		//unfortunately, the or_else function of result doesn't accept async cloasures or cloasures with async blocks, so we can't use lazy loading here at the moment. The performance penalty is minimal however, because this should be in cache anyway
134		let label = if let Some(n) = name.as_deref() {
135			n.to_string()
136		} else if let Some(d) = description.as_deref() {
137			d.to_string()
138		//otherwise, if this is empty too, we try to use the relations set to find the element labeling this one
139		} else {
140			relation_set
141				.iter()
142				// we only need entries which contain the wanted relationship, only labeled by for now
143				.filter(|elem| elem.0 == RelationType::LabelledBy)
144				.cloned()
145				// we have to remove the first item of the entries, because it's constant now that we filtered by it
146				//furthermore, after doing this, we'd have something like Vec<Vec<Item>>, which is not what we need, we need something like Vec<Item>, so we do both the flattening of the structure and the mapping in one function call
147				.flat_map(|this| this.1)
148				// from a collection of items, to a collection of strings, in this case the text of those labels, because yes, technically there can be more than one
149				.map(|this| this.text)
150				// gather all that into a string, separated by newlines or spaces I think
151				.collect()
152		};
153		utterance_buffer += &label;
154	} else {
155		//then just append to the buffer and be done with it
156		utterance_buffer += text;
157	}
158	let role = state_changed.item.role;
159	//there has to be a space between the accessible name of an object and its role, so insert it now
160	write!(utterance_buffer, " {}", role.name()).expect("Able to write to string");
161	Ok(vec![
162		Focus(state_changed.item.object).into(),
163		Speak(utterance_buffer, Priority::Text).into(),
164	])
165}

191#[tracing::instrument(ret, err)]
192async fn caret_moved(
193	caret_moved: CacheEvent<TextCaretMovedEvent>,
194	LastCaretPos(last_pos): LastCaretPos,
195	LastFocused(last_focus): LastFocused,
196) -> Result<Vec<OdiliaCommand>, OdiliaError> {
197	let mut commands: Vec<OdiliaCommand> =
198		vec![CaretPos(caret_moved.inner.position.try_into()?).into()];
199
200	if last_focus == caret_moved.item.object {
201		let start = min(caret_moved.inner.position.try_into()?, last_pos);
202		let end = max(caret_moved.inner.position.try_into()?, last_pos);
203		if let Some(text) = caret_moved.item.text.get(start..end) {
204			commands.extend((Priority::Text, text.to_string()).into_commands());
205		} else {
206			return Err(OdiliaError::Generic(format!(
207				"Slide {}..{} could not be created from {}",
208				start, end, caret_moved.item.text
209			)));
210		}
211	} else {
212		let (text, _, _) = caret_moved
213			.item
214			.get_string_at_offset(
215				caret_moved.inner.position.try_into()?,
216				Granularity::Line,
217			)
218			.await?;
219		commands.extend((Priority::Text, text).into_commands());
220	}
221	Ok(commands)
222}

224#[tokio::main]
225async fn main() -> eyre::Result<()> {
226	let args = Args::parse();
227
228	//initialize the primary token for task cancelation
229	let token = CancellationToken::new();
230
231	//initialize a task tracker, which will allow us to wait for all tasks to finish
232	let tracker = TaskTracker::new();
233
234	//initializing configuration
235	let config = load_configuration(args.config)?;
236	//initialize logging, with the provided config
237	logging::init(&config)?;
238
239	tracing::info!(?config, "this configuration was used to prepair odilia");
240
241	// Make sure applications with dynamic accessibility support do expose their AT-SPI2 interfaces.
242	if let Err(e) = atspi_connection::set_session_accessibility(true)
243		.instrument(tracing::info_span!("setting accessibility enabled flag"))
244		.await
245	{
246		tracing::error!("Could not set AT-SPI2 IsEnabled property because: {}", e);
247	}
248	// this is the channel which handles all SSIP commands. If SSIP is not allowed to operate on a separate task, then waiting for the receiving message can block other long-running operations like structural navigation.
249	// Although in the future, this may possibly be resolved through a proper cache, I think it still makes sense to separate SSIP's IO operations to a separate task.
250	//  it is very important that this is *never* full, since it can cause deadlocking if the other task sending the request is working with zbus.
251	let (ssip_req_tx, ssip_req_rx) = mpsc::channel::<ssip_client_async::Request>(128);
252	let (mut ev_tx, ev_rx) =
253		futures::channel::mpsc::channel::<Result<atspi::Event, atspi::AtspiError>>(10_000);
254	// Initialize state
255	let state = Arc::new(ScreenReaderState::new(ssip_req_tx, config).await?);
256	let ssip = odilia_tts::create_ssip_client().await?;
257
258	if state.say(Priority::Message, "Welcome to Odilia!".to_string()).await {
259		tracing::debug!("Welcome message spoken.");
260	} else {
261		tracing::error!("Welcome message failed. Odilia is not able to continue in this state. Exiting now.");
262		state.close_speech().await;
263		exit(1);
264	}
265
266	// Register events
267	tokio::try_join!(
268		state.register_event::<object::StateChangedEvent>(),
269		state.register_event::<object::TextCaretMovedEvent>(),
270		state.register_event::<object::ChildrenChangedEvent>(),
271		state.register_event::<object::TextChangedEvent>(),
272		state.register_event::<document::LoadCompleteEvent>(),
273		state.add_cache_match_rule(),
274	)?;
275
276	// load handlers
277	let handlers = Handlers::new(state.clone())
278		.command_listener(speak)
279		.command_listener(new_focused_item)
280		.command_listener(new_caret_pos)
281		.atspi_listener(doc_loaded)
282		.atspi_listener(caret_moved)
283		.atspi_listener(focused)
284		.atspi_listener(unfocused);
285	let ssip_event_receiver =
286		odilia_tts::handle_ssip_commands(ssip, ssip_req_rx, token.clone())
287			.map(|r| r.wrap_err("Could no process SSIP request"));
288	let notification_task = notifications_monitor(Arc::clone(&state), token.clone())
289		.map(|r| r.wrap_err("Could not process signal shutdown."));
290	let mut stream = state.atspi.event_stream();
291	// There is a reason we are not reading from the event stream directly.
292	// This `MessageStream` can only store 64 events in its buffer.
293	// And, even if it could store more (it can via options), `zbus` specifically states that:
294	// You must ensure a MessageStream is continuously polled or you will experience hangs.
295	// So, we continually poll it here, then receive it on the other end.
296	// Additioanlly, since sending is not async, but simply errors when there is an issue, this will
297	// help us avoid hangs.
298	let event_send_task = async move {
299		std::pin::pin!(&mut stream);
300		while let Some(ev) = stream.next().await {
301			if let Err(e) = ev_tx.try_send(ev) {
302				tracing::error!("Error sending event across channel! {e:?}");
303			}
304		}
305	};
306	let atspi_handlers_task = handlers.atspi_handler(ev_rx);
307
308	tracker.spawn(ssip_event_receiver);
309	tracker.spawn(notification_task);
310	tracker.spawn(atspi_handlers_task);
311	tracker.spawn(event_send_task);
312	tracker.close();
313	let _ = sigterm_signal_watcher(token, tracker)
314		.await
315		.wrap_err("can not process interrupt signal");
316	Ok(())
317}

319fn load_configuration(cli_overide: Option<PathBuf>) -> Result<ApplicationConfig, eyre::Report> {
320	// In order, do  a configuration file specified via cli, XDG_CONFIG_HOME, the usual location for system wide configuration(/etc/odilia/config.toml)
321	// If XDG_CONFIG_HOME based configuration wasn't found, create one by combining default values with the system provided ones, if available, for the user to alter, for the next run of odilia
322	//default configuration first, because that doesn't affect the priority outlined above
323	let figment = Figment::from(Serialized::defaults(ApplicationConfig::default()));
324	//cli override, if applicable
325	let figment =
326		if let Some(path) = cli_overide { figment.join(Toml::file(path)) } else { figment };
327	//create a config.toml file in `XDG_CONFIG_HOME`, to make it possible for the user to edit the default values, if it doesn't exist already
328	let xdg_dirs = xdg::BaseDirectories::with_prefix("odilia").expect(
329			"unable to find the odilia config directory according to the xdg dirs specification",
330		);
331
332	let config_path = xdg_dirs
333		.place_config_file("config.toml")
334		.expect("unable to place configuration file. Maybe your system is readonly?");
335
336	let figment = figment
337		//next, the configuration system wide, in /etc/odilia/config.toml
338		.admerge(Toml::file("/etc/odilia/config.toml"))
339		//finally, the xdg configuration
340		.admerge(Toml::file(&config_path));
341	//realise the configuration and freeze it into place
342	let config: ApplicationConfig = figment.extract()?;
343	if !config_path.exists() {
344		let toml = toml::to_string(&config)?;
345		fs::write(&config_path, toml).expect("Unable to create default config file.");
346	}
347	Ok(config)
