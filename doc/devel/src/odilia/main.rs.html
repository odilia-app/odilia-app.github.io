65/// Try to spawn the `odilia-input-server-*` binary.
66#[tracing::instrument]
67fn try_spawn_input_server(input: &InputMethod) -> eyre::Result<Child> {
68	let bin_name = format!(
69		"{}-{}",
70		"odilia-input-server",
71		match input {
72			InputMethod::Keyboard => "keyboard",
73			InputMethod::Custom(s) => &s,
74		}
75	);
76	if which::which(&bin_name).is_err() {
77		tracing::info!("Unable to find {bin_name} in $PATH; trying hardcoded paths.");
78	}
79	let mut found = which::which_in_global(
80		bin_name.clone(),
81		Some("./target/debug/:./target/release/"),
82	)?;
83	let child = match found.next() {
84		None => {
85			return Err(Report::msg(format!("Unable to find {bin_name} in $PATH or any hardcoded paths (for development). This means Odilia is uncontrollable by any mechanism!")));
86		}
87		Some(path) => {
88			tracing::info!("Input server path: {:?}", path);
89			ProcCommand::new(path).spawn()?
90		}
91	};
92	Ok(child)
93}

95#[tracing::instrument(skip(state, shutdown))]
96async fn notifications_monitor(
97	state: Arc<ScreenReaderState>,
98	shutdown: CancellationToken,
99) -> eyre::Result<()> {
100	let mut stream = listen_to_dbus_notifications()
101		.instrument(tracing::info_span!("creating notification listener"))
102		.await?;
103	loop {
104		tokio::select! {
105		    Some(notification) = stream.next() => {
106		      let notification_message =
107			format!("new notification: {}, {}, {}.", notification.app_name, notification.title, notification.body);
108		      state.say(Priority::Important, notification_message).await;
109		    },
110		    () = shutdown.cancelled() => {
111		      tracing::debug!("Shutting down notification task.");
112		      break;
113		    },
114		}
115	}
116	Ok(())
117}
118#[tracing::instrument]
119async fn sigterm_signal_watcher(
120	token: CancellationToken,
121	tracker: TaskTracker,
122	state: Arc<ScreenReaderState>,
123) -> eyre::Result<()> {
124	let timeout_duration = Duration::from_millis(500); //todo: perhaps take this from the configuration file at some point
125	let mut c = signal(SignalKind::interrupt())?;
126	c.recv().instrument(tracing::debug_span!("Watching for Ctrl+C")).await;
127	tracing::debug!("Asking all processes to stop.");
128	(*state.children_pids.lock().expect("Able to lock mutex!"))
129		.iter_mut()
130		.try_for_each(Child::kill)
131		.expect("Able to kill child processes");
132	tracing::debug!("cancelling all tokens");
133	token.cancel();
134	tracing::debug!(?timeout_duration, "waiting for all tasks to finish");
135	timeout(timeout_duration, tracker.wait()).await?;
136	tracing::debug!("All listeners have stopped.");
137	tracing::debug!("Goodbye, Odilia!");
138	Ok(())
139}

164#[tracing::instrument(ret)]
165async fn focused(
166	state_changed: CacheEvent<Focused>,
167	EventProp(name): EventProp<Name>,
168	EventProp(description): EventProp<Description>,
169	EventProp(relation_set): EventProp<RelationSet>,
170) -> impl TryIntoCommands {
171	//because the current command implementation doesn't allow for multiple speak commands without interrupting the previous utterance, this is more or less an accumulating buffer for that utterance
172	let mut utterance_buffer = String::new();
173	//does this have a text or a name?
174	// in order for the borrow checker to not scream that we move ownership of item.text, therefore making item partially moved, we only take a reference here, because in truth the only thing that we need to know is if the string is empty, because the extending of the buffer will imply a clone anyway
175	let text = &state_changed.item.text;
176	if text.is_empty() {
177		//then the label can either be the accessible name, the description, or the relations set, aka labeled by another object
178		//unfortunately, the or_else function of result doesn't accept async cloasures or cloasures with async blocks, so we can't use lazy loading here at the moment. The performance penalty is minimal however, because this should be in cache anyway
179		let label = if let Some(n) = name.as_deref() {
180			n.to_string()
181		} else if let Some(d) = description.as_deref() {
182			d.to_string()
183		//otherwise, if this is empty too, we try to use the relations set to find the element labeling this one
184		} else {
185			relation_set
186				.iter()
187				// we only need entries which contain the wanted relationship, only labeled by for now
188				.filter(|elem| elem.0 == RelationType::LabelledBy)
189				.cloned()
190				// we have to remove the first item of the entries, because it's constant now that we filtered by it
191				//furthermore, after doing this, we'd have something like Vec<Vec<Item>>, which is not what we need, we need something like Vec<Item>, so we do both the flattening of the structure and the mapping in one function call
192				.flat_map(|this| this.1)
193				// from a collection of items, to a collection of strings, in this case the text of those labels, because yes, technically there can be more than one
194				.map(|this| this.text)
195				// gather all that into a string, separated by newlines or spaces I think
196				.collect()
197		};
198		utterance_buffer += &label;
199	} else {
200		//then just append to the buffer and be done with it
201		utterance_buffer += text;
202	}
203	let role = state_changed.item.role;
204	//there has to be a space between the accessible name of an object and its role, so insert it now
205	write!(utterance_buffer, " {}", role.name()).expect("Able to write to string");
206	Ok(vec![
207		Focus(state_changed.item.object).into(),
208		Speak(utterance_buffer, Priority::Text).into(),
209	])
210}

251#[tracing::instrument(ret, err)]
252async fn caret_moved(
253	caret_moved: CacheEvent<TextCaretMovedEvent>,
254	LastCaretPos(last_pos): LastCaretPos,
255	LastFocused(last_focus): LastFocused,
256) -> Result<Vec<OdiliaCommand>, OdiliaError> {
257	let mut commands: Vec<OdiliaCommand> =
258		vec![CaretPos(caret_moved.inner.position.try_into()?).into()];
259
260	if last_focus == caret_moved.item.object {
261		let start = min(caret_moved.inner.position.try_into()?, last_pos);
262		let end = max(caret_moved.inner.position.try_into()?, last_pos);
263		if let Some(text) = caret_moved.item.text.get(start..end) {
264			commands.extend((Priority::Text, text.to_string()).into_commands());
265		} else {
266			return Err(OdiliaError::Generic(format!(
267				"Slide {}..{} could not be created from {}",
268				start, end, caret_moved.item.text
269			)));
270		}
271	} else {
272		let (text, _, _) = caret_moved
273			.item
274			.get_string_at_offset(
275				caret_moved.inner.position.try_into()?,
276				Granularity::Line,
277			)
278			.await?;
279		commands.extend((Priority::Text, text).into_commands());
280	}
281	Ok(commands)
282}

284#[tokio::main]
285async fn main() -> eyre::Result<()> {
286	let args = Args::parse();
287
288	//initialize the primary token for task cancelation
289	let token = CancellationToken::new();
290
291	//initialize a task tracker, which will allow us to wait for all tasks to finish
292	let tracker = TaskTracker::new();
293
294	//initializing configuration
295	let config = load_configuration(args.config)?;
296	//initialize logging, with the provided config
297	logging::init(&config)?;
298
299	tracing::info!(?config, "this configuration was used to prepair odilia");
300
301	// Make sure applications with dynamic accessibility support do expose their AT-SPI2 interfaces.
302	if let Err(e) = atspi_connection::set_session_accessibility(true)
303		.instrument(tracing::info_span!("setting accessibility enabled flag"))
304		.await
305	{
306		tracing::error!("Could not set AT-SPI2 IsEnabled property because: {}", e);
307	}
308	// this is the channel which handles all SSIP commands. If SSIP is not allowed to operate on a separate task, then waiting for the receiving message can block other long-running operations like structural navigation.
309	// Although in the future, this may possibly be resolved through a proper cache, I think it still makes sense to separate SSIP's IO operations to a separate task.
310	//  it is very important that this is *never* full, since it can cause deadlocking if the other task sending the request is working with zbus.
311	let (ssip_req_tx, ssip_req_rx) = mpsc::channel::<ssip_client_async::Request>(128);
312	let (mut ev_tx, ev_rx) =
313		futures::channel::mpsc::channel::<Result<atspi::Event, atspi::AtspiError>>(10_000);
314	let (input_tx, input_rx) = mpsc::channel::<ScreenReaderEvent>(255);
315	// Initialize state
316	let state = Arc::new(ScreenReaderState::new(ssip_req_tx, config).await?);
317	let ssip = odilia_tts::create_ssip_client().await?;
318
319	if state.say(Priority::Message, "Welcome to Odilia!".to_string()).await {
320		tracing::debug!("Welcome message spoken.");
321	} else {
322		tracing::error!("Welcome message failed. Odilia is not able to continue in this state. Exiting now.");
323		state.close_speech().await;
324		exit(1);
325	}
326
327	// Register events
328	tokio::try_join!(
329		state.register_event::<object::StateChangedEvent>(),
330		state.register_event::<object::TextCaretMovedEvent>(),
331		state.register_event::<object::ChildrenChangedEvent>(),
332		state.register_event::<object::TextChangedEvent>(),
333		state.register_event::<document::LoadCompleteEvent>(),
334		state.add_cache_match_rule(),
335	)?;
336
337	// load handlers
338	let handlers = Handlers::new(state.clone())
339		.command_listener(speak)
340		.command_listener(new_focused_item)
341		.command_listener(new_caret_pos)
342		.atspi_listener(doc_loaded)
343		.atspi_listener(caret_moved)
344		.atspi_listener(focused)
345		.atspi_listener(unfocused)
346		.input_listener(stop_speech)
347		.input_listener(change_mode)
348		.input_listener(structural_nav);
349
350	let ssip_event_receiver =
351		odilia_tts::handle_ssip_commands(ssip, ssip_req_rx, token.clone())
352			.map(|r| r.wrap_err("Could no process SSIP request"));
353	let notification_task = notifications_monitor(Arc::clone(&state), token.clone())
354		.map(|r| r.wrap_err("Could not process signal shutdown."));
355	let mut stream = state.atspi.event_stream();
356	// There is a reason we are not reading from the event stream directly.
357	// This `MessageStream` can only store 64 events in its buffer.
358	// And, even if it could store more (it can via options), `zbus` specifically states that:
359	// You must ensure a MessageStream is continuously polled or you will experience hangs.
360	// So, we continually poll it here, then receive it on the other end.
361	// Additioanlly, since sending is not async, but simply errors when there is an issue, this will
362	// help us avoid hangs.
363	let event_send_task = async move {
364		std::pin::pin!(&mut stream);
365		while let Some(ev) = stream.next().await {
366			if let Err(e) = ev_tx.try_send(ev) {
367				tracing::error!("Error sending event across channel! {e:?}");
368			}
369		}
370	};
371	let atspi_handlers_task = handlers.clone().atspi_handler(ev_rx);
372	let listener = odilia_input::setup_input_server()
373		.await
374		.expect("We should be able to set up input server; without it, Odilia cannot be controlled via input methods");
375	let input_task = odilia_input::sr_event_receiver(listener, input_tx, token.clone());
376	let input_handler = handlers.input_handler(input_rx);
377	let child = try_spawn_input_server(&state.config.input.method)?;
378	state.add_child_proc(child).expect("Able to add child to process!");
379
380	tracker.spawn(ssip_event_receiver);
381	tracker.spawn(notification_task);
382	tracker.spawn(atspi_handlers_task);
383	tracker.spawn(event_send_task);
384	tracker.spawn(input_task);
385	tracker.spawn(input_handler);
386	tracker.close();
387	let _ = sigterm_signal_watcher(token, tracker, Arc::clone(&state))
388		.await
389		.wrap_err("can not process interrupt signal");
390	Ok(())
391}

393fn load_configuration(cli_overide: Option<PathBuf>) -> Result<ApplicationConfig, eyre::Report> {
394	// In order, do  a configuration file specified via cli, XDG_CONFIG_HOME, the usual location for system wide configuration(/etc/odilia/config.toml)
395	// If XDG_CONFIG_HOME based configuration wasn't found, create one by combining default values with the system provided ones, if available, for the user to alter, for the next run of odilia
396	//default configuration first, because that doesn't affect the priority outlined above
397	let figment = Figment::from(Serialized::defaults(ApplicationConfig::default()));
398	//cli override, if applicable
399	let figment =
400		if let Some(path) = cli_overide { figment.join(Toml::file(path)) } else { figment };
401	//create a config.toml file in `XDG_CONFIG_HOME`, to make it possible for the user to edit the default values, if it doesn't exist already
402	let xdg_dirs = xdg::BaseDirectories::with_prefix("odilia").expect(
403			"unable to find the odilia config directory according to the xdg dirs specification",
404		);
405
406	let config_path = xdg_dirs
407		.place_config_file("config.toml")
408		.expect("unable to place configuration file. Maybe your system is readonly?");
409
410	let figment = figment
411		//next, the configuration system wide, in /etc/odilia/config.toml
412		.admerge(Toml::file("/etc/odilia/config.toml"))
413		//finally, the xdg configuration
414		.admerge(Toml::file(&config_path));
415	//realise the configuration and freeze it into place
416	let config: ApplicationConfig = figment.extract()?;
417	if !config_path.exists() {
418		let toml = toml::to_string(&config)?;
419		fs::write(&config_path, toml).expect("Unable to create default config file.");
420	}
421	Ok(config)
