59fn find_it<P>(exe_name: P) -> Option<PathBuf>
60where
61	P: AsRef<Path>,
62{
63	find_it_in(exe_name, [""].into_iter())
64}
65fn find_it_in<P, I, Item>(exe_name: P, paths: I) -> Option<PathBuf>
66where
67	P: AsRef<Path>,
68	I: Iterator<Item = Item>,
69	Item: Into<std::ffi::OsString>,
70{
71	let os_strings = paths
72		.map(Into::into)
73		.chain([env::var_os("PATH")?])
74		.collect::<Vec<std::ffi::OsString>>();
75	os_strings.iter().find_map(|paths| {
76		env::split_paths(&paths).find_map(|dir| {
77			let full_path = dir.join(&exe_name);
78			if full_path.is_file() {
79				Some(full_path)
80			} else {
81				None
82			}
83		})
84	})
85}

87async fn or_cancel<F>(f: F, token: &CancellationToken) -> Result<F::Output, std::io::Error>
88where
89	F: std::future::Future,
90{
91	token.cancelled()
92		.map(|()| Err(std::io::ErrorKind::TimedOut.into()))
93		.or(f.map(Ok))
94		.await
95}

97/// Try to spawn the `odilia-input-server-*` binary.
98#[tracing::instrument]
99fn try_spawn_input_server(input: &InputMethod) -> Result<Child, OdiliaError> {
100	let bin_name = format!(
101		"{}-{}",
102		"odilia-input-server",
103		match input {
104			InputMethod::Keyboard => "keyboard",
105			InputMethod::Custom(s) => &s,
106		}
107	);
108	if find_it(&bin_name).is_none() {
109		tracing::info!("Unable to find {bin_name} in $PATH; trying hardcoded paths.");
110	}
111	let found = find_it_in(
112		bin_name.clone(),
113		["./target/debug/", "./target/release/", "../target/debug/", "../target/release/"]
114			.into_iter(),
115	);
116	let child = match found {
117		None => {
118			return Err(format!("Unable to find {bin_name} in $PATH or any hardcoded paths (for development). This means Odilia is uncontrollable by any mechanism!").into());
119		}
120		Some(path) => {
121			tracing::info!("Input server path: {:?}", path);
122			ProcCommand::new(path).spawn()?
123		}
124	};
125	Ok(child)
126}

128#[tracing::instrument(skip(state, shutdown))]
129async fn notifications_monitor(
130	state: Arc<ScreenReaderState>,
131	shutdown: CancellationToken,
132) -> Result<(), OdiliaError> {
133	let mut stream = listen_to_dbus_notifications()
134		.instrument(tracing::info_span!("creating notification listener"))
135		.await?;
136	loop {
137		let maybe_timeout = or_cancel(stream.next(), &shutdown).await;
138		let Ok(maybe_notification) = maybe_timeout else {
139			tracing::debug!("Shutting down notification task.");
140			break;
141		};
142		let Some(notification) = maybe_notification else {
143			continue;
144		};
145		let notification_message = format!(
146			"new notification: {}, {}, {}.",
147			notification.app_name, notification.title, notification.body
148		);
149		state.say(Priority::Important, notification_message).await;
150	}
151	Ok(())
152}
153#[tracing::instrument]
154async fn sigterm_signal_watcher(
155	token: CancellationToken,
156	state: Arc<ScreenReaderState>,
157) -> Result<(), OdiliaError> {
158	let timeout_duration = Duration::from_secs(5); //todo: perhaps take this from the configuration file at some point
159	let mut signals = Signals::new([Signal::Int])?;
160	signals.next()
161		.instrument(tracing::debug_span!("Watching for Ctrl+C"))
162		.await;
163	tracing::debug!("Asking all processes to stop.");
164	(*state.children_pids.lock().expect("Unable to lock mutex!"))
165		.iter_mut()
166		.try_for_each(Child::kill)
167		.expect("Able to kill child processes");
168	tracing::debug!("cancelling all tokens");
169	token.cancel();
170	tracing::debug!(?timeout_duration, "waiting for all tasks to finish");
171	Ok(())
172}

197#[tracing::instrument(ret)]
198async fn focused(
199	state_changed: CacheEvent<Focused>,
200	EventProp(relation_set): EventProp<RelationSet>,
201) -> impl TryIntoCommands {
202	//because the current command implementation doesn't allow for multiple speak commands without interrupting the previous utterance, this is more or less an accumulating buffer for that utterance
203	let mut utterance_buffer = String::new();
204	let item = state_changed.item;
205	//does this have a text or a name?
206	// in order for the borrow checker to not scream that we move ownership of item.text, therefore making item partially moved, we only take a reference here, because in truth the only thing that we need to know is if the string is empty, because the extending of the buffer will imply a clone anyway
207	let text = &item.text;
208	let name = item.name;
209	let description = item.description;
210	if text.is_empty() {
211		//then the label can either be the accessible name, the description, or the relations set, aka labeled by another object
212		//unfortunately, the or_else function of result doesn't accept async cloasures or cloasures with async blocks, so we can't use lazy loading here at the moment. The performance penalty is minimal however, because this should be in cache anyway
213		let label = if let Some(n) = name.as_deref() {
214			n.to_string()
215		} else if let Some(d) = description.as_deref() {
216			d.to_string()
217		//otherwise, if this is empty too, we try to use the relations set to find the element labeling this one
218		} else {
219			relation_set
220				.iter()
221				// we only need entries which contain the wanted relationship, only labeled by for now
222				.filter(|elem| elem.0 == RelationType::LabelledBy)
223				.cloned()
224				// we have to remove the first item of the entries, because it's constant now that we filtered by it
225				//furthermore, after doing this, we'd have something like Vec<Vec<Item>>, which is not what we need, we need something like Vec<Item>, so we do both the flattening of the structure and the mapping in one function call
226				.flat_map(|this| this.1)
227				// from a collection of items, to a collection of strings, in this case the text of those labels, because yes, technically there can be more than one
228				.map(|this| this.text)
229				// gather all that into a string, separated by newlines or spaces I think
230				.collect()
231		};
232		utterance_buffer += &label;
233	} else {
234		//then just append to the buffer and be done with it
235		utterance_buffer += text;
236	}
237	let role = item.role;
238	//there has to be a space between the accessible name of an object and its role, so insert it now
239	write!(utterance_buffer, " {}", role.name()).expect("Able to write to string");
240	Ok(vec![Focus(item.object).into(), Speak(utterance_buffer, Priority::Text).into()])
241}

342async fn async_main() -> Result<(), OdiliaError> {
343	let ex = &EXECUTOR;
344	let args = Args::from_cli_args()?;
345
346	//initialize the primary token for task cancelation
347	let token = CancellationToken::new();
348
349	//initializing configuration
350	let config = load_configuration(args.config)?;
351	//initialize logging, with the provided config
352	logging::init(&config)?;
353
354	tracing::info!(?config, "this configuration was used to prepair odilia");
355
356	// Make sure applications with dynamic accessibility support do expose their AT-SPI2 interfaces.
357	if let Err(e) = atspi::connection::set_session_accessibility(true)
358		.instrument(tracing::info_span!("setting accessibility enabled flag"))
359		.await
360	{
361		tracing::error!("Could not set AT-SPI2 IsEnabled property because: {}", e);
362	}
363	// this is the channel which handles all SSIP commands. If SSIP is not allowed to operate on a separate task, then waiting for the receiving message can block other long-running operations like structural navigation.
364	// Although in the future, this may possibly be resolved through a proper cache, I think it still makes sense to separate SSIP's IO operations to a separate task.
365	//  it is very important that this is *never* full, since it can cause deadlocking if the other task sending the request is working with zbus.
366	let (ssip_req_tx, ssip_req_rx) = bounded::<ssip_client_async::Request>(128);
367	let (ev_tx, ev_rx) = bounded::<Result<atspi::Event, atspi::AtspiError>>(10_000);
368	let (input_tx, input_rx) = bounded::<ScreenReaderEvent>(255);
369	// Initialize state
370	let state = Arc::new(ScreenReaderState::new(ssip_req_tx, config).await?);
371	let ssip = odilia_tts::create_ssip_client().await?;
372
373	if state.say(Priority::Message, "Welcome to Odilia!".to_string()).await {
374		tracing::debug!("Welcome message spoken.");
375	} else {
376		tracing::error!("Welcome message failed. Odilia is not able to continue in this state. Exiting now.");
377		state.close_speech().await;
378		exit(1);
379	}
380
381	// Register events
382	(
383		state.register_event::<object::StateChangedEvent>(),
384		state.register_event::<object::TextCaretMovedEvent>(),
385		state.register_event::<object::ChildrenChangedEvent>(),
386		state.register_event::<object::TextChangedEvent>(),
387		state.register_event::<document::LoadCompleteEvent>(),
388		state.add_cache_match_rule(),
389	)
390		.try_join()
391		.await?;
392
393	// load handlers
394	let handlers = Handlers::new(state.clone())
395		.command_listener(speak)
396		.command_listener(new_focused_item)
397		.command_listener(new_caret_pos)
398		.command_listener(set_state)
399		.atspi_listener(doc_loaded)
400		.atspi_listener(caret_moved)
401		.atspi_listener(focused)
402		.atspi_listener(state_set)
403		.input_listener(stop_speech)
404		.input_listener(change_mode)
405		.input_listener(structural_nav);
406
407	let ssip_event_receiver =
408		odilia_tts::handle_ssip_commands(ssip, ssip_req_rx, token.clone());
409	let notification_task = notifications_monitor(Arc::clone(&state), token.clone());
410	let mut stream = state.atspi.event_stream();
411	// There is a reason we are not reading from the event stream directly.
412	// This `MessageStream` can only store 64 events in its buffer.
413	// And, even if it could store more (it can via options), `zbus` specifically states that:
414	// You must ensure a MessageStream is continuously polled or you will experience hangs.
415	// So, we continually poll it here, then receive it on the other end.
416	// Additioanlly, since sending is not async, but simply errors when there is an issue, this will
417	// help us avoid hangs.
418	let token_clone = token.clone();
419	let event_send_task = async move {
420		std::pin::pin!(&mut stream);
421		loop {
422			let maybe = or_cancel(stream.next(), &token_clone).await;
423			let Ok(maybe_ev) = maybe else {
424				return;
425			};
426			if let Some(ev) = maybe_ev {
427				if let Err(e) = ev_tx.try_send(ev) {
428					tracing::error!(
429						"Error sending event across channel! {e:?}"
430					);
431				}
432			}
433		}
434	};
435	let atspi_handlers_task = handlers.clone().atspi_handler(ev_rx, token.clone());
436	let listener = odilia_input::setup_input_server()
437		.await
438		.expect("We should be able to set up input server; without it, Odilia cannot be controlled via input methods");
439	let input_task = odilia_input::sr_event_receiver(listener, input_tx, token.clone())
440		.for_each(|fut| {
441			ex.spawn(fut).detach();
442		});
443	let input_handler = handlers.input_handler(input_rx, token.clone());
444	let child = try_spawn_input_server(&state.config.input.method)?;
445	state.add_child_proc(child).expect("Able to add child to process!");
446
447	let joined_tasks = (
448		ssip_event_receiver,
449		notification_task,
450		atspi_handlers_task,
451		event_send_task,
452		input_task,
453		input_handler,
454	)
455		.join();
456	ex.spawn(sigterm_signal_watcher(token, Arc::clone(&state))).detach();
457	let _ = joined_tasks.await;
458	tracing::debug!("All listeners have stopped.");
459	tracing::debug!("Goodbye, Odilia!");
460
461	Ok(())
462}

464fn load_configuration(cli_overide: Option<PathBuf>) -> Result<ApplicationConfig, OdiliaError> {
465	// In order, do  a configuration file specified via cli, XDG_CONFIG_HOME, the usual location for system wide configuration(/etc/odilia/config.toml)
466	// If XDG_CONFIG_HOME based configuration wasn't found, create one by combining default values with the system provided ones, if available, for the user to alter, for the next run of odilia
467	//default configuration first, because that doesn't affect the priority outlined above
468	let xdg_dirs = xdg::BaseDirectories::with_prefix("odilia").expect(
469			"unable to find the odilia config directory according to the xdg dirs specification",
470		);
471
472	let config_path = xdg_dirs
473		.place_config_file("config.toml")
474		.expect("unable to place configuration file. Maybe your system is readonly?");
475
476	let mut config = config::Config::builder()
477		.add_source(config::Config::try_from(&ApplicationConfig::default())?)
478		// env vars
479		.add_source(config::Environment::with_prefix("ODILIA"))
480		//next, the configuration system wide, in /etc/odilia/config.toml
481		.add_source(config::File::with_name("/etc/odilia/config"))
482		//finally, the xdg configuration
483		.add_source(config::File::with_name(
484			config_path.to_str().expect("Valid UTF-8 path"),
485		));
486	if let Some(path) = cli_overide {
487		// if a path overide was given, use that
488		config = config.add_source(config::File::with_name(
489			path.to_str().expect("Valid UTF-8 path"),
490		));
491	}
492	Ok(config.build()?.try_deserialize()?)
493}
