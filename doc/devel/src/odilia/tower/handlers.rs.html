60	pub async fn command_handler(mut self, mut commands: Receiver<Command>) {
61		loop {
62			let maybe_cmd = commands.recv().await;
63			let Some(cmd) = maybe_cmd else {
64				tracing::error!("Error cmd: {maybe_cmd:?}");
65				continue;
66			};
67			// NOTE: Why not use join_all(...) ?
68			// Because this drives the futures concurrently, and we want ordered handlers.
69			// Otherwise, we cannot guarentee that the caching functions get run first.
70			// we could move caching to a separate, ordered system, then parallelize the other functions,
71			// if we determine this is a performance problem.
72			if let Err(e) = self.command.call(cmd).await {
73				tracing::error!("{e:?}");
74			}
75		}
76	}
77	#[tracing::instrument(skip_all)]
78	pub async fn atspi_handler<R>(mut self, mut events: R)
79	where
80		R: Stream<Item = Result<Event, AtspiError>> + Unpin,
81	{
82		std::pin::pin!(&mut events);
83		loop {
84			let maybe_ev = events.next().await;
85			let Some(Ok(ev)) = maybe_ev else {
86				tracing::error!("Error in processing {maybe_ev:?}");
87				continue;
88			};
89			if let Err(e) = self.atspi.call(ev).await {
90				tracing::error!("{e:?}");
91			}
92		}
93	}

108	pub fn command_listener<H, T, C, R>(mut self, handler: H) -> Self
109	where
110		H: Handler<T, Response = R> + Send + Clone + 'static,
111		<H as Handler<T>>::Future: Send,
112		C: CommandType + ChooserStatic<CommandDiscriminants> + Send + 'static,
113		Command: TryInto<C>,
114		OdiliaError: From<<Command as TryInto<C>>::Error>
115			+ From<<T as TryFromState<Arc<ScreenReaderState>, C>>::Error>
116			+ From<<T as AsyncTryFrom<(Arc<ScreenReaderState>, C)>>::Error>,
117		R: Into<Result<(), Error>> + Send + 'static,
118		T: TryFromState<Arc<ScreenReaderState>, C> + Send + 'static,
119		<T as TryFromState<Arc<ScreenReaderState>, C>>::Future: Send,
120		<T as TryFromState<Arc<ScreenReaderState>, C>>::Error: Send,
121		T: AsyncTryFrom<(Arc<ScreenReaderState>, C)>,
122		<T as AsyncTryFrom<(Arc<ScreenReaderState>, C)>>::Future: std::marker::Send,
123	{
124		let bs = handler
125			.into_service()
126			.map_response_into::<R, (), OdiliaError>()
127			.request_async_try_from()
128			.with_state(Arc::clone(&self.state))
129			.request_try_from()
130			.boxed_clone();
131		self.command.entry(C::identifier()).or_default().push(bs);
132		Self {
133			state: self.state,
134			atspi: self.atspi,
135			command: self.command,
136			input: self.input,
137		}
138	}
139	pub fn input_listener<H, T, R, E>(mut self, handler: H) -> Self
140	where
141		H: Handler<T, Response = R> + Send + Clone + 'static,
142		<H as Handler<T>>::Future: Send,
143		E: EventType + ChooserStatic<ScreenReaderEventDiscriminants> + Send + 'static,
144		ScreenReaderEvent: TryInto<E>,
145		OdiliaError: From<<ScreenReaderEvent as TryInto<E>>::Error>
146			+ From<<T as TryFromState<Arc<ScreenReaderState>, E>>::Error>,
147		R: TryIntoCommands + Send + 'static,
148		T: TryFromState<Arc<ScreenReaderState>, E> + Send + 'static,
149		<T as TryFromState<Arc<ScreenReaderState>, E>>::Future: Send,
150		<T as TryFromState<Arc<ScreenReaderState>, E>>::Error: Send,
151	{
152		let bs = handler
153			.into_service()
154			.map_response_try_into_command()
155			.request_async_try_from()
156			.with_state(Arc::clone(&self.state))
157			.request_try_from()
158			.iter_into(self.command.clone())
159			.map_result(|res: OdiliaResult<Vec<OdiliaResult<ResultList>>>| {
160				res?.into_iter() // Remove outer result
161					.flatten() // Flatten out first vec
162					.flatten() // Flatten out ResultList
163					.collect::<Result<(), OdiliaError>>()
164			})
165			.boxed_clone();
166		self.input.entry(E::identifier()).or_default().push(bs);
167		Self {
168			state: self.state,
169			atspi: self.atspi,
170			command: self.command,
171			input: self.input,
172		}
173	}
