75#[inline]
76async fn convert_to_new_type<
77	'a,
78	'b,
79	T: From<Proxy<'b>> + ProxyDefault,
80	U: ProxyImpl<'a> + ProxyDefault,
81>(
82	from: &U,
83) -> zbus::Result<T> {
84	let from = from.inner();
85
86	// first thing is first, we need to creat an accessible to query the interfaces.
87	let accessible = AccessibleProxy::builder(from.connection())
88		.destination(from.destination())?
89		.cache_properties(CacheProperties::No)
90		.path(from.path())?
91		.build()
92		.await?;
93	// if the interface we're trying to convert to is not available as an interface; this can be problematic because the interface we're passing in could potentially be different from what we're converting to.
94	let new_interface_name_ref: &InterfaceName = <T as ProxyDefault>::INTERFACE
95		.as_ref()
96		.ok_or(Error::InterfaceNotFound)?;
97	let new_interface_name: Interface = serde_plain::from_str(new_interface_name_ref)
98		.map_err(|_| Error::InterfaceNotFound)?;
99	if !accessible.get_interfaces().await?.contains(new_interface_name) {
100		return Err(Error::InterfaceNotFound);
101	}
102	// otherwise, make a new Proxy with the related type.
103	let path = from.path().to_owned();
104	let dest = from.destination().to_owned();
105
106	ProxyBuilder::<T>::new(from.connection())
107		.interface(
108			<T as ProxyDefault>::INTERFACE
109				.as_ref()
110				.ok_or(Error::InterfaceNotFound)?,
111		)?
112		.destination(dest)?
113		.cache_properties(CacheProperties::No)
114		.path(path)?
115		.build()
116		.await
117}
