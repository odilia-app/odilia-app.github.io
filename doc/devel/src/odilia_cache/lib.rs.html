57#[derive(Clone, Eq, PartialEq, Debug, Serialize, Deserialize)]
58#[repr(transparent)]
59pub struct RelationSet(Vec<(RelationType, Vec<Link>)>);
60
61impl RelationSet {
62	/// Turn the `Link` items into `CacheItem`s.
63	///
64	/// This function ignores [`Link::Unlinked`] variants.
65	#[must_use]
66	pub fn unchecked_into_cache_items(&self, c: &Cache) -> Vec<(RelationType, Vec<CacheItem>)> {
67		self.0.iter()
68			.map(|(rt, links)| {
69				(
70					*rt,
71					links.iter()
72						.filter_map(|link| match link {
73							Link::Unlinked(_) => None,
74							Link::Linked(nid) => c.get_id(*nid),
75						})
76						.collect::<Vec<CacheItem>>(),
77				)
78			})
79			.collect()
80	}
81	fn try_link_values(&mut self, cache: &Cache) -> Result<(), Vec<AccessiblePrimitive>> {
82		let relations_map = self.0.iter_mut();
83		let link_map = relations_map.flat_map(|(_rt, links)| links);
84		let mut unlinked = Vec::new();
85		for link in link_map {
86			if let Link::Unlinked(ap) = link {
87				if let Some(nid) = cache.id_lookup.get(ap) {
88					*link = Link::Linked(*nid);
89				} else {
90					unlinked.push(ap.clone());
91				}
92			}
93		}
94
95		if unlinked.is_empty() {
96			return Ok(());
97		}
98		Err(unlinked)
99	}
100}

150impl CacheItem {
151	/// Creates a `CacheItem` from an [`atspi::Event`] type.
152	/// # Errors
153	/// This can fail under three possible conditions:
154	///
155	/// 1. We are unable to convert information from the event into an [`AccessiblePrimitive`] hashmap key. This should never happen.
156	/// 2. We are unable to convert the [`AccessiblePrimitive`] to an [`atspi_proxies::accessible::AccessibleProxy`].
157	/// 3. The `accessible_to_cache_item` function fails for any reason. This also shouldn't happen.
158	#[tracing::instrument(level = "trace", skip_all, ret, err)]
159	pub async fn from_atspi_event<T: EventProperties, E: CacheSideEffect>(
160		event: &T,
161		external: &E,
162	) -> OdiliaResult<Self> {
163		let a11y_prim = AccessiblePrimitive::from_event(event);
164		external.lookup_external(&a11y_prim).await
165	}
166	/// Convert an [`atspi::LegacyCacheItem`] into a [`crate::CacheItem`].
167	/// This requires calls to `DBus`, which is quite expensive. Beware calling this too often.
168	/// # Errors
169	/// This function can fail under the following conditions:
170	///
171	/// 1. The [`atspi::CacheItem`] can not be turned into a [`crate::AccessiblePrimitive`]. This should never happen.
172	/// 2. The [`crate::AccessiblePrimitive`] can not be turned into a [`atspi_proxies::accessible::AccessibleProxy`]. This should never happen.
173	/// 3. Getting children from the `AccessibleProxy` fails. This should never happen.
174	///
175	/// The only time these can fail is if the item is removed on the application side before the conversion to `AccessibleProxy`.
176	#[tracing::instrument(level = "trace", skip(connection), ret, err)]
177	pub async fn from_atspi_legacy_cache_item(
178		atspi_cache_item: atspi_common::LegacyCacheItem,
179		connection: &zbus::Connection,
180	) -> OdiliaResult<Self> {
181		let acc = AccessiblePrimitive::from(atspi_cache_item.object.clone())
182			.into_accessible(connection)
183			.await?;
184		let index: i32 = acc.get_index_in_parent().await?;
185		let rs = acc.get_relation_set().await?.into();
186		let name = acc.name().await.map(|s| if s.is_empty() { None } else { Some(s) })?;
187		let desc =
188			acc.description()
189				.await
190				.map(|s| if s.is_empty() { None } else { Some(s) })?;
191		Ok(Self {
192			object: atspi_cache_item.object.into(),
193			app: atspi_cache_item.app.into(),
194			parent: atspi_cache_item.parent.into(),
195			index: index.try_into().ok(),
196			children_num: Some(atspi_cache_item.children.len()),
197			interfaces: atspi_cache_item.ifaces,
198			role: atspi_cache_item.role,
199			states: atspi_cache_item.states,
200			text: atspi_cache_item.name,
201			children: atspi_cache_item.children.into_iter().map_into::<_>().collect(),
202			relation_set: rs,
203			name,
204			description: desc,
205		})
206	}
207}

209/// An internal cache used within Odilia.
210///
211/// This contains (mostly) all accessibles in the entire accessibility tree, and
212/// they are referenced by their IDs. If you are having issues with incorrect or
213/// invalid accessibles trying to be accessed, this is code is probably the issue.
214#[derive(Clone)]
215pub struct Cache {
216	pub tree: ThreadSafeCache,
217	pub id_lookup: IdLookupTable,
218	pub connection: zbus::Connection,
219}

301// N.B.: we are using std RwLockes internally here, within the cache hashmap
302// entries. When adding async methods, take care not to hold these mutexes
303// across .await points.
304impl Cache {
305	/// create a new, fresh cache
306	#[must_use]
307	#[tracing::instrument(level = "debug", ret, skip_all)]
308	pub fn new(conn: zbus::Connection) -> Self {
309		Self {
310			tree: Arc::new(RwLock::new(Arena::with_capacity(10_000))),
311			id_lookup: Arc::new(DashMap::with_capacity_and_hasher(
312				10_000,
313				FxBuildHasher::default(),
314			)),
315			connection: conn,
316		}
317	}
318	/// Add an item via a reference instead of creating the reference.
319	#[tracing::instrument(level = "trace", ret, err)]
320	pub fn add(&self, mut cache_item: CacheItem) -> OdiliaResult<()> {
321		// Do not create new items when not necessary.
322		if let Some(self_id) = self.id_lookup.get(&cache_item.object) {
323			return Err(CacheError::DuplicateItem(*self_id).into());
324		}
325		let maybe_index = cache_item.index;
326		let key = cache_item.object.clone();
327		let parent_key = cache_item.parent.clone();
328		let mut cache = self.tree.write();
329		let unlinked_related_items = cache_item.relation_set.try_link_values(self);
330		let children = cache_item.children.clone();
331		let id = cache.new_node(cache_item);
332		self.id_lookup.insert(key, id);
333		// no need to connect to the rest of the graph, because it's the first item.
334		if self.id_lookup.len() == 1 {
335			return Ok(());
336		}
337		let Some(parent_id) = self.id_lookup.get(&parent_key) else {
338			return Err(CacheError::MoreData(Vec::from([parent_key])).into());
339		};
340		if let Err(unlinked_related_items) = unlinked_related_items {
341			return Err(CacheError::MoreData(unlinked_related_items).into());
342		}
343		let Some(sibling_index) = maybe_index else {
344			return Ok(());
345		};
346		if sibling_index == 0 {
347			parent_id.checked_prepend(id, &mut cache)?;
348		} else if sibling_index == parent_id.children(&cache).count() {
349			parent_id.checked_append(id, &mut cache)?;
350		} else {
351			match parent_id.children(&cache).nth(sibling_index) {
352				Some(left_sibling) => {
353					left_sibling.checked_insert_after(id, &mut cache)?;
354				}
355				// TODO: specific child?
356				None => return Err(CacheError::MoreData(children).into()),
357			}
358		}
359		Ok(())
360	}

362	/// Remove a single cache item. This function can not fail.
363	#[tracing::instrument(level = "trace", ret)]
364	pub fn remove(&self, id: &CacheKey) {
365		// if the item is not found in the lookup table, just return.
366		let Some((_key, node_id)) = self.id_lookup.remove(id) else {
367			tracing::warn!("Attempted to remove an item that doesn't exist: {id:?}");
368			return;
369		};
370		let mut cache = self.tree.write();
371		// Remove the item from the tree structure.
372		node_id.remove(&mut cache);
373	}

375	/// Get a single item from the cache by ID.
376	///
377	/// This will allow you to get the item without holding any locks to it,
378	/// at the cost of (1) a clone and (2) no guarantees that the data is kept up-to-date.
379	#[must_use]
380	#[tracing::instrument(level = "trace", ret)]
381	pub fn get_id(&self, id: NodeId) -> Option<CacheItem> {
382		let cache = self.tree.read();
383		let ref_item = cache.get(id)?;
384		// clone the reference into an owned value
385		Some(ref_item.get().to_owned())
386	}

388	/// Get a single item from the cache.
389	///
390	/// This will allow you to get the item without holding any locks to it,
391	/// at the cost of (1) a clone and (2) no guarantees that the data is kept up-to-date.
392	#[must_use]
393	#[tracing::instrument(level = "trace", ret)]
394	pub fn get(&self, id: &CacheKey) -> Option<CacheItem> {
395		let cache = self.tree.read();
396		let node_id = self.id_lookup.get(id)?;
397		let ref_item = cache.get(*node_id)?;
398		// clone the reference into an owned value
399		Some(ref_item.get().to_owned())
400	}

429	/// Edit a mutable `CacheItem`. Returns true if the update was successful.
430	///
431	/// Note: an exclusive lock for the given cache item will be placed for the
432	/// entire length of the passed function, so try to avoid any compute in it.
433	///
434	/// # Errors
435	///
436	/// An [`odilia_common::errors::OdiliaError::PoisoningError`] may be returned if a write lock can not be acquired on the `CacheItem` being modified.
437	#[tracing::instrument(level = "trace", skip(modify), ret, err)]
438	pub fn modify_item<F>(&self, id: &CacheKey, modify: F) -> OdiliaResult<bool>
439	where
440		F: FnOnce(&mut CacheItem),
441	{
442		let Some(node_id) = self.id_lookup.get(id) else {
443			tracing::warn!("The lookup table does not contain this item: {:?}", id);
444			return Ok(false);
445		};
446		let mut cache = self.tree.write();
447		let Some(node) = cache.get_mut(*node_id) else {
448			tracing::warn!("The tree cache does not contain this item: {:?}", node_id);
449			return Ok(false);
450		};
451		let cache_item = node.get_mut();
452		modify(cache_item);
453		Ok(true)
454	}

456	/// Get a single item from the cache (note that this copies some integers to a new struct).
457	/// If the `CacheItem` is not found, create one, add it to the cache, and return it.
458	/// # Errors
459	/// The function will return an error if:
460	/// 1. The `accessible` can not be turned into an `AccessiblePrimitive`. This should never happen, but is technically possible.
461	/// 2. The [`Self::add`] function fails.
462	/// 3. The [`accessible_to_cache_item`] function fails.
463	#[tracing::instrument(level = "debug", ret, err, skip(connection))]
464	pub async fn get_or_create(
465		&self,
466		key: &AccessiblePrimitive,
467		connection: &zbus::Connection,
468		cache: Arc<Cache>,
469	) -> OdiliaResult<CacheItem> {
470		// if the item already exists in the cache, return it
471		if let Some(cache_item) = self.get(key) {
472			return Ok(cache_item);
473		}
474		// otherwise, build a cache item
475		// NOTE: recursion CAN NOT be used here due to the size of some a11y trees and the size of
476		// async stack frames.
477		let mut stack: VecDeque<AccessiblePrimitive> = vec![key.to_owned()].into();
478		let mut first_ci = None;
479		while let Some(item) = stack.pop_front() {
480			let accessible = AccessibleProxy::builder(connection)
481				.destination(item.sender)?
482				.cache_properties(CacheProperties::No)
483				.path(item.id)?
484				.build()
485				.await?;
486			let start = std::time::Instant::now();
487			let cache_item = accessible_to_cache_item(&accessible).await?;
488			if first_ci.is_none() {
489				first_ci = Some(cache_item.clone());
490			}
491			let end = std::time::Instant::now();
492			let diff = end - start;
493			tracing::debug!("Time to create cache item: {:?}", diff);
494			if let Err(OdiliaError::Cache(CacheError::MoreData(items))) =
495				self.add(cache_item)
496			{
497				stack.extend(items);
498			}
499		}
500		// return that same cache item
501		// SAFETY: this is okay because we always have one item in the stack, and guarantee that any
502		// errors along the way to setting the value cause an early return.
503		Ok(first_ci.expect("Able to extract first CacheItem!"))
504	}

506	/// Clears the cache completely.
507	pub fn clear(&self) {
508		self.tree.write().clear();
509		self.id_lookup.clear();
510	}

513/// Convert an [`atspi_proxies::accessible::AccessibleProxy`] into a [`crate::CacheItem`].
514/// This runs a bunch of long-awaiting code and can take quite some time; use this sparingly.
515/// This takes most properties and some function calls through the `AccessibleProxy` structure and generates a new `CacheItem`, which will be written to cache before being sent back.
516///
517/// # Errors
518///
519/// Will return an `Err(_)` variant when:
520///
521/// 1. The `cache` parameter does not reference an active cache once the `Weak` is upgraded to an `Option<Arc<_>>`.
522/// 2. Any of the function calls on the `accessible` fail.
523/// 3. Any `(String, OwnedObjectPath) -> AccessiblePrimitive` conversions fail. This *should* never happen, but technically it is possible.
524#[tracing::instrument(level = "trace", ret, err)]
525pub async fn accessible_to_cache_item(accessible: &AccessibleProxy<'_>) -> OdiliaResult<CacheItem> {
526	let (app, parent, index, children_num, interfaces, role, states, children) = tokio::try_join!(
527		accessible.get_application(),
528		accessible.parent(),
529		accessible.get_index_in_parent(),
530		accessible.child_count(),
531		accessible.get_interfaces(),
532		accessible.get_role(),
533		accessible.get_state(),
534		accessible.get_children(),
535	)?;
536	let rs = accessible.get_relation_set().await?.into();
537	let name = accessible
538		.name()
539		.await
540		.map(|s| if s.is_empty() { None } else { Some(s) })?;
541	let desc = accessible
542		.description()
543		.await
544		.map(|s| if s.is_empty() { None } else { Some(s) })?;
545	// if it implements the Text interface
546	let text = match accessible.to_text().await {
547		// get *all* the text
548		Ok(text_iface) => text_iface.get_all_text().await,
549		// otherwise, use the name instaed
550		Err(_) => Ok(accessible.name().await?),
551	}?;
552	Ok(CacheItem {
553		object: accessible.try_into()?,
554		app: app.into(),
555		parent: parent.into(),
556		index: index.try_into().ok(),
557		children_num: children_num.try_into().ok(),
558		interfaces,
559		role,
560		states,
561		text,
562		children: children.into_iter().map_into().collect(),
563		relation_set: rs,
564		name,
565		description: desc,
566	})
