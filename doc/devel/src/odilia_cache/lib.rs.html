82impl CacheItem {
83	/// Return a *reference* to a parent. This is *much* cheaper than getting the parent element outright via [`Self::parent`].
84	/// # Errors
85	/// This method will return a [`CacheError::NoItem`] if no item is found within the cache.
86	#[tracing::instrument(level = "trace", ret, err)]
87	pub fn parent_ref(&mut self) -> OdiliaResult<Arc<std::sync::RwLock<CacheItem>>> {
88		let parent_ref = Weak::upgrade(&self.parent.item);
89		if let Some(p) = parent_ref {
90			Ok(p)
91		} else {
92			let cache = strong_cache(&self.cache)?;
93			let arc_mut_parent = cache
94				.get_ref(&self.parent.key.clone())
95				.ok_or(CacheError::NoItem)?;
96			self.parent.item = Arc::downgrade(&arc_mut_parent);
97			Ok(arc_mut_parent)
98		}
99	}
100	/// Creates a `CacheItem` from an [`atspi::Event`] type.
101	/// # Errors
102	/// This can fail under three possible conditions:
103	///
104	/// 1. We are unable to convert information from the event into an [`AccessiblePrimitive`] hashmap key. This should never happen.
105	/// 2. We are unable to convert the [`AccessiblePrimitive`] to an [`atspi_proxies::accessible::AccessibleProxy`].
106	/// 3. The `accessible_to_cache_item` function fails for any reason. This also shouldn't happen.
107	#[tracing::instrument(level = "trace", skip_all, ret, err)]
108	pub async fn from_atspi_event<T: EventProperties>(
109		event: &T,
110		cache: Arc<Cache>,
111		connection: &zbus::Connection,
112	) -> OdiliaResult<Self> {
113		let a11y_prim = AccessiblePrimitive::from_event(event);
114		accessible_to_cache_item(
115			&a11y_prim.into_accessible(connection).await?,
116			Arc::downgrade(&cache),
117		)
118		.await
119	}
120	/// Convert an [`atspi::CacheItem`] into a [`crate::CacheItem`].
121	/// This requires calls to `DBus`, which is quite expensive. Beware calling this too often.
122	/// # Errors
123	/// This function can fail under the following conditions:
124	///
125	/// 1. The [`atspi::CacheItem`] can not be turned into a [`crate::AccessiblePrimitive`]. This should never happen.
126	/// 2. The [`crate::AccessiblePrimitive`] can not be turned into a [`atspi_proxies::accessible::AccessibleProxy`]. This should never happen.
127	/// 3. Getting children from the `AccessibleProxy` fails. This should never happen.
128	///
129	/// The only time these can fail is if the item is removed on the application side before the conversion to `AccessibleProxy`.
130	#[tracing::instrument(level = "trace", skip_all, ret, err)]
131	pub async fn from_atspi_cache_item(
132		atspi_cache_item: atspi_common::CacheItem,
133		cache: Weak<Cache>,
134		connection: &zbus::Connection,
135	) -> OdiliaResult<Self> {
136		let children: Vec<CacheRef> =
137			AccessiblePrimitive::from(atspi_cache_item.object.clone())
138				.into_accessible(connection)
139				.await?
140				.get_children()
141				.await?
142				.into_iter()
143				.map(|child_object_pair| CacheRef::new(child_object_pair.into()))
144				.collect();
145		Ok(Self {
146			object: atspi_cache_item.object.into(),
147			app: atspi_cache_item.app.into(),
148			parent: CacheRef::new(atspi_cache_item.parent.into()),
149			index: atspi_cache_item.index.try_into().ok(),
150			children_num: atspi_cache_item.children.try_into().ok(),
151			interfaces: atspi_cache_item.ifaces,
152			role: atspi_cache_item.role,
153			states: atspi_cache_item.states,
154			text: atspi_cache_item.name,
155			cache,
156			children,
157		})
158	}
159	/// Convert an [`atspi::LegacyCacheItem`] into a [`crate::CacheItem`].
160	/// This requires calls to `DBus`, which is quite expensive. Beware calling this too often.
161	/// # Errors
162	/// This function can fail under the following conditions:
163	///
164	/// 1. The [`atspi::CacheItem`] can not be turned into a [`crate::AccessiblePrimitive`]. This should never happen.
165	/// 2. The [`crate::AccessiblePrimitive`] can not be turned into a [`atspi_proxies::accessible::AccessibleProxy`]. This should never happen.
166	/// 3. Getting children from the `AccessibleProxy` fails. This should never happen.
167	///
168	/// The only time these can fail is if the item is removed on the application side before the conversion to `AccessibleProxy`.
169	#[tracing::instrument(level = "trace", skip_all, ret, err)]
170	pub async fn from_atspi_legacy_cache_item(
171		atspi_cache_item: atspi_common::LegacyCacheItem,
172		cache: Weak<Cache>,
173		connection: &zbus::Connection,
174	) -> OdiliaResult<Self> {
175		let index: i32 = AccessiblePrimitive::from(atspi_cache_item.object.clone())
176			.into_accessible(connection)
177			.await?
178			.get_index_in_parent()
179			.await?;
180		Ok(Self {
181			object: atspi_cache_item.object.into(),
182			app: atspi_cache_item.app.into(),
183			parent: CacheRef::new(atspi_cache_item.parent.into()),
184			index: index.try_into().ok(),
185			children_num: Some(atspi_cache_item.children.len()),
186			interfaces: atspi_cache_item.ifaces,
187			role: atspi_cache_item.role,
188			states: atspi_cache_item.states,
189			text: atspi_cache_item.name,
190			cache,
191			children: atspi_cache_item
192				.children
193				.into_iter()
194				.map(|or| CacheRef::new(or.into()))
195				.collect(),
196		})
197	}
198	// Same as [`AccessibleProxy::get_children`], just offered as a non-async version.
199	/// Get a `Vec` of children with the same type as `Self`.
200	/// # Errors
201	/// 1. Will return an `Err` variant if `self.cache` does not reference an active cache. This should never happen, but it is technically possible.
202	/// 2. Any children keys' values are not found in the cache itself.
203	#[tracing::instrument(level = "trace", skip_all, ret, err)]
204	pub fn get_children(&self) -> OdiliaResult<Vec<Self>> {
205		let derefed_cache: Arc<Cache> = strong_cache(&self.cache)?;
206		let children = self
207			.children
208			.iter()
209			.map(|child_ref| {
210				child_ref
211					.clone_inner()
212					.or_else(|| derefed_cache.get(&child_ref.key))
213					.ok_or(CacheError::NoItem)
214			})
215			.collect::<Result<Vec<_>, _>>()?;
216		Ok(children)
217	}
218}

220/// A composition of an accessible ID and (possibly) a reference
221/// to its `CacheItem`, if the item has not been dropped from the cache yet.
222/// TODO if desirable, we could make one direction strong references (e.g. have
223/// the parent be an Arc, or have the children be Arcs). Might even be possible to have both.
224/// BUT - is it even desirable to keep an item pinned in an Arc from its
225/// relatives after it has been removed from the cache?
226#[derive(Debug, Clone, Deserialize, Serialize)]
227pub struct CacheRef {
228	pub key: CacheKey,
229	#[serde(skip)]
230	item: Weak<RwLock<CacheItem>>,
231}

233impl CacheRef {
234	#[must_use]
235	#[tracing::instrument(level = "trace", skip_all, ret)]
236	pub fn new(key: AccessiblePrimitive) -> Self {
237		Self { key, item: Weak::new() }
238	}
239
240	#[must_use]
241	pub fn clone_inner(&self) -> Option<CacheItem> {
242		Some(self.item.upgrade().as_ref()?.read().ok()?.clone())
243	}
244}

317	/// - The items mentioned are not in the cache.
318	pub async fn get_relation_set(
319		&self,
320	) -> Result<Vec<(RelationType, Vec<Self>)>, OdiliaError> {
321		let cache = strong_cache(&self.cache)?;
322		let ipc_rs = as_accessible(self).await?.get_relation_set().await?;
323		let mut relations = Vec::new();
324		for (relation, object_pairs) in ipc_rs {
325			let mut cache_keys = Vec::new();
326			for object_pair in object_pairs {
327				let cached = cache.get_ipc(&object_pair.into()).await?;
328				cache_keys.push(cached);
329			}
330			relations.push((relation, cache_keys));
331		}
332		Ok(relations)
333	}

338		self.get_children()?

342	}

447	pub async fn get_string_at_offset(
448		&self,
449		offset: usize,
450		granularity: Granularity,
451	) -> Result<(String, usize, usize), OdiliaError> {
452		// optimisations that don't call out to DBus.
453		if granularity == Granularity::Paragraph {
454			return Ok((self.text.clone(), 0, self.text.len()));
455		} else if granularity == Granularity::Char {
456			let range = offset..=offset;
457			return Ok((
458				self.text
459					.get(range)
460					.ok_or(CacheError::TextBoundsError)?
461					.to_string(),
462				offset,
463				offset + 1,
464			));
465		} else if granularity == Granularity::Word {
466			return Ok(self
467				.text
468				// [char]
469				.split_whitespace()
470				// [(word, start, end)]
471				.filter_map(|word| {
472					let start = self
473						.text
474						// [(idx, char)]
475						.char_indices()
476						// [(idx, char)]: uses pointer arithmatic to find start index
477						.find(|&(idx, _)| {
478							idx == word.as_ptr() as usize
479								- self.text.as_ptr() as usize
480						})
481						// [idx]
482						.map(|(idx, _)| idx)?;
483					// calculate based on start
484					let end = start + word.len();
485					// if the offset if within bounds
486					if offset >= start && offset <= end {
487						Some((word.to_string(), start, end))
488					} else {
489						None
490					}
491				})
492				// get "all" words that match; there should be only one result
493				.collect::<Vec<_>>()
494				.first()
495				// if there's no matching word (out of bounds)
496				.ok_or_else(|| OdiliaError::Generic("Out of bounds".to_string()))?
497				// clone the reference into a value
498				.clone());
499		}
500		// any other variations, in particular, Granularity::Line, will need to call out to DBus. It's just too complex to calculate, get updates for bounding boxes, etc.
501		// this variation does NOT get a semantic line. It gets a visual line.
502		let dbus_version = as_text(self)
503			.await?
504			.get_string_at_offset(offset.try_into()?, granularity)
505			.await?;
506		Ok((dbus_version.0, dbus_version.1.try_into()?, dbus_version.2.try_into()?))
507	}

597/// An internal cache used within Odilia.
598///
599/// This contains (mostly) all accessibles in the entire accessibility tree, and
600/// they are referenced by their IDs. If you are having issues with incorrect or
601/// invalid accessibles trying to be accessed, this is code is probably the issue.
602#[derive(Clone)]
603pub struct Cache {
604	pub by_id: ThreadSafeCache,
605	pub connection: zbus::Connection,
606}
607
608impl std::fmt::Debug for Cache {
609	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
610		f.write_str(&format!("Cache {{ by_id: ...{} items..., .. }}", self.by_id.len()))
611	}
612}

614pub trait CacheExt {
615	/// Get a single item from the cache. This will also get the information from `DBus` if it does not
616	/// exist in the cache.
617	fn get_ipc(
618		&self,
619		id: &CacheKey,
620	) -> impl Future<Output = Result<CacheItem, OdiliaError>> + Send;
621	fn item_from_event<T: EventProperties + Sync>(
622		&self,
623		ev: &T,
624	) -> impl Future<Output = OdiliaResult<CacheItem>> + Send;
625}
626
627impl CacheExt for Arc<Cache> {
628	#[tracing::instrument(level = "trace", ret)]
629	async fn get_ipc(&self, id: &CacheKey) -> Result<CacheItem, OdiliaError> {
630		if let Some(ci) = self.get(id) {
631			return Ok(ci);
632		}
633		let acc = id.clone().into_accessible(&self.connection).await?;
634		accessible_to_cache_item(&acc, Arc::downgrade(self)).await
635	}
636	async fn item_from_event<T: EventProperties + Sync>(
637		&self,
638		ev: &T,
639	) -> OdiliaResult<CacheItem> {
640		let a11y_prim = AccessiblePrimitive::from_event(ev);
641		accessible_to_cache_item(
642			&a11y_prim.into_accessible(&self.connection).await?,
643			Arc::downgrade(self),
644		)
645		.await
646	}
647}

649// N.B.: we are using std RwLockes internally here, within the cache hashmap
650// entries. When adding async methods, take care not to hold these mutexes
651// across .await points.
652impl Cache {
653	/// create a new, fresh cache
654	#[must_use]
655	#[tracing::instrument(level = "debug", ret, skip_all)]
656	pub fn new(conn: zbus::Connection) -> Self {
657		Self {
658			by_id: Arc::new(DashMap::with_capacity_and_hasher(
659				10_000,
660				FxBuildHasher::default(),
661			)),
662			connection: conn,
663		}
664	}
665	/// add a single new item to the cache. Note that this will empty the bucket
666	/// before inserting the `CacheItem` into the cache (this is so there is
667	/// never two items with the same ID stored in the cache at the same time).
668	/// # Errors
669	/// Fails if the internal call to [`Self::add_ref`] fails.
670	#[tracing::instrument(level = "trace", ret, err)]
671	pub fn add(&self, cache_item: CacheItem) -> OdiliaResult<()> {
672		let id = cache_item.object.clone();
673		self.add_ref(id, &Arc::new(RwLock::new(cache_item)))
674	}
675
676	/// Add an item via a reference instead of creating the reference.
677	/// # Errors
678	/// Can error if [`Cache::populate_references`] errors. The insertion is guarenteed to succeed.
679	#[tracing::instrument(level = "trace", ret, err)]
680	pub fn add_ref(
681		&self,
682		id: CacheKey,
683		cache_item: &Arc<RwLock<CacheItem>>,
684	) -> OdiliaResult<()> {
685		self.by_id.insert(id, Arc::clone(cache_item));
686		Self::populate_references(&self.by_id, cache_item)
687	}
688
689	/// Remove a single cache item. This function can not fail.
690	#[tracing::instrument(level = "trace", ret)]
691	pub fn remove(&self, id: &CacheKey) {
692		self.by_id.remove(id);
693	}
694
695	/// Get a single item from the cache, this only gets a reference to an item, not the item itself.
696	/// You will need to either get a read or a write lock on any item returned from this function.
697	/// It also may return `None` if a value is not matched to the key.
698	#[must_use]
699	#[tracing::instrument(level = "trace", ret)]
700	pub fn get_ref(&self, id: &CacheKey) -> Option<Arc<RwLock<CacheItem>>> {
701		self.by_id.get(id).as_deref().cloned()
702	}
703
704	/// Get a single item from the cache.
705	///
706	/// This will allow you to get the item without holding any locks to it,
707	/// at the cost of (1) a clone and (2) no guarantees that the data is kept up-to-date.
708	#[must_use]
709	#[tracing::instrument(level = "trace", ret)]
710	pub fn get(&self, id: &CacheKey) -> Option<CacheItem> {
711		Some(self.by_id.get(id).as_deref()?.read().ok()?.clone())
712	}
713
714	/// get a many items from the cache; this only creates one read handle (note that this will copy all data you would like to access)
715	#[must_use]
716	#[tracing::instrument(level = "trace", ret)]
717	pub fn get_all(&self, ids: &[CacheKey]) -> Vec<Option<CacheItem>> {
718		ids.iter().map(|id| self.get(id)).collect()
719	}
720
721	/// Bulk add many items to the cache; only one accessible should ever be
722	/// associated with an id.
723	/// # Errors
724	/// An `Err(_)` variant may be returned if the [`Cache::populate_references`] function fails.
725	#[tracing::instrument(level = "trace", ret, err)]
726	pub fn add_all(&self, cache_items: Vec<CacheItem>) -> OdiliaResult<()> {
727		cache_items
728			.into_iter()
729			.map(|cache_item| {
730				let id = cache_item.object.clone();
731				let arc = Arc::new(RwLock::new(cache_item));
732				self.by_id.insert(id, Arc::clone(&arc));
733				arc
734			})
735			.collect::<Vec<_>>() // Insert all items before populating
736			.into_iter()
737			.try_for_each(|item| Self::populate_references(&self.by_id, &item))
738	}
739	/// Bulk remove all ids in the cache; this only refreshes the cache after removing all items.
740	#[tracing::instrument(level = "trace", ret)]
741	pub fn remove_all(&self, ids: &Vec<CacheKey>) {
742		for id in ids {
743			self.by_id.remove(id);
744		}
745	}
746
747	/// Edit a mutable `CacheItem`. Returns true if the update was successful.
748	///
749	/// Note: an exclusive lock for the given cache item will be placed for the
750	/// entire length of the passed function, so try to avoid any compute in it.
751	///
752	/// # Errors
753	///
754	/// An [`odilia_common::errors::OdiliaError::PoisoningError`] may be returned if a write lock can not be acquired on the `CacheItem` being modified.
755	#[tracing::instrument(level = "trace", skip(modify), ret, err)]
756	pub fn modify_item<F>(&self, id: &CacheKey, modify: F) -> OdiliaResult<bool>
757	where
758		F: FnOnce(&mut CacheItem),
759	{
760		// I wonder if `get_mut` vs `get` makes any difference here? I suppose
761		// it will just rely on the dashmap write access vs mutex lock access.
762		// Let's default to the fairness of the mutex.
763		let entry = if let Some(i) = self.by_id.get(id) {
764			// Drop the dashmap reference immediately, at the expense of an Arc clone.
765			(*i).clone()
766		} else {
767			tracing::trace!("The cache does not contain the requested item: {:?}", id);
768			return Ok(false);
769		};
770		let mut cache_item = entry.write()?;
771		modify(&mut cache_item);
772		Ok(true)
773	}
774
775	/// Get a single item from the cache (note that this copies some integers to a new struct).
776	/// If the `CacheItem` is not found, create one, add it to the cache, and return it.
777	/// # Errors
778	/// The function will return an error if:
779	/// 1. The `accessible` can not be turned into an `AccessiblePrimitive`. This should never happen, but is technically possible.
780	/// 2. The [`Self::add`] function fails.
781	/// 3. The [`accessible_to_cache_item`] function fails.
782	#[tracing::instrument(level = "debug", ret, err)]
783	pub async fn get_or_create(
784		&self,
785		accessible: &AccessibleProxy<'_>,
786		cache: Arc<Cache>,
787	) -> OdiliaResult<CacheItem> {
788		// if the item already exists in the cache, return it
789		let primitive = accessible.try_into()?;
790		if let Some(cache_item) = self.get(&primitive) {
791			return Ok(cache_item);
792		}
793		// otherwise, build a cache item
794		let start = std::time::Instant::now();
795		let cache_item =
796			accessible_to_cache_item(accessible, Arc::downgrade(&cache)).await?;
797		let end = std::time::Instant::now();
798		let diff = end - start;
799		tracing::debug!("Time to create cache item: {:?}", diff);
800		// add a clone of it to the cache
801		self.add(cache_item.clone())?;
802		// return that same cache item
803		Ok(cache_item)
804	}
805
806	/// Populate children and parent references given a cache and an `Arc<RwLock<CacheItem>>`.
807	/// This will unlock the `RwLock<_>`, update the references for children and parents, then go to the parent and children and do the same: update the parent for the children, then update the children referneces for the parent.
808	/// # Errors
809	/// If any references, either the ones passed in through the `item_ref` parameter, any children references, or the parent reference are unable to be unlocked, an `Err(_)` variant will be returned.
810	/// Technically it can also fail if the index of the `item_ref` in its parent exceeds `usize` on the given platform, but this is highly improbable.
811	#[tracing::instrument(level = "trace", ret, err)]
812	pub fn populate_references(
813		cache: &ThreadSafeCache,
814		item_ref: &Arc<RwLock<CacheItem>>,
815	) -> Result<(), OdiliaError> {
816		let item_wk_ref = Arc::downgrade(item_ref);
817		let mut item = item_ref.write()?;
818		let item_key = item.object.clone();
819
820		let parent_key = item.parent.key.clone();
821		let parent_ref_opt = cache.get(&parent_key);
822
823		// Update this item's children references
824		for child_ref in &mut item.children {
825			if let Some(child_arc) = cache.get(&child_ref.key).as_ref() {
826				child_ref.item = Arc::downgrade(child_arc);
827				child_arc.write()?.parent.item = Weak::clone(&item_wk_ref);
828			}
829		}
830
831		// TODO: Should there be errors for the non let bindings?
832		if let Some(parent_ref) = parent_ref_opt {
833			item.parent.item = Arc::downgrade(&parent_ref);
834			if let Some(ix) = item.index {
835				if let Some(cache_ref) = parent_ref
836					.write()?
837					.children
838					.get_mut(ix)
839					.filter(|i| i.key == item_key)
840				{
841					cache_ref.item = Weak::clone(&item_wk_ref);
842				}
843			}
844		}
845		Ok(())
846	}
847}

849/// Convert an [`atspi_proxies::accessible::AccessibleProxy`] into a [`crate::CacheItem`].
850/// This runs a bunch of long-awaiting code and can take quite some time; use this sparingly.
851/// This takes most properties and some function calls through the `AccessibleProxy` structure and generates a new `CacheItem`, which will be written to cache before being sent back.
852///
853/// # Errors
854///
855/// Will return an `Err(_)` variant when:
856///
857/// 1. The `cache` parameter does not reference an active cache once the `Weak` is upgraded to an `Option<Arc<_>>`.
858/// 2. Any of the function calls on the `accessible` fail.
859/// 3. Any `(String, OwnedObjectPath) -> AccessiblePrimitive` conversions fail. This *should* never happen, but technically it is possible.
860#[tracing::instrument(level = "trace", ret, err)]
861pub async fn accessible_to_cache_item(
862	accessible: &AccessibleProxy<'_>,
863	cache: Weak<Cache>,
864) -> OdiliaResult<CacheItem> {
865	let (app, parent, index, children_num, interfaces, role, states, children) = tokio::try_join!(
866		accessible.get_application(),
867		accessible.parent(),
868		accessible.get_index_in_parent(),
869		accessible.child_count(),
870		accessible.get_interfaces(),
871		accessible.get_role(),
872		accessible.get_state(),
873		accessible.get_children(),
874	)?;
875	// if it implements the Text interface
876	let text = match accessible.to_text().await {
877		// get *all* the text
878		Ok(text_iface) => text_iface.get_all_text().await,
879		// otherwise, use the name instaed
880		Err(_) => Ok(accessible.name().await?),
881	}?;
882	Ok(CacheItem {
883		object: accessible.try_into()?,
884		app: app.into(),
885		parent: CacheRef::new(parent.into()),
886		index: index.try_into().ok(),
887		children_num: children_num.try_into().ok(),
888		interfaces,
889		role,
890		states,
891		text,
892		children: children.into_iter().map(|k| CacheRef::new(k.into())).collect(),
893		cache,
894	})
