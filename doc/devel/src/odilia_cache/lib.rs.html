45async fn or_cancel<F>(f: F, token: &CancellationToken) -> Result<F::Output, std::io::Error>
46where
47	F: std::future::Future,
48{
49	token.cancelled()
50		.map(|()| Err(std::io::ErrorKind::TimedOut.into()))
51		.or(f.map(Ok))
52		.await
53}

86impl CacheActor {
87	/// Request the [`CacheRequest`] from the cache.
88	///
89	/// # Errors
90	///
91	/// The possible errors are outlined in [`CacheError`].
92	///
93	/// # Panics
94	///
95	/// If the receiver for the response is dropped.
96	pub async fn request(&self, req: CacheRequest) -> Result<CacheResponse, OdiliaError> {
97		let (reply, recv) = bounded(1);
98		self.send
99			.send((req, reply))
100			.await
101			.expect("Unable to send a message on the channel; this is bad!");
102		recv.recv().await.expect("Unable to get response from channel!")
103	}
104}

110pub async fn cache_handler_task<D: CacheDriver + Send>(
111	recv: ActorRecv,
112	shutdown: CancellationToken,
113	mut cache: Cache<D>,
114) {
115	loop {
116		let Ok(maybe_request) = or_cancel(recv.recv(), &shutdown).await else {
117			tracing::info!("Shutting down cache service due to cancellation token!");
118			break;
119		};
120		tracing::trace!("MR: {maybe_request:?}");
121		let (request, response) = match maybe_request {
122			Err(e) => {
123				tracing::error!(error = %e, "Error receiving cache request");
124				continue;
125			}
126			Ok(req) => req,
127		};
128		tracing::trace!("REQ: {request:?}");
129		let maybe_cache_item = cache.request(request).await;
130		match response.send(maybe_cache_item).await {
131			Ok(()) => tracing::trace!("Successful sending cache item back!"),
132			Err(e) => {
133				tracing::error!(error = %e, "Error sending cache item back to requester!");
134			}
135		}
136	}
137}

152impl NewCache {
153	fn has_app(&self, key: &CacheKey) -> bool {
154		self.0.keys().any(|k| *k == key.sender)
155	}
156	fn get(&self, key: &CacheKey) -> Option<&CacheItem> {
157		self.0.get(&key.sender)?.get(&key.id)

159	fn get_mut(&mut self, key: &CacheKey) -> Option<&mut CacheItem> {
160		self.0.get_mut(&key.sender)?.get_mut(&key.id)

163		self.0.entry(key.sender)
164			.or_default()
165			// Above we go from Map<Map<...>> to Map<...>
166			.entry(key.id)
167			.or_insert(cache_item);
168	}
169	fn remove(&mut self, key: &CacheKey) -> Option<CacheItem> {
170		self.0.get_mut(&key.sender)?.remove(&key.id)
171	}
172	fn len(&self) -> usize {
173		self.0.values().map(HashMap::len).sum()
174	}

240/// A method of performing I/O side-effects outside the cache itself.
241/// This is made an explicit trait such that we can either:
242///
243/// 1. Call out to `DBus` (production), or
244/// 2. Use a fixed set of items (testing).
245/// 3. Panic when called.
246///
247/// Feel free to implement your own at your convenience.
248pub trait CacheDriver {
249	/// Lookup a given [`CacheKey`] that was not found in the cache..
250	fn lookup_external(
251		&self,
252		key: &CacheKey,
253	) -> impl Future<Output = OdiliaResult<CacheItem>> + Send;
254	/// Bulk query an application based on the [`CacheKey.sender`] field.
255	fn lookup_bulk(
256		&self,
257		key: &CacheKey,
258	) -> impl Future<Output = OdiliaResult<Vec<CacheItem>>> + Send;
259	/// A seperate method from [`lookup_external`] for getting relation sets.
260	/// This is separate because it can have rather large results and should only be called when
261	/// absolutely necessary.
262	fn lookup_relations(
263		&self,
264		key: &CacheKey,
265		ty: RelationType,
266	) -> impl Future<Output = OdiliaResult<Vec<CacheKey>>> + Send;
267	/// A separate method from [`lookup_external`] for converting an [`atspi::CacheItem`] into a
268	/// [`CacheItem`].
269	/// This will call out to `DBus` for the remaining details.
270	fn lookup_from_cache_item(
271		&self,
272		cache_item: atspi::CacheItem,
273	) -> impl Future<Output = OdiliaResult<CacheItem>> + Send;
274	/// A separate method from [`lookup_external`] for converting an [`atspi::LegacyCacheItem`] into a
275	/// [`CacheItem`].
276	/// This will call out to `DBus` for the remaining details.
277	fn lookup_from_legacy_cache_item(
278		&self,
279		cache_item: atspi::LegacyCacheItem,
280	) -> impl Future<Output = OdiliaResult<CacheItem>> + Send;
281}

295	async fn lookup_bulk(&self, key: &CacheKey) -> OdiliaResult<Vec<CacheItem>> {
296		let cache = CacheProxy::builder(self)
297			.destination(key.sender.clone())?
298			.cache_properties(CacheProperties::No)
299			// the fixed path to the cache object
300			.path("/org/a11y/atspi/cache")?
301			.build()
302			.await?;
303		tracing::error!("NAME: {}", cache.inner().destination());
304		tracing::error!("TRY TO GET CACHE ITEM!");

306		tracing::error!("ITEMS: {maybe_items:?}");
307		let futs: Vec<_> = maybe_items?

315		&self,

317		ty: RelationType,
318	) -> OdiliaResult<Vec<CacheKey>> {

320			.destination(key.sender.clone())?

339		&self,

341	) -> OdiliaResult<CacheItem> {
342		let key: CacheKey = cache_item.object.clone().into();
343		tracing::trace!("DST: {key:?}");
344		let accessible = AccessibleProxy::builder(self)
345			.destination(key.sender.clone())?
346			.cache_properties(CacheProperties::No)
347			.path(key.id.clone())?
348			.build()
349			.await?;
350		let (description, help_text, text, children) = (

354			accessible

372			.try_join()

374		Ok(CacheItem {

447impl<D: CacheDriver + Send> Cache<D> {
448	async fn handle_event(&mut self, ev: Event) -> Result<CacheItem, OdiliaError> {
449		ev.handle_event(self).await
450	}
451	async fn request(&mut self, req: CacheRequest) -> Result<CacheResponse, OdiliaError> {
452		tracing::trace!("REQ: {req:?}");
453		match req {
454			CacheRequest::Item(ref key) => {
455				self.get_or_create(key)
456					.map_ok(|ci| CacheResponse::Item(Item(ci)))
457					.await
458			}
459			CacheRequest::Parent(ref key) => {
460				self.get_or_create(key)
461					.map_ok(|ci| CacheResponse::Parent(Parent(ci)))
462					.await
463			}
464			CacheRequest::Children(ref key) => {
465				let children_vec = self.get_or_create(key).await?.children;
466				let children = self.get_or_create_all(children_vec).await?;
467				Ok(CacheResponse::Children(Children(children)))
468			}
469			CacheRequest::Relation(ref key, ty) => {
470				let rel_ids = self.driver.lookup_relations(key, ty).await?;
471				let rels = self.get_or_create_all(rel_ids).await?;
472				Ok(CacheResponse::Relations(Relations(ty, rels)))
473			}
474			CacheRequest::EventHandler(event) => self
475				.handle_event(*event)
476				.await
477				.map(|item| CacheResponse::Item(Item(item))),
478			CacheRequest::AddAll(items) => {
479				let _ = self.add_all(items);
480				Ok(CacheResponse::AddAll)
481			}
482		}
483	}
484	pub fn tree(
485		&self,
486	) -> &HashMap<String, HashMap<String, CacheItem, FxBuildHasher>, FxBuildHasher> {
487		&self.tree.0
488	}
489}

494impl<D: CacheDriver> Cache<D> {
495	/// create a new, fresh cache
496	#[must_use]
497	#[tracing::instrument(level = "debug", ret, skip_all)]
498	pub fn new(driver: D) -> Self {
499		Self { tree: NewCache(HashMap::with_hasher(FxBuildHasher::default())), driver }
500	}

502	/// Remove a single cache item. This function can not fail.
503	#[tracing::instrument(level = "trace", skip(self))]
504	pub fn remove(&mut self, id: &CacheKey) -> Option<CacheItem> {
505		let Some(item) = self.tree.remove(id) else {
506			tracing::warn!("Attempted to remove an item that doesn't exist: {id:?}");
507			return None;
508		};
509		Some(item)
510	}

512	/// Get a single item from the cache.
513	///
514	/// This will allow you to get the item without holding any locks to it,
515	/// at the cost of (1) a clone and (2) no guarantees that the data is kept up-to-date.
516	#[must_use]
517	#[tracing::instrument(level = "trace", skip(self), ret)]
518	pub fn get(&self, id: &CacheKey) -> Option<CacheItem> {
519		let ref_item = self.tree.get(id)?;
520		// clone the reference into an owned value
521		Some(ref_item.clone())
522	}
523
524	/// get a many items from the cache; this only creates one read handle (note that this will copy all data you would like to access)
525	#[must_use]
526	#[tracing::instrument(level = "trace", skip(self), ret)]
527	pub fn get_all(&self, ids: &[CacheKey]) -> Vec<Option<CacheItem>> {
528		ids.iter().map(|id| self.get(id)).collect()
529	}
530
531	/// Bulk remove all ids in the cache; this only refreshes the cache after removing all items.
532	#[tracing::instrument(level = "trace", ret, skip(self))]
533	pub fn remove_all(&mut self, ids: &[CacheKey]) {
534		for id in ids {
535			self.remove(id);
536		}
537	}
538
539	pub fn add(&mut self, ci: CacheItem) -> CacheItem {
540		self.tree.insert(ci.object.clone(), ci.clone());
541		ci
542	}
543	pub fn add_all(&mut self, cis: Vec<CacheItem>) -> Vec<CacheItem> {
544		let clone = cis.clone();
545		for ci in cis {
546			self.add(ci);
547		}
548		clone
549	}
550
551	async fn prefetch_app(&mut self, key: &CacheKey) -> OdiliaResult<CacheItem> {
552		let items = self.driver.lookup_bulk(key).await?;
553		for item in items {
554			self.tree.insert(key.clone(), item);
555		}
556		// this should always succeed since we just bulk added
557		self.get(key).ok_or(CacheError::NoItem.into())
558	}
559
560	async fn get_or_create_all(&mut self, keys: Vec<CacheKey>) -> OdiliaResult<Vec<CacheItem>> {
561		let mut found = vec![];
562		let mut not_found = vec![];
563		for key in keys {
564			match self.tree.get(&key) {
565				Some(cache_item) => found.push(cache_item.clone()),
566				None => not_found.push(key),
567			}
568		}
569		for key in not_found {
570			let item = self.get_or_create(&key).await?;
571			found.push(item);
572		}
573		Ok(self.add_all(found))
574	}

576	/// Modify the given item with closure [`F`] if it was already contained in the cache.
577	/// Otherwise, fetch a new item over the [`CacheDriver`].
578	///
579	/// # Errors
580	///
581	/// See: [`get_or_create`]
582	pub async fn modify_if_not_new<F>(
583		&mut self,
584		key: &AccessiblePrimitive,
585		f: F,
586	) -> OdiliaResult<CacheItem>
587	where
588		F: FnOnce(&mut CacheItem),
589	{
590		// if the item already exists in the cache, modify it
591		if let Some(cache_item) = self.tree.get_mut(key) {
592			f(cache_item);
593			return Ok(cache_item.clone());
594		}
595		self.get_or_create(key).await
596	}

598	/// Get a single item from the cache (note that this copies some integers to a new struct).
599	/// If the `CacheItem` is not found, create one, add it to the cache, and return it.
600	/// # Errors
601	/// The function will return an error if:
602	/// 1. The `accessible` can not be turned into an `AccessiblePrimitive`. This should never happen, but is technically possible.
603	/// 2. The [`Self::add`] function fails.
604	/// 3. The [`accessible_to_cache_item`] function fails.
605	///
606	/// # Panics
607	///
608	/// This function technically has a `.expect()` which could panic. But we gaurs against this.
609	#[tracing::instrument(level = "trace", ret, err(level = "warn"), skip(self))]
610	async fn get_or_create(&mut self, key: &AccessiblePrimitive) -> OdiliaResult<CacheItem> {
611		// if the item already exists in the cache, return it
612		if let Some(cache_item) = self.get(key) {
613			return Ok(cache_item);
614		}
615		// if the item's app has never had an item added to cache, bulk query it
616		if !self.tree.has_app(key) {
617			return self.prefetch_app(key).await;
618		}
619		let cache_item = self.driver.lookup_external(key).await?;
620		self.tree.insert(key.clone(), cache_item.clone());
621		Ok(cache_item)
622	}
623
624	/// Same as [`get_or_create`] but starts with an initial [`atspi::CacheItem`].
625	/// # Errors
626	/// The function will return an error if:
627	/// 1. The `accessible` can not be turned into an `AccessiblePrimitive`. This should never happen, but is technically possible.
628	/// 2. The [`Self::add`] function fails.
629	/// 3. The [`accessible_to_cache_item`] function fails.
630	///
631	/// # Panics
632	///
633	/// This function technically has a `.expect()` which could panic. But we gaurs against this.
634	#[tracing::instrument(level = "trace", ret, err(level = "warn"), skip(self))]
635	async fn get_or_create_from_cache_item(
636		&mut self,
637		ci: atspi::CacheItem,
638	) -> OdiliaResult<CacheItem> {
639		let key = ci.object.clone().into();
640		// if the item already exists in the cache, return it
641		if let Some(cache_item) = self.get(&key) {
642			return Ok(cache_item);
643		}
644		let cache_item = self.driver.lookup_from_cache_item(ci).await?;
645		self.tree.insert(key, cache_item.clone());
646		Ok(cache_item)
647	}
648
649	/// Same as [`get_or_create`] but starts with an initial [`atspi::LegacyCacheItem`].
650	/// # Errors
651	/// The function will return an error if:
652	/// 1. The `accessible` can not be turned into an `AccessiblePrimitive`. This should never happen, but is technically possible.
653	/// 2. The [`Self::add`] function fails.
654	/// 3. The [`accessible_to_cache_item`] function fails.
655	///
656	/// # Panics
657	///
658	/// This function technically has a `.expect()` which could panic. But we gaurs against this.
659	#[tracing::instrument(level = "trace", ret, err(level = "warn"), skip(self))]
660	async fn get_or_create_from_legacy_cache_item(
661		&mut self,
662		ci: atspi::LegacyCacheItem,
663	) -> OdiliaResult<CacheItem> {
664		let key = ci.object.clone().into();
665		// if the item already exists in the cache, return it
666		if let Some(cache_item) = self.get(&key) {
667			return Ok(cache_item);
668		}
669		let cache_item = self.driver.lookup_from_legacy_cache_item(ci).await?;
670		self.tree.insert(key, cache_item.clone());
671		Ok(cache_item)
672	}

675		self.tree.0.clear();

679/// Convert an [`atspi::proxy::accessible::AccessibleProxy`] into a [`crate::CacheItem`].
680/// This runs a bunch of long-awaiting code and can take quite some time; use this sparingly.
681/// This takes most properties and some function calls through the `AccessibleProxy` structure and generates a new `CacheItem`, which will be written to cache before being sent back.
682///
683/// # Errors
684///
685/// Will return an `Err(_)` variant when:
686///
687/// 1. The `cache` parameter does not reference an active cache once the `Weak` is upgraded to an `Option<Arc<_>>`.
688/// 2. Any of the function calls on the `accessible` fail.
689/// 3. Any `(String, OwnedObjectPath) -> AccessiblePrimitive` conversions fail. This *should* never happen, but technically it is possible.
690#[tracing::instrument(level = "trace", ret, err(level = "warn"))]
691pub async fn accessible_to_cache_item(accessible: &AccessibleProxy<'_>) -> OdiliaResult<CacheItem> {
692	let props = (
693		accessible.get_application(),
694		accessible.parent(),
695		accessible.get_index_in_parent(),
696		accessible.child_count(),
697		accessible.get_interfaces(),
698		accessible.get_role(),
699		accessible.get_state(),
700		accessible.get_children(),
701	)
702		.try_join();
703	let maps = (
704		accessible
705			.name()
706			.map_ok(|s| if s.is_empty() { None } else { Some(s) }),
707		accessible
708			.description()
709			.map_ok(|s| if s.is_empty() { None } else { Some(s) }),
710		accessible
711			.help_text()
712			.map_ok(|s| if s.is_empty() { None } else { Some(s) }),
713		accessible
714			.to_text()
715			.and_then(|text_proxy| {
716				text_proxy.get_all_text().map_ok(|s| {
717					if s.is_empty() {
718						None
719					} else {
720						Some(s)
721					}
722				})
723			})
724			.unwrap_or_else(|_| None)
725			.map(Ok),
726	)
727		.try_join();
728	let (
729		(app, parent, index, children_num, interfaces, role, states, children),
730		(name, description, help_text, text),
731	) = (props, maps).try_join().await?;
732
733	let ci = CacheItem {
734		object: accessible.into(),
735		app: app.into(),
736		parent: parent.into(),
737		index: index.try_into().ok(),
738		children_num: children_num.try_into().ok(),
739		interfaces,
740		role,
741		states,
742		children: children.into_iter().map(Into::into).collect(),
743		name,
744		description,
745		help_text,
746		text,
747	};
748	Ok(ci)
749}
