11/// A struct which represents the bare minimum of an accessible for purposes of caching.
12/// This makes some *possibly eronious* assumptions about what the sender is.
13pub struct AccessiblePrimitive {
14	/// The accessible ID, which is an arbitrary string specified by the application.
15	/// It is guaranteed to be unique per application.
16	/// Examples:
17	/// * /org/a11y/atspi/accessible/1234
18	/// * /org/a11y/atspi/accessible/null
19	/// * /org/a11y/atspi/accessible/root
20	/// * /org/Gnome/GTK/abab22-bbbb33-2bba2
21	pub id: String,
22	/// Assuming that the sender is ":x.y", this stores the (x,y) portion of this sender.
23	/// Examples:
24	/// * :1.1 (the first window has opened)
25	/// * :2.5 (a second session exists, where at least 5 applications have been lauinched)
26	/// * :1.262 (many applications have been started on this bus)
27	pub sender: smartstring::alias::String,
28}
29
30impl AccessiblePrimitive {
31	/// Turns any `atspi::event` type into an `AccessiblePrimitive`, the basic type which is used for keys in the cache.
32	#[cfg_attr(feature = "tracing", tracing::instrument(skip_all, level = "trace", ret, err))]
33	pub fn from_event<T: EventProperties>(event: &T) -> Self {
34		let sender = event.sender();
35		let path = event.path();
36		let id = path.to_string();
37		Self { id, sender: sender.as_str().into() }
38	}
39
40	/// Convert into an [`atspi_proxies::accessible::AccessibleProxy`]. Must be async because the creation of an async proxy requires async itself.
41	/// # Errors
42	/// Will return a [`zbus::Error`] in the case of an invalid destination, path, or failure to create a `Proxy` from those properties.
43	#[cfg_attr(feature = "tracing", tracing::instrument(skip_all, level = "trace", ret, err))]
44	pub async fn into_accessible<'a>(
45		self,
46		conn: &zbus::Connection,
47	) -> zbus::Result<AccessibleProxy<'a>> {
48		let id = self.id;
49		let sender = self.sender.clone();
50		let path: ObjectPath<'a> = id.try_into()?;
51		ProxyBuilder::new(conn)
52			.path(path)?
53			.destination(sender.as_str().to_owned())?
54			.cache_properties(CacheProperties::No)
55			.build()
56			.await
57	}
58	/// Convert into an [`atspi_proxies::text::TextProxy`]. Must be async because the creation of an async proxy requires async itself.
59	/// # Errors
60	/// Will return a [`zbus::Error`] in the case of an invalid destination, path, or failure to create a `Proxy` from those properties.
61	#[cfg_attr(feature = "tracing", tracing::instrument(skip_all, level = "trace", ret, err))]
62	pub async fn into_text<'a>(self, conn: &zbus::Connection) -> zbus::Result<TextProxy<'a>> {
63		let id = self.id;
64		let sender = self.sender.clone();
65		let path: ObjectPath<'a> = id.try_into()?;
66		ProxyBuilder::new(conn)
67			.path(path)?
68			.destination(sender.as_str().to_owned())?
69			.cache_properties(CacheProperties::No)
70			.build()
71			.await
72	}
73}

101impl TryFrom<&AccessibleProxy<'_>> for AccessiblePrimitive {
102	type Error = AccessiblePrimitiveConversionError;
103
104	#[cfg_attr(feature = "tracing", tracing::instrument(skip_all, level = "trace", ret, err))]
105	fn try_from(accessible: &AccessibleProxy<'_>) -> Result<AccessiblePrimitive, Self::Error> {
106		let accessible = accessible.inner();
107		let sender = accessible.destination().as_str().into();
108		let id = accessible.path().as_str().into();
109		Ok(AccessiblePrimitive { id, sender })
110	}
