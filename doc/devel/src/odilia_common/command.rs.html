13pub trait TryIntoCommands {
14	type Error: Into<OdiliaError>;
15	type Iter: Iterator<Item = OdiliaCommand> + Send;
16	/// Fallibly returns an iterator of [`OdiliaCommand`]s to run.
17	///
18	/// # Errors
19	///
20	/// When implemented, the function is allowed to fail with any type that can be converted into
21	/// [`OdiliaError`], but conversion should between these types should be done from the
22	/// implementers' side, liekly using `?`.
23	fn try_into_commands(self) -> Result<Self::Iter, OdiliaError>;
24}
25impl TryIntoCommands for Result<Vec<OdiliaCommand>, OdiliaError> {
26	type Error = OdiliaError;
27	type Iter = std::vec::IntoIter<OdiliaCommand>;
28	fn try_into_commands(self) -> Result<Self::Iter, OdiliaError> {
29		self.map(IntoIterator::into_iter)
30	}
31}
32impl<T: IntoCommands> TryIntoCommands for T {
33	type Error = Infallible;
34	type Iter = T::Iter;
35	fn try_into_commands(self) -> Result<Self::Iter, OdiliaError> {
36		Ok(self.into_commands())
37	}
38}
39impl<T: IntoCommands, E: Into<OdiliaError>> TryIntoCommands for Result<T, E> {
40	type Error = E;
41	type Iter = T::Iter;
42	fn try_into_commands(self) -> Result<Self::Iter, OdiliaError> {
43		match self {
44			Ok(ok) => Ok(ok.into_commands()),
45			Err(err) => Err(err.into()),
46		}
47	}
48}
