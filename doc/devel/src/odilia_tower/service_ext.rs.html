1//! Allow postfix notation for building services from existing ones.
2
3use crate::{
4	async_try::{AsyncTryInto, AsyncTryIntoLayer, AsyncTryIntoService},
5	state_svc::{StateLayer, StateService},
6	sync_try::{TryIntoLayer, TryIntoService},
7};
8use tower::{Layer, Service};

10/// Use postfix notation on your [`tower::Service`]s to produce nested services.
11pub trait ServiceExt<Request>: Service<Request> {
12	/// Map a new input type into your service's input using a [`TryInto`] impl.
13	///
14	/// ```
15	/// use assert_matches::assert_matches;
16	/// use odilia_tower::service_ext::ServiceExt;
17	/// use tower::{service_fn, Service};
18	/// use futures::executor::block_on;
19	/// use std::num::TryFromIntError;
20	/// // NOTE: the associated error type must implement From<E>, where E is the error in converting
21	/// // from the new input type (in this example, u64) to the inner one (u8).
22	/// #[derive(Debug, PartialEq, Eq)]
23	/// pub enum Error {
24	///   IntConv(TryFromIntError)
25	/// }
26	/// impl From<TryFromIntError> for Error {
27	///   fn from(ie: TryFromIntError) -> Error {
28	///       Error::IntConv(ie)
29	///   }
30	/// }
31	///
32	/// async fn double(input: u8) -> Result<u8, Error> {
33	///   Ok(input * 2)
34	/// }
35	/// let mut service = service_fn(double)
36	///     // try to convert u64 to u8 before passing to the service function
37	///     .request_try_from();
38	/// assert_eq!(block_on(service.call(100u64)), Ok(200u8));
39	/// // can not be successfully converted to u8!
40	/// assert_matches!(block_on(service.call(300u64)), Err(_));
41	/// ```
42	fn request_try_from<I, R, Fut1>(self) -> TryIntoService<Request, I, Self, R, Fut1>
43	where
44		Self: Sized,
45		I: TryInto<Request>,
46		Self: Service<Request, Response = R, Future = Fut1>,
47	{
48		TryIntoLayer::new().layer(self)
49	}
50	/// Map a new input type into your service's input using a custom [`AsyncTryInto`] impl.
51	/// This is mostly used in conjunction with state extraction, but doesn't inherently have to.
52	///
53	/// ```
54	/// use assert_matches::assert_matches;
55	/// use odilia_tower::{
56	///   service_ext::ServiceExt,
57	///   async_try::AsyncTryFrom,
58	/// };
59	/// use tower::{service_fn, Service};
60	/// use futures::{
61	///   executor::block_on,
62	///   future::{ready, Ready},
63	/// };
64	/// use std::num::TryFromIntError;
65	/// // Used to get around "foreign traits on foreign types" rule.
66	/// #[derive(Debug, PartialEq, Eq)]
67	/// struct U8(u8);
68	/// impl From<u8> for U8 {
69	///   fn from(inner: u8) -> U8 {
70	///     U8(inner)
71	///   }
72	/// }
73	/// #[derive(Debug, PartialEq, Eq)]
74	/// struct U64(u64);
75	/// impl From<u64> for U64 {
76	///   fn from(inner: u64) -> U64 {
77	///     U64(inner)
78	///   }
79	/// }
80	///
81	/// impl AsyncTryFrom<U64> for U8 {
82	///   type Error = TryFromIntError;
83	///   type Future = Ready<Result<U8, Self::Error>>;
84	///   fn try_from_async(big: U64) -> Self::Future {
85	///       ready(big.0.try_into().map(U8))
86	///   }
87	/// }
88	/// // NOTE: the associated error type must implement From<E>, where E is the error in converting
89	/// // from the new input type (in this example, u64) to the inner one (u8).
90	/// #[derive(Debug, PartialEq, Eq)]
91	/// pub enum Error {
92	///   IntConv(TryFromIntError)
93	/// }
94	/// impl From<TryFromIntError> for Error {
95	///   fn from(ie: TryFromIntError) -> Error {
96	///       Error::IntConv(ie)
97	///   }
98	/// }
99	///
100	/// async fn double(input: U8) -> Result<U8, Error> {
101	///   Ok((input.0 * 2).into())
102	/// }
103	/// let mut service = service_fn(double)
104	///     // try to convert u64 to u8 before passing to the service function
105	///     // this time, use the async function we described above
106	///     .request_async_try_from::<U64, U8, _>();
107	/// assert_eq!(block_on(service.call(100u64.into())), Ok(200u8.into()));
108	/// // can not be successfully converted to u8!
109	/// assert_matches!(block_on(service.call(300u64.into())), Err(_));
110	/// ```
111	fn request_async_try_from<I, R, Fut1>(
112		self,
113	) -> AsyncTryIntoService<Request, I, Self, R, Fut1>
114	where
115		I: AsyncTryInto<Request>,
116		Self: Service<Request, Response = R, Future = Fut1> + Clone,
117	{
118		AsyncTryIntoLayer::new().layer(self)
119	}
120	/// Inject a clonable state into each invocation of the inner service.
121	/// NOTE:
122	///
123	/// - since [`tower::Service`] only accepts functions with one parameter, this is passed as
124	///   `(S, P)` where `S` is the state type and `P` is the parameter type.
125	/// - The `S` parameter will be cloned upon each invocation to [`Service::call`]. It should be
126	///   relatively cheap to clone.
127	///
128	/// ```
129	/// use odilia_tower::service_ext::ServiceExt;
130	/// use tower::{service_fn, Service};
131	/// use futures::executor::block_on;
132	/// use std::{convert::Infallible, sync::Arc};
133	/// // a stand in for some comlpex type, don't actually do this.
134	/// type State = Arc<usize>;
135	///
136	/// // NOTE: `tower` does not allow multiple parameters to services.
137	/// async fn state_and_param((state, param): (Arc<usize>, u8)) -> Result<Vec<u8>, Infallible> {
138	///     let mut vec = Vec::with_capacity(*state);
139	///     for _ in 0..*state {
140	///         vec.push(param);
141	///     }
142	///     Ok(vec)
143	/// }
144	/// let mut service = service_fn(state_and_param)
145	///     .with_state(Arc::new(5));
146	/// assert_eq!(block_on(service.call(2)), Ok(vec![2u8, 2, 2, 2, 2]));
147	/// ```
148	fn with_state<S>(self, s: S) -> StateService<Self, S>
149	where
150		Self: Sized,
151		S: Clone,
152	{
153		StateLayer::new(s).layer(self)
154	}
155}
156
157impl<T: ?Sized, Request> ServiceExt<Request> for T where T: Service<Request> {}
