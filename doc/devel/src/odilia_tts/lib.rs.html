27/// Creates a new async SSIP client which can be sent commends, and can await responses to.
28/// # Errors
29/// There may be errors when trying to send the initial registration command, or when parsing the response.
30#[tracing::instrument(level = "debug", err)]
31pub async fn create_ssip_client(
32) -> eyre::Result<AsyncClient<BufReader<OwnedReadHalf>, BufWriter<OwnedWriteHalf>>> {
33	tracing::debug!("Attempting to register SSIP client odilia:speech");
34	let mut ssip_core =
35		match Builder::new().build().await {
36			Ok(ssip) => ssip,
37			Err(e) => {
38				if e.kind() == ErrorKind::ConnectionRefused {
39					tracing::debug!("Speech dispatcher is not active. Attempting to spawn it.");
40					Command::new("speech-dispatcher")
41              .arg("--spawn")
42              .stdin(Stdio::null())
43              .stdout(Stdio::null())
44              .stderr(Stdio::null())
45              .spawn()
46              .context("Error running `speech-dispatcher --spawn`; this is a fatal error.")
47			?;
48					tracing::debug!(
49						"Attempting to connect to speech-dispatcher again!"
50					);
51					tokio::time::sleep(time::Duration::from_secs(1)).await;
52					Builder::new().build().await?
53				} else {
54					tracing::debug!("Speech dispatcher could not be started.");
55					exit(1);
56				}
57			}
58		};
59	tracing::debug!("Client created. Setting name");
60	ssip_core
61		.set_client_name(ClientName::new("odilia", "speech"))
62		.await?
63		.check_client_name_set()
64		.await?;
65	tracing::debug!("SSIP client registered as odilia:speech");
66	Ok(ssip_core)
67}

69/// A handler task for incoming SSIP requests
70/// This function will run untill it gets canceled via the cancellation token
71///
72/// # Errors
73///
74/// This function will return an error if anything within it fails. It may fail to read a value from the channel, it may fail to run an SSIP command, or fail to parse the response.
75/// Errors may also be returned during cleanup via the `cancellation_token` parameter, since shutting down the connection to speech dispatcher can also potentially error.
76/// Any of these failures will result in this function exiting with an `Err(_)` variant.
77#[tracing::instrument(level = "debug", skip_all, err)]
78pub async fn handle_ssip_commands(
79	mut client: AsyncClient<BufReader<OwnedReadHalf>, BufWriter<OwnedWriteHalf>>,
80	requests: Receiver<Request>,
81	shutdown: CancellationToken,
82) -> eyre::Result<()> {
83	tokio::pin!(requests);
84	loop {
85		tokio::select! {
86				      request_option = requests.recv() => {
87					      if let Some(request) = request_option {
88		  tracing::debug!(?request, "SSIP command received");
89		  let response = client
90		    .send(request).await?
91		    .receive().await?;
92		  tracing::debug!(?response, "Recieved response from server");
93		}
94				      }
95				      () = shutdown.cancelled() => {
96		      tracing::debug!("Saying goodbye message.");
97		      client
98			      .send(Request::Speak).await?
99			      .receive().await?;
100		      client
101			      .send(Request::SendLines(Vec::from(["Quitting Odilia".to_string()]))).await?
102			      .receive().await?;
103		      tracing::debug!("Attempting to quit SSIP.");
104		      let response = client
105			.send(Request::Quit).await?
106			.receive().await?;
107		      tracing::debug!(?response, "Recieved response from server");
108					      tracing::debug!("SSIP command interpreter shutdown completed");
109					      break;
110				      }
111			      }
112	}
113	Ok(())
114}
