191fn evdev_event_to_rdev_event(event: &InputEvent) -> Option<EventType> {
192    match &event.event_code {
193        EventCode::EV_KEY(key) => {
194            if let Some(button) = evdev_key_to_rdev_button(key) {
195                // first check if pressed key is a mouse button
196                match event.value {
197                    0 => Some(EventType::ButtonRelease(button)),
198                    _ => Some(EventType::ButtonPress(button)),
199                }
200            } else if let Some(key) = evdev_key_to_rdev_key(key) {
201                // check if pressed key is a keyboard key
202                match event.value {
203                    0 => Some(EventType::KeyRelease(key)),
204                    _ => Some(EventType::KeyPress(key)),
205                }
206            } else {
207                // if neither mouse button nor keyboard key, return none
208                None
209            }
210        }
211        // Other event_codes cannot be represented by rdev,
212        // and some never will e.g. EV_SYN
213        _ => None,
214    }
215}

264pub fn grab<T, S>(callback: T, mut state: S) -> Result<(), GrabError>
265where
266    T: Fn(Event, &mut S) -> Option<Event> + 'static,
267{
268    //let mut kb = Keyboard::new().ok_or(GrabError::KeyboardError)?;
269    //let display = Display::new().ok_or(GrabError::MissingDisplayError)?;
270    //let (width, height) = display.get_size().ok_or(GrabError::MissingDisplayError)?;
271    //let (current_x, current_y) = display
272    //    .get_mouse_pos()
273    //    .ok_or(GrabError::MissingDisplayError)?;
274    //let mut x = current_x as f64;
275    //let mut y = current_y as f64;
276    //let w = width as f64;
277    //let h = height as f64;
278    filter_map_events(|event| {
279        let event_type = match evdev_event_to_rdev_event(&event) {
280            Some(rdev_event) => rdev_event,
281            // If we can't convert event, simulate it
282            None => return (Some(event), GrabStatus::Continue),
283        };
284        let rdev_event = Event {
285            time: SystemTime::now(),
286            // always empty, never used
287            name: Some(""),
288            event_type,
289        };
290        if callback(rdev_event, &mut state).is_some() {
291            (Some(event), GrabStatus::Continue)
292        } else {
293            // callback returns None, swallow the event
294            (None, GrabStatus::Continue)
295        }
296    })?;
297    Ok(())
298}

300pub fn filter_map_events<F>(mut func: F) -> io::Result<()>
301where
302    F: FnMut(InputEvent) -> (Option<InputEvent>, GrabStatus),
303{
304    let (epoll_fd, mut devices, output_devices) = setup_devices()?;
305    let mut inotify = setup_inotify(epoll_fd, &devices)?;
306
307    //grab devices
308    devices
309        .iter_mut()
310        .try_for_each(|device| device.grab(evdev_rs::GrabMode::Grab))?;
311
312    // create buffer for epoll to fill
313    let mut epoll_buffer = [epoll::Event::new(epoll::Events::empty(), 0); 4];
314    let mut inotify_buffer = vec![0_u8; 4096];
315    'event_loop: loop {
316        let num_events = epoll::wait(epoll_fd, -1, &mut epoll_buffer)?;
317
318        //map and simulate events, dealing with
319        'events: for event in &epoll_buffer[0..num_events] {
320            // new device file created
321            if event.data == INOTIFY_DATA {
322                for event in inotify.read_events(&mut inotify_buffer)? {
323                    assert!(
324                        event.mask.contains(inotify::EventMask::CREATE),
325                        "inotify is listening for events other than file creation"
326                    );
327                    add_device_to_epoll_from_inotify_event(epoll_fd, event, &mut devices)?;
328                }
329            } else {
330                // Input device recieved event
331                let device_idx = event.data as usize;
332                let device = devices.get(device_idx).unwrap();
333                while device.has_event_pending() {
334                    //TODO: deal with EV_SYN::SYN_DROPPED
335                    let (_, event) = match device.next_event(evdev_rs::ReadFlag::NORMAL) {
336                        Ok(event) => event,
337                        Err(_) => {
338                            let device_fd = device.fd().unwrap().into_raw_fd();
339                            let empty_event = epoll::Event::new(epoll::Events::empty(), 0);
340                            epoll::ctl(epoll_fd, EPOLL_CTL_DEL, device_fd, empty_event)?;
341                            continue 'events;
342                        }
343                    };
344                    let (event, grab_status) = func(event);
345
346                    if let (Some(event), Some(out_device)) = (event, output_devices.get(device_idx))
347                    {
348                        out_device.write_event(&event)?;
349                    }
350                    if grab_status == GrabStatus::Stop {
351                        break 'event_loop;
352                    }
353                }
354            }
355        }
356    }
357
358    for device in devices.iter_mut() {
359        //ungrab devices, ignore errors
360        device.grab(evdev_rs::GrabMode::Ungrab).ok();
361    }
362
363    epoll::close(epoll_fd)?;
364    Ok(())
365}

368const INOTIFY_DATA: u64 = u64::max_value();
369const EPOLLIN: epoll::Events = epoll::Events::EPOLLIN;

371/// Whether to continue grabbing events or to stop
372/// Used in `filter_map_events` (and others)
373#[derive(Debug, Eq, PartialEq, Hash)]
374pub enum GrabStatus {
375    /// Stop grabbing
376    Continue,
377    /// ungrab events
378    Stop,
379}

381fn get_device_files<T>(path: T) -> io::Result<Vec<File>>
382where
383    T: AsRef<Path>,
384{
385    let mut res = Vec::new();
386    for entry in read_dir(path)? {
387        let entry = entry?;
388        // /dev/input files are character devices
389        if !entry.file_type()?.is_char_device() {
390            continue;
391        }
392
393        let path = entry.path();
394        let file_name_bytes = match path.file_name() {
395            Some(file_name) => file_name.as_bytes(),
396            None => continue, // file_name was "..", should be impossible
397        };
398        // skip filenames matching "mouse.* or mice".
399        // these files don't play nice with libevdev, not sure why
400        // see: https://askubuntu.com/questions/1043832/difference-between-dev-input-mouse0-and-dev-input-mice
401        if file_name_bytes == OsStr::new("mice").as_bytes()
402            || file_name_bytes
403                .get(0..=1)
404                .map(|s| s == OsStr::new("js").as_bytes())
405                .unwrap_or(false)
406            || file_name_bytes
407                .get(0..=4)
408                .map(|s| s == OsStr::new("mouse").as_bytes())
409                .unwrap_or(false)
410        {
411            continue;
412        }
413        res.push(File::open(path)?);
414    }
415    Ok(res)
416}

418fn epoll_watch_all<'a, T>(device_files: T) -> io::Result<RawFd>
419where
420    T: Iterator<Item = &'a File>,
421{
422    let epoll_fd = epoll::create(true)?;
423    // add file descriptors to epoll
424    for (file_idx, file) in device_files.enumerate() {
425        let epoll_event = epoll::Event::new(EPOLLIN, file_idx as u64);
426        epoll::ctl(epoll_fd, EPOLL_CTL_ADD, file.as_raw_fd(), epoll_event)?;
427    }
428    Ok(epoll_fd)
429}

431fn inotify_devices() -> io::Result<Inotify> {
432    let mut inotify = Inotify::init()?;
433    inotify.add_watch(DEV_PATH, WatchMask::CREATE)?;
434    Ok(inotify)
435}

437fn add_device_to_epoll_from_inotify_event(
438    epoll_fd: RawFd,
439    event: inotify::Event<&OsStr>,
440    devices: &mut Vec<Device>,
441) -> io::Result<()> {
442    let mut device_path = OsString::from(DEV_PATH);
443    device_path.push(OsString::from("/"));
444    device_path.push(event.name.unwrap());
445    // new plug events
446    let file = File::open(device_path)?;
447    let fd = file.as_raw_fd();
448    let device = Device::new_from_fd(file)?;
449    let event = epoll::Event::new(EPOLLIN, devices.len() as u64);
450    devices.push(device);
451    epoll::ctl(epoll_fd, EPOLL_CTL_ADD, fd, event)?;
452    Ok(())
453}

455/// Returns tuple of epoll_fd, all devices, and uinput devices, where
456/// uinputdevices is the same length as devices, and each uinput device is
457/// a libevdev copy of its corresponding device.The epoll_fd is level-triggered
458/// on any available data in the original devices.
459fn setup_devices() -> io::Result<(RawFd, Vec<Device>, Vec<UInputDevice>)> {
460    let device_files = get_device_files(DEV_PATH)?;
461    let epoll_fd = epoll_watch_all(device_files.iter())?;
462    let devices = device_files
463        .into_iter()
464        .map(Device::new_from_fd)
465        .collect::<io::Result<Vec<Device>>>()?;
466    let output_devices = devices
467        .iter()
468        .map(UInputDevice::create_from_device)
469        .collect::<io::Result<Vec<UInputDevice>>>()?;
470    Ok((epoll_fd, devices, output_devices))
471}

473/// Creates an inotify instance looking at /dev/input and adds it to an epoll instance.
474/// Ensures devices isnt too long, which would make the epoll data ambigious.
475fn setup_inotify(epoll_fd: RawFd, devices: &[Device]) -> io::Result<Inotify> {
476    //Ensure there is space for inotify at last epoll index.
477    if devices.len() as u64 >= INOTIFY_DATA {
478        eprintln!("number of devices: {}", devices.len());
479        return Err(io::Error::other("too many device files!"));
480    }
481    // Set up inotify to listen for new devices being plugged in
482    let inotify = inotify_devices()?;
483    let epoll_event = epoll::Event::new(EPOLLIN, INOTIFY_DATA);
484    epoll::ctl(epoll_fd, EPOLL_CTL_ADD, inotify.as_raw_fd(), epoll_event)?;
485    Ok(inotify)
