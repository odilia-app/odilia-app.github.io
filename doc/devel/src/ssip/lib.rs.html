418/// Synthesis voice
419#[derive(Debug, PartialEq, Eq, Clone, Hash)]
420pub struct SynthesisVoice {
421    pub name: String,
422    pub language: Option<String>,
423    pub dialect: Option<String>,
424}
425
426impl SynthesisVoice {
427    pub fn new(name: &str, language: Option<&str>, dialect: Option<&str>) -> SynthesisVoice {
428        SynthesisVoice {
429            name: name.to_string(),
430            language: language.map(|s| s.to_string()),
431            dialect: dialect.map(|s| s.to_string()),
432        }
433    }
434    /// Parse Option::None or string "none" into Option::None
435    fn parse_none(token: Option<&str>) -> Option<String> {
436        match token {
437            Some(s) => match s {
438                "none" => None,
439                s => Some(s.to_string()),
440            },
441            None => None,
442        }
443    }
444}
445
446impl FromStr for SynthesisVoice {
447    type Err = ClientError;
448
449    fn from_str(s: &str) -> Result<Self, Self::Err> {
450        let mut iter = s.split('\t');
451        match iter.next() {
452            Some(name) => Ok(SynthesisVoice {
453                name: name.to_string(),
454                language: SynthesisVoice::parse_none(iter.next()),
455                dialect: SynthesisVoice::parse_none(iter.next()),
456            }),
457            None => Err(ClientError::unexpected_eof("missing synthesis voice name")),
458        }
459    }
460}

462/// Command status line
463///
464/// Consists in a 3-digits code and a message. It can be a success or a failure.
465///
466/// Examples:
467/// - 216 OK OUTPUT MODULE SET
468/// - 409 ERR RATE TOO HIGH
469#[derive(Debug, PartialEq, Eq)]
470#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
471#[cfg_attr(feature = "dbus", derive(zvariant::Type))]
472pub struct StatusLine {
473    pub code: ReturnCode,
474    pub message: String,
475}
476
477impl fmt::Display for StatusLine {
478    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
479        write!(f, "{} {}", self.code, self.message)
480    }
481}
482/// Client error, either I/O error or SSIP error.
483#[derive(Debug)]
484pub enum ClientError {
485    Io(io::Error),
486    NotReady,
487    Ssip(StatusLine),
488    TooFewLines,
489    TooManyLines,
490    UnexpectedStatus(ReturnCode),
491}
492
493impl fmt::Display for ClientError {
494    fn fmt(&self, fmt: &mut fmt::Formatter) -> fmt::Result {
495        match self {
496            ClientError::Io(ioe) => {
497                fmt.write_str("I/O: ")?;
498                ioe.fmt(fmt)
499            }
500            ClientError::NotReady => fmt.write_str("Not ready"),
501            ClientError::Ssip(status_line) => {
502                fmt.write_str("SSIP: ")?;
503                status_line.fmt(fmt)
504            }
505            ClientError::TooFewLines => fmt.write_str("Too few lines"),
506            ClientError::TooManyLines => fmt.write_str("Too many lines"),
507            ClientError::UnexpectedStatus(return_code) => {
508                fmt.write_str("Unexpected status: ")?;
509                return_code.fmt(fmt)
510            }
511        }
512    }
513}
514impl Error for ClientError {}
515
516impl ClientError {
517    /// Create I/O error
518    pub fn io_error(kind: io::ErrorKind, msg: &str) -> Self {
519        Self::Io(io::Error::new(kind, msg))
520    }
521
522    /// Invalid data I/O error
523    pub fn invalid_data(msg: &str) -> Self {
524        ClientError::io_error(io::ErrorKind::InvalidData, msg)
525    }
526
527    /// Unexpected EOF I/O error
528    pub fn unexpected_eof(msg: &str) -> Self {
529        ClientError::io_error(io::ErrorKind::UnexpectedEof, msg)
530    }
531}
532
533impl From<io::Error> for ClientError {
534    fn from(err: io::Error) -> Self {
535        if err.kind() == io::ErrorKind::WouldBlock {
536            ClientError::NotReady
537        } else {
538            ClientError::Io(err)
539        }
540    }
541}

680/// History client status
681#[derive(Debug, PartialEq, Eq, Clone, Hash)]
682#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
683#[cfg_attr(feature = "dbus", derive(zvariant::Type))]
684pub struct HistoryClientStatus {
685    pub id: ClientId,
686    pub name: String,
687    pub connected: bool,
688}
689
690impl HistoryClientStatus {
691    pub fn new(id: ClientId, name: &str, connected: bool) -> Self {
692        Self {
693            id,
694            name: name.to_string(),
695            connected,
696        }
697    }
698}
699
700impl FromStr for HistoryClientStatus {
701    type Err = ClientError;
702
703    fn from_str(s: &str) -> Result<Self, Self::Err> {
704        let mut iter = s.splitn(3, ' ');
705        match iter.next() {
706            Some("") => Err(ClientError::unexpected_eof("expecting client id")),
707            Some(client_id) => match client_id.parse::<u32>() {
708                Ok(id) => match iter.next() {
709                    Some(name) => match iter.next() {
710                        Some("0") => Ok(HistoryClientStatus::new(id, name, false)),
711                        Some("1") => Ok(HistoryClientStatus::new(id, name, true)),
712                        Some(_) => Err(ClientError::invalid_data("invalid client status")),
713                        None => Err(ClientError::unexpected_eof("expecting client status")),
714                    },
715                    None => Err(ClientError::unexpected_eof("expecting client name")),
716                },
717                Err(_) => Err(ClientError::invalid_data("invalid client id")),
718            },
719            None => Err(ClientError::unexpected_eof("expecting client id")),
720        }
721    }
722}

855        assert_eq!("Portuguese (Portugal)+Kaukovalta", v1.name);

862        assert_eq!("eo", v2.language.unwrap());
863        assert!(v2.dialect.is_none());

880    #[test]
881    fn parse_history_client_status() {
882        assert_eq!(
883            HistoryClientStatus::new(10, "joe:speechd_client:main", false),
884            HistoryClientStatus::from_str("10 joe:speechd_client:main 0").unwrap()
885        );
886        assert_eq!(
887            HistoryClientStatus::new(11, "joe:speechd_client:main", true),
888            HistoryClientStatus::from_str("11 joe:speechd_client:main 1").unwrap()
889        );
890        for line in &[
891            "9 joe:speechd_client:main xxx",
892            "xxx joe:speechd_client:main 1",
893        ] {
894            match HistoryClientStatus::from_str(line) {
895                Ok(_) => panic!("parsing should have failed"),
896                Err(ClientError::Io(err)) if err.kind() == io::ErrorKind::InvalidData => (),
897                Err(_) => panic!("expecting error 'invalid data' parsing \"{line}\""),
898            }
899        }
900        for line in &["8 joe:speechd_client:main", "8", ""] {
901            match HistoryClientStatus::from_str(line) {
902                Ok(_) => panic!("parsing should have failed"),
903                Err(ClientError::Io(err)) if err.kind() == io::ErrorKind::UnexpectedEof => (),
904                Err(_) => panic!("expecting error 'unexpected EOF' parsing \"{line}\""),
905            }
906        }
907    }
