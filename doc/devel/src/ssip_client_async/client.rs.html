26/// Convert boolean to ON or OFF
27fn on_off(value: bool) -> &'static str {
28    if value {
29        "on"
30    } else {
31        "off"
32    }
33}

35macro_rules! send_one_line {
36    ($self:expr, $fmt:expr, $( $arg:expr ),+) => {
37        flush_lines(&mut $self.output, &[format!($fmt, $( $arg ),+).as_str()])
38    };
39    ($self:expr, $fmt:expr) => {
40        flush_lines(&mut $self.output, &[$fmt])
41    }
42}
43
44macro_rules! send_toggle {
45    ($output:expr, $fmt:expr, $val:expr) => {
46        send_one_line!($output, $fmt, on_off($val))
47    };
48    ($output:expr, $fmt:expr, $arg:expr, $val:expr) => {
49        send_one_line!($output, $fmt, $arg, on_off($val))
50    };
51}
52
53macro_rules! send_range {
54    ($output:expr, $fmt:expr, $scope:expr, $val:expr) => {
55        send_one_line!(
56            $output,
57            $fmt,
58            $scope,
59            std::cmp::max(-100, std::cmp::min(100, $val))
60        )
61    };
62}

64/// SSIP client on generic stream
65///
66/// There are two ways to send requests and receive responses:
67/// * Either with the generic [`Client::send`] and [`Client::receive`]
68/// * Or with the specific methods such as [`Client::set_rate`], ..., [`Client::get_rate`], ...
69pub struct Client<S: Read + Write + Source> {
70    input: io::BufReader<S>,
71    output: io::BufWriter<S>,
72}
73
74impl<S: Read + Write + Source> Client<S> {
75    /// Create a SSIP client on the reader and writer.
76    pub(crate) fn new(input: io::BufReader<S>, output: io::BufWriter<S>) -> Self {
77        // https://stackoverflow.com/questions/58467659/how-to-store-tcpstream-with-bufreader-and-bufwriter-in-a-data-structure
78        Self { input, output }
79    }

93    /// Send lines of text (terminated by a single dot).
94    pub fn send_lines(&mut self, lines: &[String]) -> ClientResult<&mut Self> {
95        const END_OF_DATA: [&str; 1] = ["."];
96        write_lines(
97            &mut self.output,
98            lines
99                .iter()
100                .map(|s| s.as_str())
101                .collect::<Vec<&str>>()
102                .as_slice(),
103        )?;
104        flush_lines(&mut self.output, &END_OF_DATA)?;
105        Ok(self)
106    }
107
108    /// Send one line of text (terminated by a single dot).
109    pub fn send_line(&mut self, line: &str) -> ClientResult<&mut Self> {
110        const END_OF_DATA: &str = ".";
111        flush_lines(&mut self.output, &[line, END_OF_DATA])?;
112        Ok(self)
113    }

115    /// Send a request
116    pub fn send(&mut self, request: Request) -> ClientResult<&mut Self> {
117        match request {
118            Request::SetName(client_name) => send_one_line!(
119                self,
120                "SET self CLIENT_NAME {}:{}:{}",
121                client_name.user,
122                client_name.application,
123                client_name.component
124            ),
125            Request::Speak => send_one_line!(self, "SPEAK"),
126            Request::SendLine(line) => self.send_line(&line).map(|_| ()),
127            Request::SendLines(lines) => self.send_lines(&lines).map(|_| ()),
128            Request::SpeakChar(ch) => send_one_line!(self, "CHAR {}", ch),
129            Request::SpeakKey(key) => send_one_line!(self, "KEY {}", key),
130            Request::Stop(scope) => send_one_line!(self, "STOP {}", scope),
131            Request::Cancel(scope) => send_one_line!(self, "CANCEL {}", scope),
132            Request::Pause(scope) => send_one_line!(self, "PAUSE {}", scope),
133            Request::Resume(scope) => send_one_line!(self, "RESUME {}", scope),
134            Request::SetPriority(prio) => send_one_line!(self, "SET self PRIORITY {}", prio),
135            Request::SetDebug(value) => send_toggle!(self, "SET all DEBUG {}", value),
136            Request::SetOutputModule(scope, value) => {
137                send_one_line!(self, "SET {} OUTPUT_MODULE {}", scope, value)
138            }
139            Request::GetOutputModule => send_one_line!(self, "GET OUTPUT_MODULE"),
140            Request::ListOutputModules => send_one_line!(self, "LIST OUTPUT_MODULES"),
141            Request::SetLanguage(scope, lang) => {
142                send_one_line!(self, "SET {} LANGUAGE {}", scope, lang)
143            }
144            Request::GetLanguage => send_one_line!(self, "GET LANGUAGE"),
145            Request::SetSsmlMode(value) => send_toggle!(self, "SET self SSML_MODE {}", value),
146            Request::SetPunctuationMode(scope, mode) => {
147                send_one_line!(self, "SET {} PUNCTUATION {}", scope, mode)
148            }
149            Request::SetSpelling(scope, value) => {
150                send_toggle!(self, "SET {} SPELLING {}", scope, value)
151            }
152            Request::SetCapitalLettersRecognitionMode(scope, mode) => {
153                send_one_line!(self, "SET {} CAP_LET_RECOGN {}", scope, mode)
154            }
155            Request::SetVoiceType(scope, value) => {
156                send_one_line!(self, "SET {} VOICE_TYPE {}", scope, value)
157            }
158            Request::GetVoiceType => send_one_line!(self, "GET VOICE_TYPE"),
159            Request::ListVoiceTypes => send_one_line!(self, "LIST VOICES"),
160            Request::SetSynthesisVoice(scope, value) => {
161                send_one_line!(self, "SET {} SYNTHESIS_VOICE {}", scope, value)
162            }
163            Request::ListSynthesisVoices => send_one_line!(self, "LIST SYNTHESIS_VOICES"),
164            Request::SetRate(scope, value) => send_range!(self, "SET {} RATE {}", scope, value),
165            Request::GetRate => send_one_line!(self, "GET RATE"),
166            Request::SetPitch(scope, value) => send_range!(self, "SET {} PITCH {}", scope, value),
167            Request::GetPitch => send_one_line!(self, "GET PITCH"),
168            Request::SetVolume(scope, value) => {
169                send_range!(self, "SET {} VOLUME {}", scope, value)
170            }
171            Request::GetVolume => send_one_line!(self, "GET VOLUME"),
172            Request::SetPauseContext(scope, value) => {
173                send_one_line!(self, "SET {} PAUSE_CONTEXT {}", scope, value)
174            }
175            Request::SetHistory(scope, value) => {
176                send_toggle!(self, "SET {} HISTORY {}", scope, value)
177            }
178            Request::SetNotification(ntype, value) => {
179                send_toggle!(self, "SET self NOTIFICATION {} {}", ntype, value)
180            }
181            Request::Begin => send_one_line!(self, "BLOCK BEGIN"),
182            Request::End => send_one_line!(self, "BLOCK END"),
183            Request::HistoryGetClients => send_one_line!(self, "HISTORY GET CLIENT_LIST"),
184            Request::HistoryGetClientId => send_one_line!(self, "HISTORY GET CLIENT_ID"),
185            Request::HistoryGetClientMsgs(scope, start, number) => send_one_line!(
186                self,
187                "HISTORY GET CLIENT_MESSAGES {} {}_{}",
188                scope,
189                start,
190                number
191            ),
192            Request::HistoryGetLastMsgId => send_one_line!(self, "HISTORY GET LAST"),
193            Request::HistoryGetMsg(id) => send_one_line!(self, "HISTORY GET MESSAGE {}", id),
194            Request::HistoryCursorGet => send_one_line!(self, "HISTORY CURSOR GET"),
195            Request::HistoryCursorSet(scope, pos) => {
196                send_one_line!(self, "HISTORY CURSOR SET {} {}", scope, pos)
197            }
198            Request::HistoryCursorMove(direction) => {
199                send_one_line!(self, "HISTORY CURSOR {}", direction)
200            }
201            Request::HistorySpeak(id) => send_one_line!(self, "HISTORY SAY {}", id),
202            Request::HistorySort(direction, key) => {
203                send_one_line!(self, "HISTORY SORT {} {}", direction, key)
204            }
205            Request::HistorySetShortMsgLength(length) => {
206                send_one_line!(self, "HISTORY SET SHORT_MESSAGE_LENGTH {}", length)
207            }
208            Request::HistorySetMsgTypeOrdering(ordering) => {
209                send_one_line!(
210                    self,
211                    "HISTORY SET MESSAGE_TYPE_ORDERING \"{}\"",
212                    ordering
213                        .iter()
214                        .map(|x| x.to_string())
215                        .collect::<Vec<String>>()
216                        .join(" ")
217                )
218            }
219            Request::HistorySearch(scope, condition) => {
220                send_one_line!(self, "HISTORY SEARCH {} \"{}\"", scope, condition)
221            }
222            Request::Quit => send_one_line!(self, "QUIT"),
223        }?;
224        Ok(self)
225    }

519    /// Receive one response.
520    pub fn receive(&mut self) -> ClientResult<Response> {
521        const MSG_CURSOR_SET_FIRST: &str = "OK CURSOR SET FIRST";
522        let mut lines = Vec::new();
523        let status = self.receive_answer(&mut lines)?;
524        match status.code {
525            OK_LANGUAGE_SET => Ok(Response::LanguageSet),
526            OK_PRIORITY_SET => Ok(Response::PrioritySet),
527            OK_RATE_SET => Ok(Response::RateSet),
528            OK_PITCH_SET => Ok(Response::PitchSet),
529            OK_PUNCTUATION_SET => Ok(Response::PunctuationSet),
530            OK_CAP_LET_RECOGN_SET => Ok(Response::CapLetRecognSet),
531            OK_SPELLING_SET => Ok(Response::SpellingSet),
532            OK_CLIENT_NAME_SET => Ok(Response::ClientNameSet),
533            OK_VOICE_SET => Ok(Response::VoiceSet),
534            OK_STOPPED => Ok(Response::Stopped),
535            OK_PAUSED => Ok(Response::Paused),
536            OK_RESUMED => Ok(Response::Resumed),
537            OK_CANCELED => Ok(Response::Canceled),
538            OK_TABLE_SET => Ok(Response::TableSet),
539            OK_OUTPUT_MODULE_SET => Ok(Response::OutputModuleSet),
540            OK_PAUSE_CONTEXT_SET => Ok(Response::PauseContextSet),
541            OK_VOLUME_SET => Ok(Response::VolumeSet),
542            OK_SSML_MODE_SET => Ok(Response::SsmlModeSet),
543            // Warning OK_CUR_SET_FIRST == OK_NOTIFICATION_SET == 220. Matching message to make the difference
544            OK_NOTIFICATION_SET => {
545                if status.message == MSG_CURSOR_SET_FIRST {
546                    //OK_CUR_SET_FIRST => Ok(Response::HistoryCurSetFirst)
547                    Ok(Response::HistoryCurSetFirst)
548                } else {
549                    Ok(Response::NotificationSet)
550                }
551            }
552            OK_CUR_SET_LAST => Ok(Response::HistoryCurSetLast),
553            OK_CUR_SET_POS => Ok(Response::HistoryCurSetPos),
554            OK_PITCH_RANGE_SET => Ok(Response::PitchRangeSet),
555            OK_DEBUG_SET => Ok(Response::DebugSet),
556            OK_CUR_MOV_FOR => Ok(Response::HistoryCurMoveFor),
557            OK_CUR_MOV_BACK => Ok(Response::HistoryCurMoveBack),
558            OK_MESSAGE_QUEUED => Ok(Response::MessageQueued),
559            OK_SND_ICON_QUEUED => Ok(Response::SoundIconQueued),
560            OK_MSG_CANCELED => Ok(Response::MessageCanceled),
561            OK_RECEIVING_DATA => Ok(Response::ReceivingData),
562            OK_BYE => Ok(Response::Bye),
563            OK_CLIENTS_LIST_SENT => Ok(Response::HistoryClientListSent(parse_typed_lines::<
564                HistoryClientStatus,
565            >(&lines)?)),
566            OK_MSGS_LIST_SENT => Ok(Response::HistoryMsgsListSent(lines)),
567            OK_LAST_MSG => Ok(Response::HistoryLastMsg(parse_single_value(&lines)?)),
568            OK_CUR_POS_RET => Ok(Response::HistoryCurPosRet(parse_single_value(&lines)?)),
569            OK_TABLE_LIST_SENT => Ok(Response::TableListSent(lines)),
570            OK_CLIENT_ID_SENT => Ok(Response::HistoryClientIdSent(parse_single_integer(&lines)?)),
571            OK_MSG_TEXT_SENT => Ok(Response::MessageTextSent),
572            OK_HELP_SENT => Ok(Response::HelpSent(lines)),
573            OK_VOICES_LIST_SENT => Ok(Response::VoicesListSent(
574                parse_typed_lines::<SynthesisVoice>(&lines)?,
575            )),
576            OK_OUTPUT_MODULES_LIST_SENT => Ok(Response::OutputModulesListSent(lines)),
577            OK_GET => Ok(Response::Get(parse_single_value(&lines)?)),
578            OK_INSIDE_BLOCK => Ok(Response::InsideBlock),
579            OK_OUTSIDE_BLOCK => Ok(Response::OutsideBlock),
580            OK_NOT_IMPLEMENTED => Ok(Response::NotImplemented),
581            EVENT_INDEX_MARK => match lines.len() {
582                0..=2 => Err(ClientError::TooFewLines),
583                3 => Ok(Response::EventIndexMark(
584                    parse_event_id(&lines)?,
585                    lines[2].to_owned(),
586                )),
587                _ => Err(ClientError::TooManyLines),
588            },
589            EVENT_BEGIN => Ok(Response::EventBegin(parse_event_id(&lines)?)),
590            EVENT_END => Ok(Response::EventEnd(parse_event_id(&lines)?)),
591            EVENT_CANCELED => Ok(Response::EventCanceled(parse_event_id(&lines)?)),
592            EVENT_PAUSED => Ok(Response::EventPaused(parse_event_id(&lines)?)),
593            EVENT_RESUMED => Ok(Response::EventResumed(parse_event_id(&lines)?)),
594            _ => panic!("error should have been caught earlier"),
595        }
596    }

598    /// Check status of answer, discard lines.
599    pub fn check_status(&mut self, expected_code: ReturnCode) -> ClientResult<&mut Self> {
600        crate::protocol::receive_answer(&mut self.input, None).and_then(|status| {
601            if status.code == expected_code {
602                Ok(self)
603            } else {
604                Err(ClientError::UnexpectedStatus(status.code))
605            }
606        })
607    }

673    /// Receive a notification
674    pub fn receive_event(&mut self) -> ClientResult<Event> {
675        let mut lines = Vec::new();
676        crate::protocol::receive_answer(&mut self.input, Some(&mut lines)).and_then(|status| {
677            if lines.len() < 2 {
678                Err(ClientError::unexpected_eof("event truncated"))
679            } else {
680                let message = &lines[0];
681                let client = &lines[1];
682                match status.code {
683                    700 => {
684                        if lines.len() != 3 {
685                            Err(ClientError::unexpected_eof("index markevent truncated"))
686                        } else {
687                            let mark = lines[3].to_owned();
688                            Ok(Event::index_mark(mark, message, client))
689                        }
690                    }
691                    701 => Ok(Event::begin(message, client)),
692                    702 => Ok(Event::end(message, client)),
693                    703 => Ok(Event::cancel(message, client)),
694                    704 => Ok(Event::pause(message, client)),
695                    705 => Ok(Event::resume(message, client)),
696                    _ => Err(ClientError::invalid_data("wrong status code for event")),
697                }
698            }
699        })
700    }

718    /// Register the socket for polling.
719    #[cfg(feature = "async-mio")]
720    pub fn register(
721        &mut self,
722        poll: &mio::Poll,
723        input_token: mio::Token,
724        output_token: mio::Token,
725    ) -> io::Result<()> {
726        poll.registry()
727            .register(self.input.get_mut(), input_token, mio::Interest::READABLE)?;
728        poll.registry()
729            .register(self.output.get_mut(), output_token, mio::Interest::WRITABLE)?;
730        Ok(())
731    }
