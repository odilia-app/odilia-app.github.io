10#[cfg(not(feature = "async-mio"))]
11mod synchronous {
12    use std::io::{self, BufReader, BufWriter};
13    pub use std::net::TcpStream;
14    use std::net::{SocketAddr, ToSocketAddrs};
15    use std::time::Duration;
16    use std::vec;
17
18    use crate::client::Client;
19    use crate::net::StreamMode;
20
21    struct Addresses(Vec<SocketAddr>);
22
23    impl ToSocketAddrs for Addresses {
24        type Iter = vec::IntoIter<SocketAddr>;
25        fn to_socket_addrs(&self) -> io::Result<Self::Iter> {
26            Ok(self.0.clone().into_iter())
27        }
28    }
29
30    pub struct Builder {
31        addrs: Addresses,
32        mode: StreamMode,
33    }
34
35    impl Builder {
36        pub fn new<A: ToSocketAddrs>(addrs: A) -> io::Result<Self> {
37            Ok(Self {
38                addrs: Addresses(addrs.to_socket_addrs()?.collect::<Vec<SocketAddr>>()),
39                mode: StreamMode::Blocking,
40            })
41        }
42
43        pub fn timeout(&mut self, read_timeout: Duration) -> &mut Self {
44            self.mode = StreamMode::TimeOut(read_timeout);
45            self
46        }
47
48        pub fn nonblocking(&mut self) -> &mut Self {
49            self.mode = StreamMode::NonBlocking;
50            self
51        }
52
53        pub fn build(&self) -> io::Result<Client<TcpStream>> {
54            let input = TcpStream::connect(&self.addrs)?;
55            match self.mode {
56                StreamMode::Blocking => input.set_nonblocking(false)?,
57                StreamMode::NonBlocking => input.set_nonblocking(true)?,
58                StreamMode::TimeOut(timeout) => input.set_read_timeout(Some(timeout))?,
59            }
60            let output = input.try_clone()?;
61            Ok(Client::new(BufReader::new(input), BufWriter::new(output)))
62        }
63    }
64}
65
66#[cfg(not(feature = "async-mio"))]
67pub use synchronous::{Builder, TcpStream};
68
69#[cfg(feature = "async-mio")]
70mod asynchronous {
71    pub use mio::net::TcpStream;
72    use std::io::{self, BufReader, BufWriter};
73    use std::net::SocketAddr;
74    use std::net::TcpStream as StdTcpStream;
75
76    use crate::client::Client;
77
78    pub struct Builder {
79        addr: SocketAddr,
80    }
81
82    impl Builder {
83        pub fn new(addr: SocketAddr) -> Self {
84            Self { addr }
85        }
86
87        pub fn build(&self) -> io::Result<Client<TcpStream>> {
88            let stream = StdTcpStream::connect(self.addr)?;
89            Ok(Client::new(
90                BufReader::new(TcpStream::from_std(stream.try_clone()?)),
91                BufWriter::new(TcpStream::from_std(stream)),
92            ))
93        }
94    }
95}
96
97#[cfg(feature = "async-mio")]
98pub use asynchronous::{Builder, TcpStream};
