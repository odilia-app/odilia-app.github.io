17macro_rules! send_one_line {
18    ($self:expr, $fmt:expr, $( $arg:expr ),+) => {
19        flush_lines_tokio(&mut $self.output, &[format!($fmt, $( $arg ),+).as_str()]).await
20    };
21    ($self:expr, $fmt:expr) => {
22        flush_lines_tokio(&mut $self.output, &[$fmt]).await
23    }
24}
25macro_rules! send_toggle {
26    ($output:expr, $fmt:expr, $val:expr) => {
27        send_one_line!($output, $fmt, on_off($val))
28    };
29    ($output:expr, $fmt:expr, $arg:expr, $val:expr) => {
30        send_one_line!($output, $fmt, $arg, on_off($val))
31    };
32}
33
34macro_rules! send_range {
35    ($output:expr, $fmt:expr, $scope:expr, $val:expr) => {
36        send_one_line!(
37            $output,
38            $fmt,
39            $scope,
40            std::cmp::max(-100, std::cmp::min(100, $val))
41        )
42    };
43}

47/// Convert boolean to ON or OFF
48fn on_off(value: bool) -> &'static str {
49    if value {
50        "on"
51    } else {
52        "off"
53    }

56/// SSIP client on generic async stream
57///
58/// There are two ways to send requests and receive responses:
59/// * Either with the generic [`AsyncClient::send`] and [`AsyncClient::receive`]
60/// * Or with the specific methods such as [`AsyncClient::set_rate`], ..., [`AsyncClient::get_rate`], ...
61pub struct AsyncClient<R: AsyncBufRead + Unpin, W: AsyncWrite + Unpin> {
62    input: R,
63    output: W,
64}
65impl<R: AsyncBufRead + Unpin, W: AsyncWrite + Unpin> AsyncClient<R, W> {
66    pub(crate) fn new(input: R, output: W) -> Self {
67        Self { input, output }
68    }
69    /// Send lines of text (terminated by a single dot).
70    pub async fn send_lines(&mut self, lines: &[String]) -> ClientResult<&mut Self> {
71        const END_OF_DATA: [&str; 1] = ["."];
72        write_lines_tokio(
73            &mut self.output,
74            lines
75                .iter()
76                .map(|s| s.as_str())
77                .collect::<Vec<&str>>()
78                .as_slice(),
79        )
80        .await?;
81        flush_lines_tokio(&mut self.output, &END_OF_DATA).await?;
82        Ok(self)
83    }
84    pub async fn send_line(&mut self, line: &str) -> ClientResult<&mut Self> {
85        self.send(Request::SendLine(line.to_string())).await
86    }
87    /// Receive answer from server
88    async fn receive_answer(&mut self, lines: Option<&mut Vec<String>>) -> ClientStatus {
89        crate::protocol::receive_answer_tokio(&mut self.input, lines).await
90    }
91    /// Receive one response.
92    pub async fn receive(&mut self) -> ClientResult<Response> {
93        const MSG_CURSOR_SET_FIRST: &str = "OK CURSOR SET FIRST";
94        let mut lines = Vec::new();
95        let status = self.receive_answer(Some(&mut lines)).await?;
96        match status.code {
97            OK_LANGUAGE_SET => Ok(Response::LanguageSet),
98            OK_PRIORITY_SET => Ok(Response::PrioritySet),
99            OK_RATE_SET => Ok(Response::RateSet),
100            OK_PITCH_SET => Ok(Response::PitchSet),
101            OK_PUNCTUATION_SET => Ok(Response::PunctuationSet),
102            OK_CAP_LET_RECOGN_SET => Ok(Response::CapLetRecognSet),
103            OK_SPELLING_SET => Ok(Response::SpellingSet),
104            OK_CLIENT_NAME_SET => Ok(Response::ClientNameSet),
105            OK_VOICE_SET => Ok(Response::VoiceSet),
106            OK_STOPPED => Ok(Response::Stopped),
107            OK_PAUSED => Ok(Response::Paused),
108            OK_RESUMED => Ok(Response::Resumed),
109            OK_CANCELED => Ok(Response::Canceled),
110            OK_TABLE_SET => Ok(Response::TableSet),
111            OK_OUTPUT_MODULE_SET => Ok(Response::OutputModuleSet),
112            OK_PAUSE_CONTEXT_SET => Ok(Response::PauseContextSet),
113            OK_VOLUME_SET => Ok(Response::VolumeSet),
114            OK_SSML_MODE_SET => Ok(Response::SsmlModeSet),
115            // Warning OK_CUR_SET_FIRST == OK_NOTIFICATION_SET == 220. Matching message to make the difference
116            OK_NOTIFICATION_SET => {
117                if status.message == MSG_CURSOR_SET_FIRST {
118                    //OK_CUR_SET_FIRST => Ok(Response::HistoryCurSetFirst)
119                    Ok(Response::HistoryCurSetFirst)
120                } else {
121                    Ok(Response::NotificationSet)
122                }
123            }
124            OK_CUR_SET_LAST => Ok(Response::HistoryCurSetLast),
125            OK_CUR_SET_POS => Ok(Response::HistoryCurSetPos),
126            OK_PITCH_RANGE_SET => Ok(Response::PitchRangeSet),
127            OK_DEBUG_SET => Ok(Response::DebugSet),
128            OK_CUR_MOV_FOR => Ok(Response::HistoryCurMoveFor),
129            OK_CUR_MOV_BACK => Ok(Response::HistoryCurMoveBack),
130            OK_MESSAGE_QUEUED => Ok(Response::MessageQueued),
131            OK_SND_ICON_QUEUED => Ok(Response::SoundIconQueued),
132            OK_MSG_CANCELED => Ok(Response::MessageCanceled),
133            OK_RECEIVING_DATA => Ok(Response::ReceivingData),
134            OK_BYE => Ok(Response::Bye),
135            OK_CLIENTS_LIST_SENT => Ok(Response::HistoryClientListSent(parse_typed_lines::<
136                HistoryClientStatus,
137            >(&lines)?)),
138            OK_MSGS_LIST_SENT => Ok(Response::HistoryMsgsListSent(lines)),
139            OK_LAST_MSG => Ok(Response::HistoryLastMsg(parse_single_value(&lines)?)),
140            OK_CUR_POS_RET => Ok(Response::HistoryCurPosRet(parse_single_value(&lines)?)),
141            OK_TABLE_LIST_SENT => Ok(Response::TableListSent(lines)),
142            OK_CLIENT_ID_SENT => Ok(Response::HistoryClientIdSent(parse_single_integer(&lines)?)),
143            OK_MSG_TEXT_SENT => Ok(Response::MessageTextSent),
144            OK_HELP_SENT => Ok(Response::HelpSent(lines)),
145            OK_VOICES_LIST_SENT => Ok(Response::VoicesListSent(
146                parse_typed_lines::<SynthesisVoice>(&lines)?,
147            )),
148            OK_OUTPUT_MODULES_LIST_SENT => Ok(Response::OutputModulesListSent(lines)),
149            OK_GET => Ok(Response::Get(parse_single_value(&lines)?)),
150            OK_INSIDE_BLOCK => Ok(Response::InsideBlock),
151            OK_OUTSIDE_BLOCK => Ok(Response::OutsideBlock),
152            OK_NOT_IMPLEMENTED => Ok(Response::NotImplemented),
153            EVENT_INDEX_MARK => match lines.len() {
154                0 | 1 | 2 => Err(ClientError::TooFewLines),
155                3 => Ok(Response::EventIndexMark(
156                    parse_event_id(&lines)?,
157                    lines[2].to_owned(),
158                )),
159                _ => Err(ClientError::TooManyLines),
160            },
161            EVENT_BEGIN => Ok(Response::EventBegin(parse_event_id(&lines)?)),
162            EVENT_END => Ok(Response::EventEnd(parse_event_id(&lines)?)),
163            EVENT_CANCELED => Ok(Response::EventCanceled(parse_event_id(&lines)?)),
164            EVENT_PAUSED => Ok(Response::EventPaused(parse_event_id(&lines)?)),
165            EVENT_RESUMED => Ok(Response::EventResumed(parse_event_id(&lines)?)),
166            _ => panic!("error should have been caught earlier"),
167        }
168    }
169    /// Send a request
170    pub async fn send(&mut self, request: Request) -> ClientResult<&mut Self> {
171        match request {
172            Request::SetName(client_name) => send_one_line!(
173                self,
174                "SET self CLIENT_NAME {}:{}:{}",
175                client_name.user,
176                client_name.application,
177                client_name.component
178            ),
179            Request::Speak => send_one_line!(self, "SPEAK"),
180            Request::SendLine(line) => send_one_line!(self, &line).map(|_| ()),
181            Request::SendLines(lines) => self.send_lines(&lines).await.map(|_| ()),
182            Request::SpeakChar(ch) => send_one_line!(self, "CHAR {}", ch),
183            Request::SpeakKey(key) => send_one_line!(self, "KEY {}", key),
184            Request::Stop(scope) => send_one_line!(self, "STOP {}", scope),
185            Request::Cancel(scope) => send_one_line!(self, "CANCEL {}", scope),
186            Request::Pause(scope) => send_one_line!(self, "PAUSE {}", scope),
187            Request::Resume(scope) => send_one_line!(self, "RESUME {}", scope),
188            Request::SetPriority(prio) => send_one_line!(self, "SET self PRIORITY {}", prio),
189            Request::SetDebug(value) => send_toggle!(self, "SET all DEBUG {}", value),
190            Request::SetOutputModule(scope, value) => {
191                send_one_line!(self, "SET {} OUTPUT_MODULE {}", scope, value)
192            }
193            Request::GetOutputModule => send_one_line!(self, "GET OUTPUT_MODULE"),
194            Request::ListOutputModules => send_one_line!(self, "LIST OUTPUT_MODULES"),
195            Request::SetLanguage(scope, lang) => {
196                send_one_line!(self, "SET {} LANGUAGE {}", scope, lang)
197            }
198            Request::GetLanguage => send_one_line!(self, "GET LANGUAGE"),
199            Request::SetSsmlMode(value) => send_toggle!(self, "SET self SSML_MODE {}", value),
200            Request::SetPunctuationMode(scope, mode) => {
201                send_one_line!(self, "SET {} PUNCTUATION {}", scope, mode)
202            }
203            Request::SetSpelling(scope, value) => {
204                send_toggle!(self, "SET {} SPELLING {}", scope, value)
205            }
206            Request::SetCapitalLettersRecognitionMode(scope, mode) => {
207                send_one_line!(self, "SET {} CAP_LET_RECOGN {}", scope, mode)
208            }
209            Request::SetVoiceType(scope, value) => {
210                send_one_line!(self, "SET {} VOICE_TYPE {}", scope, value)
211            }
212            Request::GetVoiceType => send_one_line!(self, "GET VOICE_TYPE"),
213            Request::ListVoiceTypes => send_one_line!(self, "LIST VOICES"),
214            Request::SetSynthesisVoice(scope, value) => {
215                send_one_line!(self, "SET {} SYNTHESIS_VOICE {}", scope, value)
216            }
217            Request::ListSynthesisVoices => send_one_line!(self, "LIST SYNTHESIS_VOICES"),
218            Request::SetRate(scope, value) => send_range!(self, "SET {} RATE {}", scope, value),
219            Request::GetRate => send_one_line!(self, "GET RATE"),
220            Request::SetPitch(scope, value) => send_range!(self, "SET {} PITCH {}", scope, value),
221            Request::GetPitch => send_one_line!(self, "GET PITCH"),
222            Request::SetVolume(scope, value) => {
223                send_range!(self, "SET {} VOLUME {}", scope, value)
224            }
225            Request::GetVolume => send_one_line!(self, "GET VOLUME"),
226            Request::SetPauseContext(scope, value) => {
227                send_one_line!(self, "SET {} PAUSE_CONTEXT {}", scope, value)
228            }
229            Request::SetHistory(scope, value) => {
230                send_toggle!(self, "SET {} HISTORY {}", scope, value)
231            }
232            Request::SetNotification(ntype, value) => {
233                send_toggle!(self, "SET self NOTIFICATION {} {}", ntype, value)
234            }
235            Request::Begin => send_one_line!(self, "BLOCK BEGIN"),
236            Request::End => send_one_line!(self, "BLOCK END"),
237            Request::HistoryGetClients => send_one_line!(self, "HISTORY GET CLIENT_LIST"),
238            Request::HistoryGetClientId => send_one_line!(self, "HISTORY GET CLIENT_ID"),
239            Request::HistoryGetClientMsgs(scope, start, number) => send_one_line!(
240                self,
241                "HISTORY GET CLIENT_MESSAGES {} {}_{}",
242                scope,
243                start,
244                number
245            ),
246            Request::HistoryGetLastMsgId => send_one_line!(self, "HISTORY GET LAST"),
247            Request::HistoryGetMsg(id) => send_one_line!(self, "HISTORY GET MESSAGE {}", id),
248            Request::HistoryCursorGet => send_one_line!(self, "HISTORY CURSOR GET"),
249            Request::HistoryCursorSet(scope, pos) => {
250                send_one_line!(self, "HISTORY CURSOR SET {} {}", scope, pos)
251            }
252            Request::HistoryCursorMove(direction) => {
253                send_one_line!(self, "HISTORY CURSOR {}", direction)
254            }
255            Request::HistorySpeak(id) => send_one_line!(self, "HISTORY SAY {}", id),
256            Request::HistorySort(direction, key) => {
257                send_one_line!(self, "HISTORY SORT {} {}", direction, key)
258            }
259            Request::HistorySetShortMsgLength(length) => {
260                send_one_line!(self, "HISTORY SET SHORT_MESSAGE_LENGTH {}", length)
261            }
262            Request::HistorySetMsgTypeOrdering(ordering) => {
263                send_one_line!(
264                    self,
265                    "HISTORY SET MESSAGE_TYPE_ORDERING \"{}\"",
266                    ordering
267                        .iter()
268                        .map(|x| x.to_string())
269                        .collect::<Vec<String>>()
270                        .join(" ")
271                )
272            }
273            Request::HistorySearch(scope, condition) => {
274                send_one_line!(self, "HISTORY SEARCH {} \"{}\"", scope, condition)
275            }
276            Request::Quit => send_one_line!(self, "QUIT"),
277        }?;
278        Ok(self)
279    }

682    /// Receive a notification
683    pub async fn receive_event(&mut self) -> ClientResult<Event> {
684        let mut lines = Vec::new();
685        self.receive_answer(Some(&mut lines))
686            .await
687            .and_then(|status| {
688                if lines.len() < 2 {
689                    Err(ClientError::unexpected_eof("event truncated"))
690                } else {
691                    let message = &lines[0];
692                    let client = &lines[1];
693                    match status.code {
694                        700 => {
695                            if lines.len() != 3 {
696                                Err(ClientError::unexpected_eof("index markevent truncated"))
697                            } else {
698                                let mark = lines[3].to_owned();
699                                Ok(Event::index_mark(mark, message, client))
700                            }
701                        }
702                        701 => Ok(Event::begin(message, client)),
703                        702 => Ok(Event::end(message, client)),
704                        703 => Ok(Event::cancel(message, client)),
705                        704 => Ok(Event::pause(message, client)),
706                        705 => Ok(Event::resume(message, client)),
707                        _ => Err(ClientError::invalid_data("wrong status code for event")),
708                    }
709                }
710            })
711    }
