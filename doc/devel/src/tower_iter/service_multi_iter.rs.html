7/// Useful for running a set of services with the same signature in parallel.
8///
9/// Note that although calling the [`ServiceMultiIter::into_future`] function seems to return a
10/// future that resolves to
11/// `Result<Vec<Result<S::Response, S::Error>>, S::Error>`, the outer error is gaurenteed never to be
12/// an error. It is [`std::convert::Infallible`].
13///
14/// Your three options for handling this are:
15///
16/// 1. Use [`Result::unwrap`] in the inner service.
17/// 2. Call [`Iterator::collect::<Result<Vec<T>, E>>()`] on the result of the future.
18///
19/// ```
20/// use core::{
21///   convert::Infallible,
22///   iter::repeat_n,
23///   future::IntoFuture,
24/// };
25/// use tower::service_fn;
26/// use futures_lite::future::block_on;
27/// use tower_iter::service_multi_iter::ServiceMultiIter;
28///
29/// async fn mul_2(i: u32) -> Result<u32, Infallible> {
30///   Ok(i * 2)
31/// }
32/// let mul_svc_one = service_fn(mul_2);
33/// let mul_svc_all = repeat_n(mul_svc_one, 5);
34/// let input = [5, 10, 15, 20, 25].into_iter();
35/// let fut = ServiceMultiIter::new(mul_svc_all, input).into_future();
36///
37/// assert_eq!(block_on(fut),
38///     vec![
39///         Ok(10),
40///         Ok(20),
41///         Ok(30),
42///         Ok(40),
43///         Ok(50)
44///     ]
45/// );
46/// ```
47#[derive(Clone)]
48pub struct ServiceMultiIter<Si, Ii, S, I> {
49	s_iter: Si,
50	i_iter: Ii,
51	_marker: PhantomData<(S, I)>,
52}
53impl<Si, Ii, S, I> ServiceMultiIter<Si, Ii, S, I> {
54	pub fn new(s_iter: Si, i_iter: Ii) -> Self {
55		ServiceMultiIter { s_iter, i_iter, _marker: PhantomData }
56	}
57}
58
59impl<Si, Ii, S, I> IntoFuture for ServiceMultiIter<Si, Ii, S, I>
60where
61	S: Clone + Service<I>,
62	Ii: Iterator<Item = I>,
63	Si: Iterator<Item = S>,
64{
65	type Output = Vec<Result<S::Response, S::Error>>;
66	type IntoFuture = JoinAll<<MapServiceCall<Zip<Si, Ii>, S, I> as Iterator>::Item>;
67	fn into_future(self) -> Self::IntoFuture {
68		join_all(self.s_iter.zip(self.i_iter).map_service_call())
69	}
70}
