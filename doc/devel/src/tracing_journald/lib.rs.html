1//! # tracing-journald
2//!
3//! Support for logging [`tracing`] events natively to [journald],
4//! preserving structured information.
5//!
6//! ## Overview
7//!
8//! [`tracing`] is a framework for instrumenting Rust programs to collect
9//! scoped, structured, and async-aware diagnostics. `tracing-journald` provides a
10//! [`tracing-subscriber::Layer`] implementation for logging `tracing` spans
11//! and events to [`systemd-journald`][journald], on Linux distributions that
12//! use `systemd`.
13//!
14//! *Compiler support: [requires `rustc` 1.49+][msrv]*
15//!
16//! [msrv]: #supported-rust-versions
17//! [`tracing`]: https://crates.io/crates/tracing
18//! [journald]: https://www.freedesktop.org/software/systemd/man/systemd-journald.service.html
19//!
20//! ## Supported Rust Versions
21//!
22//! Tracing is built against the latest stable release. The minimum supported
23//! version is 1.49. The current Tracing version is not guaranteed to build on
24//! Rust versions earlier than the minimum supported version.
25//!
26//! Tracing follows the same compiler support policies as the rest of the Tokio
27//! project. The current stable Rust compiler and the three most recent minor
28//! versions before it will always be supported. For example, if the current
29//! stable compiler version is 1.45, the minimum supported version will not be
30//! increased past 1.42, three minor versions prior. Increasing the minimum
31//! supported compiler version is not considered a semver breaking change as
32//! long as doing so complies with this policy.
33//!

56/// Sends events and their fields to journald
57///
58/// [journald conventions] for structured field names differ from typical tracing idioms, and journald
59/// discards fields which violate its conventions. Hence, this layer automatically sanitizes field
60/// names by translating `.`s into `_`s, stripping leading `_`s and non-ascii-alphanumeric
61/// characters other than `_`, and upcasing.
62///
63/// Levels are mapped losslessly to journald `PRIORITY` values as follows:
64///
65/// - `ERROR` => Error (3)
66/// - `WARN` => Warning (4)
67/// - `INFO` => Notice (5)
68/// - `DEBUG` => Informational (6)
69/// - `TRACE` => Debug (7)
70///
71/// The standard journald `CODE_LINE` and `CODE_FILE` fields are automatically emitted. A `TARGET`
72/// field is emitted containing the event's target.
73///
74/// For events recorded inside spans, an additional `SPAN_NAME` field is emitted with the name of
75/// each of the event's parent spans.
76///
77/// User-defined fields other than the event `message` field have a prefix applied by default to
78/// prevent collision with standard fields.
79///
80/// [journald conventions]: https://www.freedesktop.org/software/systemd/man/systemd.journal-fields.html
81pub struct Layer {
82    #[cfg(unix)]
83    socket: UnixDatagram,
84    field_prefix: Option<String>,
85    syslog_identifier: String,
86}
87
88#[cfg(unix)]
89const JOURNALD_PATH: &str = "/run/systemd/journal/socket";
90
91impl Layer {
92    /// Construct a journald layer
93    ///
94    /// Fails if the journald socket couldn't be opened. Returns a `NotFound` error unconditionally
95    /// in non-Unix environments.
96    pub fn new() -> io::Result<Self> {
97        #[cfg(unix)]
98        {
99            let socket = UnixDatagram::unbound()?;
100            let layer = Self {
101                socket,
102                field_prefix: Some("F".into()),
103                syslog_identifier: std::env::current_exe()
104                    .ok()
105                    .as_ref()
106                    .and_then(|p| p.file_name())
107                    .map(|n| n.to_string_lossy().into_owned())
108                    // If we fail to get the name of the current executable fall back to an empty string.
109                    .unwrap_or_else(String::new),
110            };
111            // Check that we can talk to journald, by sending empty payload which journald discards.
112            // However if the socket didn't exist or if none listened we'd get an error here.
113            layer.send_payload(&[])?;
114            Ok(layer)
115        }
116        #[cfg(not(unix))]
117        Err(io::Error::new(
118            io::ErrorKind::NotFound,
119            "journald does not exist in this environment",
120        ))
121    }
122
123    /// Sets the prefix to apply to names of user-defined fields other than the event `message`
124    /// field. Defaults to `Some("F")`.
125    pub fn with_field_prefix(mut self, x: Option<String>) -> Self {
126        self.field_prefix = x;
127        self
128    }
129
130    /// Sets the syslog identifier for this logger.
131    ///
132    /// The syslog identifier comes from the classic syslog interface (`openlog()`
133    /// and `syslog()`) and tags log entries with a given identifier.
134    /// Systemd exposes it in the `SYSLOG_IDENTIFIER` journal field, and allows
135    /// filtering log messages by syslog identifier with `journalctl -t`.
136    /// Unlike the unit (`journalctl -u`) this field is not trusted, i.e. applications
137    /// can set it freely, and use it e.g. to further categorize log entries emitted under
138    /// the same systemd unit or in the same process.  It also allows to filter for log
139    /// entries of processes not started in their own unit.
140    ///
141    /// See [Journal Fields](https://www.freedesktop.org/software/systemd/man/systemd.journal-fields.html)
142    /// and [journalctl](https://www.freedesktop.org/software/systemd/man/journalctl.html)
143    /// for more information.
144    ///
145    /// Defaults to the file name of the executable of the current process, if any.
146    pub fn with_syslog_identifier(mut self, identifier: String) -> Self {
147        self.syslog_identifier = identifier;
148        self
149    }
150
151    /// Returns the syslog identifier in use.
152    pub fn syslog_identifier(&self) -> &str {
153        &self.syslog_identifier
154    }
155
156    #[cfg(not(unix))]
157    fn send_payload(&self, _opayload: &[u8]) -> io::Result<()> {
158        Err(io::Error::new(
159            io::ErrorKind::Unsupported,
160            "journald not supported on non-Unix",
161        ))
162    }
163
164    #[cfg(unix)]
165    fn send_payload(&self, payload: &[u8]) -> io::Result<usize> {
166        self.socket
167            .send_to(payload, JOURNALD_PATH)
168            .or_else(|error| {
169                if Some(libc::EMSGSIZE) == error.raw_os_error() {
170                    self.send_large_payload(payload)
171                } else {
172                    Err(error)
173                }
174            })
175    }
176
177    #[cfg(all(unix, not(target_os = "linux")))]
178    fn send_large_payload(&self, _payload: &[u8]) -> io::Result<usize> {
179        Err(io::Error::new(
180            io::ErrorKind::Unsupported,
181            "Large payloads not supported on non-Linux OS",
182        ))
183    }
184
185    /// Send large payloads to journald via a memfd.
186    #[cfg(target_os = "linux")]
187    fn send_large_payload(&self, payload: &[u8]) -> io::Result<usize> {
188        // If the payload's too large for a single datagram, send it through a memfd, see
189        // https://systemd.io/JOURNAL_NATIVE_PROTOCOL/
190        use std::os::unix::prelude::AsRawFd;
191        // Write the whole payload to a memfd
192        let mut mem = memfd::create_sealable()?;
193        mem.write_all(payload)?;
194        // Fully seal the memfd to signal journald that its backing data won't resize anymore
195        // and so is safe to mmap.
196        memfd::seal_fully(mem.as_raw_fd())?;
197        socket::send_one_fd_to(&self.socket, mem.as_raw_fd(), JOURNALD_PATH)
198    }
199}

208impl<S> tracing_subscriber::Layer<S> for Layer
209where
210    S: Subscriber + for<'span> LookupSpan<'span>,
211{
212    fn on_new_span(&self, attrs: &Attributes, id: &Id, ctx: Context<'_, S>) {
213        let span = ctx.span(id).expect("unknown span");
214        let mut buf = Vec::with_capacity(256);
215
216        writeln!(buf, "SPAN_NAME").unwrap();
217        put_value(&mut buf, span.name().as_bytes());
218        put_metadata(&mut buf, span.metadata(), Some("SPAN_"));
219
220        attrs.record(&mut SpanVisitor {
221            buf: &mut buf,
222            field_prefix: self.field_prefix.as_deref(),
223        });
224
225        span.extensions_mut().insert(SpanFields(buf));
226    }
227
228    fn on_record(&self, id: &Id, values: &Record, ctx: Context<S>) {
229        let span = ctx.span(id).expect("unknown span");
230        let mut exts = span.extensions_mut();
231        let buf = &mut exts.get_mut::<SpanFields>().expect("missing fields").0;
232        values.record(&mut SpanVisitor {
233            buf,
234            field_prefix: self.field_prefix.as_deref(),
235        });
236    }
237
238    fn on_event(&self, event: &Event, ctx: Context<S>) {
239        let mut buf = Vec::with_capacity(256);
240
241        // Record span fields
242        for span in ctx
243            .lookup_current()
244            .into_iter()
245            .flat_map(|span| span.scope().from_root())
246        {
247            let exts = span.extensions();
248            let fields = exts.get::<SpanFields>().expect("missing fields");
249            buf.extend_from_slice(&fields.0);
250        }
251
252        // Record event fields
253        put_priority(&mut buf, event.metadata());
254        put_metadata(&mut buf, event.metadata(), None);
255        put_field_length_encoded(&mut buf, "SYSLOG_IDENTIFIER", |buf| {
256            write!(buf, "{}", self.syslog_identifier).unwrap()
257        });
258
259        event.record(&mut EventVisitor::new(
260            &mut buf,
261            self.field_prefix.as_deref(),
262        ));
263
264        // At this point we can't handle the error anymore so just ignore it.
265        let _ = self.send_payload(&buf);
266    }
267}

269struct SpanFields(Vec<u8>);
270
271struct SpanVisitor<'a> {
272    buf: &'a mut Vec<u8>,
273    field_prefix: Option<&'a str>,
274}
275
276impl SpanVisitor<'_> {
277    fn put_span_prefix(&mut self) {
278        if let Some(prefix) = self.field_prefix {
279            self.buf.extend_from_slice(prefix.as_bytes());
280            self.buf.push(b'_');
281        }
282    }
283}
284
285impl Visit for SpanVisitor<'_> {
286    fn record_str(&mut self, field: &Field, value: &str) {
287        self.put_span_prefix();
288        put_field_length_encoded(self.buf, field.name(), |buf| {
289            buf.extend_from_slice(value.as_bytes())
290        });
291    }
292
293    fn record_debug(&mut self, field: &Field, value: &dyn fmt::Debug) {
294        self.put_span_prefix();
295        put_field_length_encoded(self.buf, field.name(), |buf| {
296            write!(buf, "{:?}", value).unwrap()
297        });
298    }
299}

301/// Helper for generating the journal export format, which is consumed by journald:
302/// https://www.freedesktop.org/wiki/Software/systemd/export/
303struct EventVisitor<'a> {
304    buf: &'a mut Vec<u8>,
305    prefix: Option<&'a str>,
306}
307
308impl<'a> EventVisitor<'a> {
309    fn new(buf: &'a mut Vec<u8>, prefix: Option<&'a str>) -> Self {
310        Self { buf, prefix }
311    }
312
313    fn put_prefix(&mut self, field: &Field) {
314        if let Some(prefix) = self.prefix {
315            if field.name() != "message" {
316                // message maps to the standard MESSAGE field so don't prefix it
317                self.buf.extend_from_slice(prefix.as_bytes());
318                self.buf.push(b'_');
319            }
320        }
321    }
322}
323
324impl Visit for EventVisitor<'_> {
325    fn record_str(&mut self, field: &Field, value: &str) {
326        self.put_prefix(field);
327        put_field_length_encoded(self.buf, field.name(), |buf| {
328            buf.extend_from_slice(value.as_bytes())
329        });
330    }
331
332    fn record_debug(&mut self, field: &Field, value: &dyn fmt::Debug) {
333        self.put_prefix(field);
334        put_field_length_encoded(self.buf, field.name(), |buf| {
335            write!(buf, "{:?}", value).unwrap()
336        });
337    }
338}

340fn put_priority(buf: &mut Vec<u8>, meta: &Metadata) {
341    put_field_wellformed(
342        buf,
343        "PRIORITY",
344        match *meta.level() {
345            Level::ERROR => b"3",
346            Level::WARN => b"4",
347            Level::INFO => b"5",
348            Level::DEBUG => b"6",
349            Level::TRACE => b"7",
350        },
351    );
352}
353
354fn put_metadata(buf: &mut Vec<u8>, meta: &Metadata, prefix: Option<&str>) {
355    if let Some(prefix) = prefix {
356        write!(buf, "{}", prefix).unwrap();
357    }
358    put_field_wellformed(buf, "TARGET", meta.target().as_bytes());
359    if let Some(file) = meta.file() {
360        if let Some(prefix) = prefix {
361            write!(buf, "{}", prefix).unwrap();
362        }
363        put_field_wellformed(buf, "CODE_FILE", file.as_bytes());
364    }
365    if let Some(line) = meta.line() {
366        if let Some(prefix) = prefix {
367            write!(buf, "{}", prefix).unwrap();
368        }
369        // Text format is safe as a line number can't possibly contain anything funny
370        writeln!(buf, "CODE_LINE={}", line).unwrap();
371    }
372}

374/// Append a sanitized and length-encoded field into `buf`.
375///
376/// Unlike `put_field_wellformed` this function handles arbitrary field names and values.
377///
378/// `name` denotes the field name. It gets sanitized before being appended to `buf`.
379///
380/// `write_value` is invoked with `buf` as argument to append the value data to `buf`.  It must
381/// not delete from `buf`, but may append arbitrary data.  This function then determines the length
382/// of the data written and adds it in the appropriate place in `buf`.
383fn put_field_length_encoded(buf: &mut Vec<u8>, name: &str, write_value: impl FnOnce(&mut Vec<u8>)) {
384    sanitize_name(name, buf);
385    buf.push(b'\n');
386    buf.extend_from_slice(&[0; 8]); // Length tag, to be populated
387    let start = buf.len();
388    write_value(buf);
389    let end = buf.len();
390    buf[start - 8..start].copy_from_slice(&((end - start) as u64).to_le_bytes());
391    buf.push(b'\n');
392}
393
394/// Mangle a name into journald-compliant form
395fn sanitize_name(name: &str, buf: &mut Vec<u8>) {
396    buf.extend(
397        name.bytes()
398            .map(|c| if c == b'.' { b'_' } else { c })
399            .skip_while(|&c| c == b'_')
400            .filter(|&c| c == b'_' || char::from(c).is_ascii_alphanumeric())
401            .map(|c| char::from(c).to_ascii_uppercase() as u8),
402    );
403}
404
405/// Append arbitrary data with a well-formed name and value.
406///
407/// `value` must not contain an internal newline, because this function writes
408/// `value` in the new-line separated format.
409///
410/// For a "newline-safe" variant, see `put_field_length_encoded`.
411fn put_field_wellformed(buf: &mut Vec<u8>, name: &str, value: &[u8]) {
412    buf.extend_from_slice(name.as_bytes());
413    buf.push(b'\n');
414    put_value(buf, value);
415}
416
417/// Write the value portion of a key-value pair, in newline separated format.
418///
419/// `value` must not contain an internal newline.
420///
421/// For a "newline-safe" variant, see `put_field_length_encoded`.
422fn put_value(buf: &mut Vec<u8>, value: &[u8]) {
423    buf.extend_from_slice(&(value.len() as u64).to_le_bytes());
424    buf.extend_from_slice(value);
425    buf.push(b'\n');
