1//! # tracing-journald
2//!
3//! Support for logging [`tracing`] events natively to [journald],
4//! preserving structured information.
5//!
6//! ## Overview
7//!
8//! [`tracing`] is a framework for instrumenting Rust programs to collect
9//! scoped, structured, and async-aware diagnostics. `tracing-journald` provides a
10//! [`tracing-subscriber::Layer`] implementation for logging `tracing` spans
11//! and events to [`systemd-journald`][journald], on Linux distributions that
12//! use `systemd`.
13//!
14//! *Compiler support: [requires `rustc` 1.63+][msrv]*
15//!
16//! [msrv]: #supported-rust-versions
17//! [`tracing`]: https://crates.io/crates/tracing
18//! [journald]: https://www.freedesktop.org/software/systemd/man/systemd-journald.service.html
19//!
20//! ## Supported Rust Versions
21//!
22//! Tracing is built against the latest stable release. The minimum supported
23//! version is 1.63. The current Tracing version is not guaranteed to build on
24//! Rust versions earlier than the minimum supported version.
25//!
26//! Tracing follows the same compiler support policies as the rest of the Tokio
27//! project. The current stable Rust compiler and the three most recent minor
28//! versions before it will always be supported. For example, if the current
29//! stable compiler version is 1.69, the minimum supported version will not be
30//! increased past 1.66, three minor versions prior. Increasing the minimum
31//! supported compiler version is not considered a semver breaking change as
32//! long as doing so complies with this policy.
33//!

56/// Sends events and their fields to journald
57///
58/// [journald conventions] for structured field names differ from typical tracing idioms, and journald
59/// discards fields which violate its conventions. Hence, this layer automatically sanitizes field
60/// names by translating `.`s into `_`s, stripping leading `_`s and non-ascii-alphanumeric
61/// characters other than `_`, and upcasing.
62///
63/// By default, levels are mapped losslessly to journald `PRIORITY` values as follows:
64///
65/// - `ERROR` => Error (3)
66/// - `WARN` => Warning (4)
67/// - `INFO` => Notice (5)
68/// - `DEBUG` => Informational (6)
69/// - `TRACE` => Debug (7)
70///
71/// These mappings can be changed with [`Subscriber::with_priority_mappings`].
72///
73/// The standard journald `CODE_LINE` and `CODE_FILE` fields are automatically emitted. A `TARGET`
74/// field is emitted containing the event's target.
75///
76/// For events recorded inside spans, an additional `SPAN_NAME` field is emitted with the name of
77/// each of the event's parent spans.
78///
79/// User-defined fields other than the event `message` field have a prefix applied by default to
80/// prevent collision with standard fields.
81///
82/// [journald conventions]: https://www.freedesktop.org/software/systemd/man/systemd.journal-fields.html
83pub struct Layer {
84    #[cfg(unix)]
85    socket: UnixDatagram,
86    field_prefix: Option<String>,
87    syslog_identifier: String,
88    additional_fields: Vec<u8>,
89    priority_mappings: PriorityMappings,
90}
91
92#[cfg(unix)]
93const JOURNALD_PATH: &str = "/run/systemd/journal/socket";

95impl Layer {
96    /// Construct a journald layer
97    ///
98    /// Fails if the journald socket couldn't be opened. Returns a `NotFound` error unconditionally
99    /// in non-Unix environments.
100    pub fn new() -> io::Result<Self> {
101        #[cfg(unix)]
102        {
103            let socket = UnixDatagram::unbound()?;
104            let layer = Self {
105                socket,
106                field_prefix: Some("F".into()),
107                syslog_identifier: std::env::current_exe()
108                    .ok()
109                    .as_ref()
110                    .and_then(|p| p.file_name())
111                    .map(|n| n.to_string_lossy().into_owned())
112                    // If we fail to get the name of the current executable fall back to an empty string.
113                    .unwrap_or_default(),
114                additional_fields: Vec::new(),
115                priority_mappings: PriorityMappings::new(),
116            };
117            // Check that we can talk to journald, by sending empty payload which journald discards.
118            // However if the socket didn't exist or if none listened we'd get an error here.
119            layer.send_payload(&[])?;
120            Ok(layer)
121        }
122        #[cfg(not(unix))]
123        Err(io::Error::new(
124            io::ErrorKind::NotFound,
125            "journald does not exist in this environment",
126        ))
127    }
128
129    /// Sets the prefix to apply to names of user-defined fields other than the event `message`
130    /// field. Defaults to `Some("F")`.
131    pub fn with_field_prefix(mut self, x: Option<String>) -> Self {
132        self.field_prefix = x;
133        self
134    }

136    /// Sets how [`tracing_core::Level`]s are mapped to [journald priorities](Priority).
137    ///
138    /// # Examples
139    ///
140    /// ```rust
141    /// use tracing_journald::{Priority, PriorityMappings};
142    /// use tracing_subscriber::prelude::*;
143    /// use tracing::error;
144    ///
145    /// let registry = tracing_subscriber::registry();
146    /// match tracing_journald::layer() {
147    ///     Ok(layer) => {
148    ///         registry.with(
149    ///             layer
150    ///                 // We can tweak the mappings between the trace level and
151    ///                 // the journal priorities.
152    ///                 .with_priority_mappings(PriorityMappings {
153    ///                     info: Priority::Informational,
154    ///                     ..PriorityMappings::new()
155    ///                 }),
156    ///         );
157    ///     }
158    ///     // journald is typically available on Linux systems, but nowhere else. Portable software
159    ///     // should handle its absence gracefully.
160    ///     Err(e) => {
161    ///         registry.init();
162    ///         error!("couldn't connect to journald: {}", e);
163    ///     }
164    /// }
165    /// ```
166    pub fn with_priority_mappings(mut self, mappings: PriorityMappings) -> Self {
167        self.priority_mappings = mappings;
168        self
169    }

171    /// Sets the syslog identifier for this logger.
172    ///
173    /// The syslog identifier comes from the classic syslog interface (`openlog()`
174    /// and `syslog()`) and tags log entries with a given identifier.
175    /// Systemd exposes it in the `SYSLOG_IDENTIFIER` journal field, and allows
176    /// filtering log messages by syslog identifier with `journalctl -t`.
177    /// Unlike the unit (`journalctl -u`) this field is not trusted, i.e. applications
178    /// can set it freely, and use it e.g. to further categorize log entries emitted under
179    /// the same systemd unit or in the same process.  It also allows to filter for log
180    /// entries of processes not started in their own unit.
181    ///
182    /// See [Journal Fields](https://www.freedesktop.org/software/systemd/man/systemd.journal-fields.html)
183    /// and [journalctl](https://www.freedesktop.org/software/systemd/man/journalctl.html)
184    /// for more information.
185    ///
186    /// Defaults to the file name of the executable of the current process, if any.
187    pub fn with_syslog_identifier(mut self, identifier: String) -> Self {
188        self.syslog_identifier = identifier;
189        self
190    }
191
192    /// Adds fields that will get be passed to journald with every log entry.
193    ///
194    /// The input values of this function are interpreted as `(field, value)` pairs.
195    ///
196    /// This can for example be used to configure the syslog facility.
197    /// See [Journal Fields](https://www.freedesktop.org/software/systemd/man/systemd.journal-fields.html)
198    /// and [journalctl](https://www.freedesktop.org/software/systemd/man/journalctl.html)
199    /// for more information.
200    ///
201    /// Fields specified using this method will be added to the journald
202    /// message alongside fields generated from the event's fields, its
203    /// metadata, and the span context. If the name of a field provided using
204    /// this method is the same as the name of a field generated by the
205    /// layer, both fields will be sent to journald.
206    ///
207    /// ```no_run
208    /// # use tracing_journald::Layer;
209    /// let layer = Layer::new()
210    ///     .unwrap()
211    ///     .with_custom_fields([("SYSLOG_FACILITY", "17")]);
212    /// ```
213    ///
214    pub fn with_custom_fields<T: AsRef<str>, U: AsRef<[u8]>>(
215        mut self,
216        fields: impl IntoIterator<Item = (T, U)>,
217    ) -> Self {
218        for (name, value) in fields {
219            put_field_length_encoded(&mut self.additional_fields, name.as_ref(), |buf| {
220                buf.extend_from_slice(value.as_ref())
221            })
222        }
223        self
224    }
225
226    /// Returns the syslog identifier in use.
227    pub fn syslog_identifier(&self) -> &str {
228        &self.syslog_identifier
229    }

231    #[cfg(not(unix))]
232    fn send_payload(&self, _opayload: &[u8]) -> io::Result<()> {
233        Err(io::Error::new(
234            io::ErrorKind::Other,
235            "journald not supported on non-Unix",
236        ))
237    }
238
239    #[cfg(unix)]
240    fn send_payload(&self, payload: &[u8]) -> io::Result<usize> {
241        self.socket
242            .send_to(payload, JOURNALD_PATH)
243            .or_else(|error| {
244                if Some(libc::EMSGSIZE) == error.raw_os_error() {
245                    self.send_large_payload(payload)
246                } else {
247                    Err(error)
248                }
249            })
250    }
251
252    #[cfg(all(unix, not(target_os = "linux")))]
253    fn send_large_payload(&self, _payload: &[u8]) -> io::Result<usize> {
254        Err(io::Error::new(
255            io::ErrorKind::Other,
256            "Large payloads not supported on non-Linux OS",
257        ))
258    }
259
260    /// Send large payloads to journald via a memfd.
261    #[cfg(target_os = "linux")]
262    fn send_large_payload(&self, payload: &[u8]) -> io::Result<usize> {
263        // If the payload's too large for a single datagram, send it through a memfd, see
264        // https://systemd.io/JOURNAL_NATIVE_PROTOCOL/
265        use std::os::unix::prelude::AsRawFd;
266        // Write the whole payload to a memfd
267        let mut mem = memfd::create_sealable()?;
268        mem.write_all(payload)?;
269        // Fully seal the memfd to signal journald that its backing data won't resize anymore
270        // and so is safe to mmap.
271        memfd::seal_fully(mem.as_raw_fd())?;
272        socket::send_one_fd_to(&self.socket, mem.as_raw_fd(), JOURNALD_PATH)
273    }
274
275    fn put_priority(&self, buf: &mut Vec<u8>, meta: &Metadata) {
276        put_field_wellformed(
277            buf,
278            "PRIORITY",
279            &[match *meta.level() {
280                Level::ERROR => self.priority_mappings.error as u8,
281                Level::WARN => self.priority_mappings.warn as u8,
282                Level::INFO => self.priority_mappings.info as u8,
283                Level::DEBUG => self.priority_mappings.debug as u8,
284                Level::TRACE => self.priority_mappings.trace as u8,
285            }],
286        );
287    }
288}

297impl<S> tracing_subscriber::Layer<S> for Layer
298where
299    S: Subscriber + for<'span> LookupSpan<'span>,
300{
301    fn on_new_span(&self, attrs: &Attributes, id: &Id, ctx: Context<'_, S>) {
302        let span = ctx.span(id).expect("unknown span");
303        let mut buf = Vec::with_capacity(256);
304
305        writeln!(buf, "SPAN_NAME").unwrap();
306        put_value(&mut buf, span.name().as_bytes());
307        put_metadata(&mut buf, span.metadata(), Some("SPAN_"));
308
309        attrs.record(&mut SpanVisitor {
310            buf: &mut buf,
311            field_prefix: self.field_prefix.as_deref(),
312        });
313
314        span.extensions_mut().insert(SpanFields(buf));
315    }
316
317    fn on_record(&self, id: &Id, values: &Record, ctx: Context<S>) {
318        let span = ctx.span(id).expect("unknown span");
319        let mut exts = span.extensions_mut();
320        let buf = &mut exts.get_mut::<SpanFields>().expect("missing fields").0;
321        values.record(&mut SpanVisitor {
322            buf,
323            field_prefix: self.field_prefix.as_deref(),
324        });
325    }
326
327    fn on_event(&self, event: &Event, ctx: Context<S>) {
328        let mut buf = Vec::with_capacity(256);
329
330        // Record span fields
331        for span in ctx
332            .lookup_current()
333            .into_iter()
334            .flat_map(|span| span.scope().from_root())
335        {
336            let exts = span.extensions();
337            let fields = exts.get::<SpanFields>().expect("missing fields");
338            buf.extend_from_slice(&fields.0);
339        }
340
341        // Record event fields
342        self.put_priority(&mut buf, event.metadata());
343        put_metadata(&mut buf, event.metadata(), None);
344        put_field_length_encoded(&mut buf, "SYSLOG_IDENTIFIER", |buf| {
345            write!(buf, "{}", self.syslog_identifier).unwrap()
346        });
347        buf.extend_from_slice(&self.additional_fields);
348
349        event.record(&mut EventVisitor::new(
350            &mut buf,
351            self.field_prefix.as_deref(),
352        ));
353
354        // At this point we can't handle the error anymore so just ignore it.
355        let _ = self.send_payload(&buf);
356    }
357}

359struct SpanFields(Vec<u8>);

361struct SpanVisitor<'a> {
362    buf: &'a mut Vec<u8>,
363    field_prefix: Option<&'a str>,
364}
365
366impl SpanVisitor<'_> {
367    fn put_span_prefix(&mut self) {
368        if let Some(prefix) = self.field_prefix {
369            self.buf.extend_from_slice(prefix.as_bytes());
370            self.buf.push(b'_');
371        }
372    }
373}
374
375impl Visit for SpanVisitor<'_> {
376    fn record_str(&mut self, field: &Field, value: &str) {
377        self.put_span_prefix();
378        put_field_length_encoded(self.buf, field.name(), |buf| {
379            buf.extend_from_slice(value.as_bytes())
380        });
381    }
382
383    fn record_debug(&mut self, field: &Field, value: &dyn fmt::Debug) {
384        self.put_span_prefix();
385        put_field_length_encoded(self.buf, field.name(), |buf| {
386            write!(buf, "{:?}", value).unwrap()
387        });
388    }
389}

391/// Helper for generating the journal export format, which is consumed by journald:
392/// https://www.freedesktop.org/wiki/Software/systemd/export/
393struct EventVisitor<'a> {
394    buf: &'a mut Vec<u8>,
395    prefix: Option<&'a str>,
396}
397
398impl<'a> EventVisitor<'a> {
399    fn new(buf: &'a mut Vec<u8>, prefix: Option<&'a str>) -> Self {
400        Self { buf, prefix }
401    }
402
403    fn put_prefix(&mut self, field: &Field) {
404        if let Some(prefix) = self.prefix {
405            if field.name() != "message" {
406                // message maps to the standard MESSAGE field so don't prefix it
407                self.buf.extend_from_slice(prefix.as_bytes());
408                self.buf.push(b'_');
409            }
410        }
411    }
412}
413
414impl Visit for EventVisitor<'_> {
415    fn record_str(&mut self, field: &Field, value: &str) {
416        self.put_prefix(field);
417        put_field_length_encoded(self.buf, field.name(), |buf| {
418            buf.extend_from_slice(value.as_bytes())
419        });
420    }
421
422    fn record_debug(&mut self, field: &Field, value: &dyn fmt::Debug) {
423        self.put_prefix(field);
424        put_field_length_encoded(self.buf, field.name(), |buf| {
425            write!(buf, "{:?}", value).unwrap()
426        });
427    }
428}

430/// A priority (called "severity code" by syslog) is used to mark the
431/// importance of a message.
432///
433/// Descriptions and examples are taken from the [Arch Linux wiki].
434/// Priorities are also documented in the
435/// [section 6.2.1 of the Syslog protocol RFC][syslog].
436///
437/// [Arch Linux wiki]: https://wiki.archlinux.org/title/Systemd/Journal#Priority_level
438/// [syslog]: https://www.rfc-editor.org/rfc/rfc5424#section-6.2.1
439#[derive(Debug, Clone, Copy, Hash, PartialEq, Eq)]
440#[repr(u8)]
441pub enum Priority {
442    /// System is unusable.
443    ///
444    /// Examples:
445    ///
446    /// - severe Kernel BUG
447    /// - systemd dumped core
448    ///
449    /// This level should not be used by applications.
450    Emergency = b'0',
451    /// Should be corrected immediately.
452    ///
453    /// Examples:
454    ///
455    /// - Vital subsystem goes out of work, data loss:
456    /// - `kernel: BUG: unable to handle kernel paging request at ffffc90403238ffc`
457    Alert = b'1',
458    /// Critical conditions
459    ///
460    /// Examples:
461    ///
462    /// - Crashe, coredumps
463    /// - `systemd-coredump[25319]: Process 25310 (plugin-container) of user 1000 dumped core`
464    Critical = b'2',
465    /// Error conditions
466    ///
467    /// Examples:
468    ///
469    /// - Not severe error reported
470    /// - `kernel: usb 1-3: 3:1: cannot get freq at ep 0x84, systemd[1]: Failed unmounting /var`
471    /// - `libvirtd[1720]: internal error: Failed to initialize a valid firewall backend`
472    Error = b'3',
473    /// May indicate that an error will occur if action is not taken.
474    ///
475    /// Examples:
476    ///
477    /// - a non-root file system has only 1GB free
478    /// - `org.freedesktop. Notifications[1860]: (process:5999): Gtk-WARNING **: Locale not supported by C library. Using the fallback 'C' locale`
479    Warning = b'4',
480    /// Events that are unusual, but not error conditions.
481    ///
482    /// Examples:
483    ///
484    /// - `systemd[1]: var.mount: Directory /var to mount over is not empty, mounting anyway`
485    /// - `gcr-prompter[4997]: Gtk: GtkDialog mapped without a transient parent. This is discouraged`
486    Notice = b'5',
487    /// Normal operational messages that require no action.
488    ///
489    /// Example: `lvm[585]: 7 logical volume(s) in volume group "archvg" now active`
490    Informational = b'6',
491    /// Information useful to developers for debugging the
492    /// application.
493    ///
494    /// Example: `kdeinit5[1900]: powerdevil: Scheduling inhibition from ":1.14" "firefox" with cookie 13 and reason "screen"`
495    Debug = b'7',
496}

498/// Mappings from tracing [`Level`]s to journald [priorities].
499///
500/// [priorities]: Priority

505    /// Priority mapped to the `WARN` level
506    pub warn: Priority,
507    /// Priority mapped to the `INFO` level
508    pub info: Priority,
509    /// Priority mapped to the `DEBUG` level
510    pub debug: Priority,
511    /// Priority mapped to the `TRACE` level

540fn put_metadata(buf: &mut Vec<u8>, meta: &Metadata, prefix: Option<&str>) {
541    if let Some(prefix) = prefix {
542        write!(buf, "{}", prefix).unwrap();
543    }
544    put_field_wellformed(buf, "TARGET", meta.target().as_bytes());
545    if let Some(file) = meta.file() {
546        if let Some(prefix) = prefix {
547            write!(buf, "{}", prefix).unwrap();
548        }
549        put_field_wellformed(buf, "CODE_FILE", file.as_bytes());
550    }
551    if let Some(line) = meta.line() {
552        if let Some(prefix) = prefix {
553            write!(buf, "{}", prefix).unwrap();
554        }
555        // Text format is safe as a line number can't possibly contain anything funny
556        writeln!(buf, "CODE_LINE={}", line).unwrap();
557    }
558}

560/// Append a sanitized and length-encoded field into `buf`.
561///
562/// Unlike `put_field_wellformed` this function handles arbitrary field names and values.
563///
564/// `name` denotes the field name. It gets sanitized before being appended to `buf`.
565///
566/// `write_value` is invoked with `buf` as argument to append the value data to `buf`.  It must
567/// not delete from `buf`, but may append arbitrary data.  This function then determines the length
568/// of the data written and adds it in the appropriate place in `buf`.
569fn put_field_length_encoded(buf: &mut Vec<u8>, name: &str, write_value: impl FnOnce(&mut Vec<u8>)) {
570    sanitize_name(name, buf);
571    buf.push(b'\n');
572    buf.extend_from_slice(&[0; 8]); // Length tag, to be populated
573    let start = buf.len();
574    write_value(buf);
575    let end = buf.len();
576    buf[start - 8..start].copy_from_slice(&((end - start) as u64).to_le_bytes());
577    buf.push(b'\n');
578}
579
580/// Mangle a name into journald-compliant form
581fn sanitize_name(name: &str, buf: &mut Vec<u8>) {
582    buf.extend(
583        name.bytes()
584            .map(|c| if c == b'.' { b'_' } else { c })
585            .skip_while(|&c| c == b'_')
586            .filter(|&c| c == b'_' || char::from(c).is_ascii_alphanumeric())
587            .map(|c| char::from(c).to_ascii_uppercase() as u8),
588    );
589}

591/// Append arbitrary data with a well-formed name and value.
592///
593/// `value` must not contain an internal newline, because this function writes
594/// `value` in the new-line separated format.
595///
596/// For a "newline-safe" variant, see `put_field_length_encoded`.
597fn put_field_wellformed(buf: &mut Vec<u8>, name: &str, value: &[u8]) {
598    buf.extend_from_slice(name.as_bytes());
599    buf.push(b'\n');
600    put_value(buf, value);
601}
602
603/// Write the value portion of a key-value pair, in newline separated format.
604///
605/// `value` must not contain an internal newline.
606///
607/// For a "newline-safe" variant, see `put_field_length_encoded`.
608fn put_value(buf: &mut Vec<u8>, value: &[u8]) {
609    buf.extend_from_slice(&(value.len() as u64).to_le_bytes());
610    buf.extend_from_slice(value);
611    buf.push(b'\n');
612}
