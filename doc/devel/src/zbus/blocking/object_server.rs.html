12/// Wrapper over an interface, along with its corresponding `SignalEmitter`
13/// instance. A reference to the underlying interface may be obtained via
14/// [`InterfaceRef::get`] and [`InterfaceRef::get_mut`].
15pub struct InterfaceRef<I> {
16    azync: crate::object_server::InterfaceRef<I>,
17}
18
19impl<I> InterfaceRef<I>
20where
21    I: 'static,
22{
23    /// Get a reference to the underlying interface.
24    ///
25    /// **WARNING:** If methods (e.g property setters) in `ObjectServer` require `&mut self`
26    /// `ObjectServer` will not be able to access the interface in question until all references
27    /// of this method are dropped; it is highly recommended that the scope of the interface
28    /// returned is restricted.
29    pub fn get(&self) -> InterfaceDeref<'_, I> {
30        block_on(self.azync.get())
31    }
32
33    /// Get a reference to the underlying interface.
34    ///
35    /// **WARNING:** Since the `ObjectServer` will not be able to access the interface in question
36    /// until the return value of this method is dropped, it is highly recommended that the scope
37    /// of the interface returned is restricted.
38    ///
39    /// # Errors
40    ///
41    /// If the interface at this instance's path is not valid, an `Error::InterfaceNotFound` error
42    /// is returned.
43    ///
44    /// # Examples
45    ///
46    /// ```no_run
47    /// # use std::error::Error;
48    /// # use async_io::block_on;
49    /// # use zbus::{blocking::Connection, interface};
50    ///
51    /// struct MyIface(u32);
52    ///
53    /// #[interface(name = "org.myiface.MyIface")]
54    /// impl MyIface {
55    ///    #[zbus(property)]
56    ///    fn count(&self) -> u32 {
57    ///        self.0
58    ///    }
59    /// }
60    /// // Set up connection and object_server etc here and then in another part of the code:
61    /// #
62    /// # let connection = Connection::session()?;
63    /// #
64    /// # let path = "/org/zbus/path";
65    /// # connection.object_server().at(path, MyIface(22))?;
66    /// let object_server = connection.object_server();
67    /// let iface_ref = object_server.interface::<_, MyIface>(path)?;
68    /// let mut iface = iface_ref.get_mut();
69    /// iface.0 = 42;
70    /// block_on(iface.count_changed(iface_ref.signal_emitter()))?;
71    /// #
72    /// # Ok::<_, Box<dyn Error + Send + Sync>>(())
73    /// ```
74    pub fn get_mut(&self) -> InterfaceDerefMut<'_, I> {
75        block_on(self.azync.get_mut())
76    }
77
78    pub fn signal_emitter(&self) -> &SignalEmitter<'static> {
79        self.azync.signal_emitter()
80    }
81}

83/// A blocking wrapper of [`crate::ObjectServer`].
84///
85/// # Example
86///
87/// This example exposes the `org.myiface.Example.Quit` method on the `/org/zbus/path`
88/// path.
89///
90/// ```no_run
91/// # use std::error::Error;
92/// use zbus::{blocking::Connection, interface};
93/// use event_listener::{Event, Listener};
94///
95/// struct Example {
96///     // Interfaces are owned by the ObjectServer. They can have
97///     // `&mut self` methods.
98///     quit_event: Event,
99/// }
100///
101/// impl Example {
102///     fn new(quit_event: Event) -> Self {
103///         Self { quit_event }
104///     }
105/// }
106///
107/// #[interface(name = "org.myiface.Example")]
108/// impl Example {
109///     // This will be the "Quit" D-Bus method.
110///     fn quit(&mut self) {
111///         self.quit_event.notify(1);
112///     }
113///
114///     // See `interface` documentation to learn
115///     // how to expose properties & signals as well.
116/// }
117///
118/// let connection = Connection::session()?;
119///
120/// let quit_event = Event::new();
121/// let quit_listener = quit_event.listen();
122/// let interface = Example::new(quit_event);
123/// connection
124///     .object_server()
125///     .at("/org/zbus/path", interface)?;
126///
127/// quit_listener.wait();
128/// # Ok::<_, Box<dyn Error + Send + Sync>>(())
129/// ```
130#[derive(Debug, Clone)]
131pub struct ObjectServer {
132    azync: crate::ObjectServer,
133}
134
135assert_impl_all!(ObjectServer: Send, Sync, Unpin);
136
137impl ObjectServer {
138    /// Create a new D-Bus `ObjectServer`.
139    pub(crate) fn new(conn: &crate::Connection) -> Self {
140        Self {
141            azync: conn.object_server().clone(),
142        }
143    }
144
145    /// Register a D-Bus [`Interface`] at a given path (see the example above).
146    ///
147    /// Typically you'd want your interfaces to be registered immediately after the associated
148    /// connection is established and therefore use
149    /// [`zbus::blocking::connection::Builder::serve_at`] instead. However, there are
150    /// situations where you'd need to register interfaces dynamically and that's where this
151    /// method becomes useful.
152    ///
153    /// If the interface already exists at this path, returns false.
154    ///
155    /// [`Interface`]: trait.Interface.html
156    pub fn at<'p, P, I>(&self, path: P, iface: I) -> Result<bool>
157    where
158        I: Interface,
159        P: TryInto<ObjectPath<'p>>,
160        P::Error: Into<Error>,
161    {
162        block_on(self.azync.at(path, iface))
163    }
164
165    /// Unregister a D-Bus [`Interface`] at a given path.
166    ///
167    /// If there are no more interfaces left at that path, destroys the object as well.
168    /// Returns whether the object was destroyed.
169    ///
170    /// [`Interface`]: trait.Interface.html
171    pub fn remove<'p, I, P>(&self, path: P) -> Result<bool>
172    where
173        I: Interface,
174        P: TryInto<ObjectPath<'p>>,
175        P::Error: Into<Error>,
176    {
177        block_on(self.azync.remove::<I, P>(path))
178    }
179
180    /// Get the interface at the given path.
181    ///
182    /// # Errors
183    ///
184    /// If the interface is not registered at the given path, an `Error::InterfaceNotFound` error is
185    /// returned.
186    ///
187    /// # Examples
188    ///
189    /// The typical use of this is to emit signals outside of a dispatched handler:
190    ///
191    /// ```no_run
192    /// # use std::error::Error;
193    /// # use zbus::block_on;
194    /// # use zbus::{
195    /// #    object_server::SignalEmitter,
196    /// #    blocking::Connection,
197    /// #    interface,
198    /// # };
199    /// #
200    /// struct MyIface;
201    /// #[interface(name = "org.myiface.MyIface")]
202    /// impl MyIface {
203    ///     #[zbus(signal)]
204    ///     async fn emit_signal(emitter: &SignalEmitter<'_>) -> zbus::Result<()>;
205    /// }
206    ///
207    /// # let connection = Connection::session()?;
208    /// #
209    /// # let path = "/org/zbus/path";
210    /// # connection.object_server().at(path, MyIface)?;
211    /// let iface_ref = connection
212    ///     .object_server()
213    ///     .interface::<_, MyIface>(path)?;
214    /// block_on(MyIface::emit_signal(iface_ref.signal_emitter()))?;
215    /// #
216    /// #
217    /// # Ok::<_, Box<dyn Error + Send + Sync>>(())
218    /// ```
219    pub fn interface<'p, P, I>(&self, path: P) -> Result<InterfaceRef<I>>
220    where
221        I: Interface,
222        P: TryInto<ObjectPath<'p>>,
223        P::Error: Into<Error>,
224    {
225        Ok(InterfaceRef {
226            azync: block_on(self.azync.interface(path))?,
227        })
228    }
229
230    /// Get a reference to the underlying async ObjectServer.
231    pub fn inner(&self) -> &crate::ObjectServer {
232        &self.azync
233    }
234
235    /// Get the underlying async ObjectServer, consuming `self`.
236    pub fn into_inner(self) -> crate::ObjectServer {
237        self.azync
238    }
239}
240
241impl From<crate::ObjectServer> for ObjectServer {
242    fn from(azync: crate::ObjectServer) -> Self {
243        Self { azync }
244    }
