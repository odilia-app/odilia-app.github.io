6/// The message field code.
7///
8/// Every [`Field`] has an associated code. This is mostly an internal D-Bus protocol detail
9/// that you would not need to ever care about when using the high-level API. When using the
10/// low-level API, this is how you can [retrieve a specific field] from [`Fields`].
11///
12/// [`Field`]: enum.Field.html
13/// [retrieve a specific field]: struct.Fields.html#method.get_field
14/// [`Fields`]: struct.Fields.html
15#[repr(u8)]
16#[derive(Copy, Clone, Debug, Deserialize_repr, PartialEq, Eq, Serialize_repr, Type)]
17pub(crate) enum FieldCode {
18    /// Code for [`Field::Path`](enum.Field.html#variant.Path).
19    Path = 1,
20    /// Code for [`Field::Interface`](enum.Field.html#variant.Interface).
21    Interface = 2,
22    /// Code for [`Field::Member`](enum.Field.html#variant.Member).
23    Member = 3,
24    /// Code for [`Field::ErrorName`](enum.Field.html#variant.ErrorName).
25    ErrorName = 4,
26    /// Code for [`Field::ReplySerial`](enum.Field.html#variant.ReplySerial).
27    ReplySerial = 5,
28    /// Code for [`Field::Destination`](enum.Field.html#variant.Destination).
29    Destination = 6,
30    /// Code for [`Field::Sender`](enum.Field.html#variant.Sender).
31    Sender = 7,
32    /// Code for [`Field::Signature`](enum.Field.html#variant.Signature).
33    Signature = 8,
34    /// Code for [`Field::UnixFDs`](enum.Field.html#variant.UnixFDs).
35    UnixFDs = 9,
36}
37
38assert_impl_all!(FieldCode: Send, Sync, Unpin);
